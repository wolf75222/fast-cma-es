143 def fitness(self, x):
144 if len(x) != len(self.get_bounds()[0]):
145 raise ValueError(
146 "Expected "

152 rvt_outs, rvt_ins, _, reso_dts, dvs = self._compute_dvs(x)
153 # compute final flyby and resulting trajectory
154 rvt_out = rvt_outs[-1].rotate(self._rotation_axis, self._theta) # rotate
155 _, _, incl, _, _, _ = rvt_out.kepler()
156 # orbit should be as polar as possible, but we do not care about prograde/retrograde
157 corrected_inclination = abs(abs(incl) % pi - pi / 2) * RAD2DEG

159 # check perihelion and aphelion bounds during the flight
160 emp_perhelion = 2 * AU
161 min_sun_distance = 2 * AU
162 max_sun_distance = 0
163
164 for i in range(len(rvt_outs)):
165 orb = rvt_outs[i]
166 tof = orb.tof(rvt_ins[i + 1])
167 transfer_a, transfer_e, _, _, _, _ = orb.kepler()
168 transfer_period = 2 * pi * sqrt(transfer_a ** 3 / orb.mu)
169 perhelion = transfer_a * (1 - transfer_e)
170 # update min and max sun distance
171 if i >= len(rvt_outs) - 3:
172 emp_perhelion = min(emp_perhelion, perhelion)
173 min_sun_distance = min(min_sun_distance, perhelion)
174 if tof > transfer_period:
175 max_sun_distance = max(max_sun_distance, transfer_a * (1 + transfer_e))
176
177 # overall time limit
178 time_all = SEC2DAY * (rvt_ins[-1].t - rvt_outs[0].t)
179 time_val = time_all
180 time_limit = self._max_mission_time # 11 years
181 if time_val > time_limit:
182 time_val += 10 * (time_val - time_limit)
183
184 distance_val = min_sun_distance / AU
185 # avoid bonus for perihelion < 0.28
186 if distance_val < 0.28:
187 distance_val += 10 * (0.28 - distance_val);
188
189 # wrong reso timing in seconds
190 reso_penalty = np.sum(reso_dts)
191
192 # wrong minimal / maximal distance
193 distance_penalty = max(0, 0.28 - min_sun_distance / AU)
194 distance_penalty += max(0, max_sun_distance / AU - 1.2)
195
196 # allow start dv
197 dvs[0] = max(0, dvs[0] - self._max_dv0)
198 dv_val = np.sum(dvs)
199
200 value = (100 * dv_val +
201 100 * corrected_inclination +
202 5000 * (max(0, distance_val - 0.28)) +
203 5000 * (max(0, emp_perhelion / AU - 0.28)) +
204 0.5 * time_val +
205 reso_penalty +
206 50000 * distance_penalty
207 )
208
209 if value < bval.value:
210 bval.value = value
211 print(str(value)
212 +" " + str(incl * RAD2DEG)
213 +" " + str(time_all / 365.25)
214 +" " + str(dv_val)
215 +" " + str(min_sun_distance / AU)
216 +" " + str(max_sun_distance / AU)
217 +" " + str(reso_penalty) + " "
218 +" " + str(distance_penalty)
219 )
220
221 return ([value])

487# propagate rvt_outs, rvt_ins, rvt_pls, dvs using MGA / Lambert
488def _dv_mga(pl1, pl2, tof, max_revs, rvt_outs, rvt_ins, rvt_pls, dvs, lps=None):
489 rvt_pl = rvt_pls[-1] # current planet
490 v_in = rvt_pl.v if rvt_ins[-1] is None else rvt_ins[-1].v
491 rvt_pl2 = rvt_planet(pl2, rvt_pl.t + tof)
492 rvt_pls.append(rvt_pl2)
493 r = rvt_pl.r
494 vpl = rvt_pl.v
495 r2 = rvt_pl2.r
496 lp = lambert_problem(r, r2, tof, rvt_pl.mu, False, max_revs)
497 lp = lambert_problem_multirev_ga(v_in, lp, pl1, vpl)
498 if not lps is None:
499 lps.append(lp)
500 v_out = lp.get_v1()[0]
501 rvt_out = rvt(r, v_out, rvt_pl.t, rvt_pl.mu)
502 rvt_outs.append(rvt_out)
503 rvt_in = rvt(r2, lp.get_v2()[0], rvt_pl.t + tof, rvt_pl.mu)
504 rvt_ins.append(rvt_in)
505 vr_in = [a - b for a, b in zip(vpl, v_in)]
506 vr_out = [a - b for a, b in zip(v_out, vpl)]
507 dv = fb_vel(vr_in, vr_out, pl1)
508 dvs.append(dv)

511# propagate rvt_outs, rvt_ins, rvt_pls, dvs using resonances
512def compute_resonance(pl, resonances, beta, safe_distance, used_resos, reso_dts, rvt_outs,
513 rvt_ins, rvt_pls, dvs, resos=None):
514 rvt_in = rvt_ins[-1] # current spaceship
515 rvt_pl = rvt_pls[-1] # current planet
516 reso = resonance(pl, rvt_in, rvt_pl, resonances)
517 reso_dt, used_reso = reso.select_resonance(beta, safe_distance)
518 if not resos is None:
519 resos.append(reso)
520 used_resos.append(used_reso)
521 reso_dts.append(reso_dt)
522 rvt_outs.append(reso.rvt_out)
523 tof = reso.tof()
524 time2 = reso.time + tof
525 rvt_pl2 = rvt_planet(pl, time2)
526 rvt_pls.append(rvt_pl2)
527 rvt_in2 = rvt(rvt_pl2.r, reso.rvt_out.v, time2, rvt_pl2.mu) # its a resonance, we arrive with same v as we started
528 rvt_ins.append(rvt_in2)
529 dvs.append(0) # # its a resonance, we don't need an impulse
