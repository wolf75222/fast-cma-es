46def read_solutions(fname):
47 ys = []

49 with open(fname) as csvfile:
50 lines = csvfile.readlines()
51 for line in lines:
52 row = line.split(' ')
53 if len(row) < 12:
54 continue
55 ys.append(float(row[0]))
56 x = []
57 i = -1

63 xi = xi[1:]
64 x.insert(0, float(xi))
65 break
66 x.insert(0, float(xi))
67 i -= 1
68 xs.append(x)

152def archipelago():
153 import pygmo as pg
154 udp = solo_mgar_udp([7000, 8000])
155 #uda = pg.sga(gen = 6000)
156 uda = pg.sade(memory=True,variant=1,gen=6000)
157 # instantiate an unconnected archipelago
158 for _ in range(1000):
159 archi = pg.archipelago(t = pg.topologies.unconnected())
160 for _ in range(32):
161 alg = pg.algorithm(uda)
162 #alg.set_verbosity(1)
163 prob = pg.problem(udp)
164 pop = pg.population(prob, 20)
165 isl = pg.island(algo=alg, pop=pop)
166 archi.push_back(isl)
167 archi.evolve()
168 archi.wait_check()
