30double sdev(vec v) {
31 return sqrt((v.array() - v.mean()).square().sum() / (v.size() - 1));
32}

34class ADAM {
35
36public:
37
38 ADAM(const vec &x0, double b1_, double b2_, double eps_,
39 double center_learning_rate_, double decay_coef_) {
40 dim = x0.size();
41 x = x0;
42 m = zeros(dim);
43 v = zeros(dim);
44 b1 = b1_;
45 b2 = b2_;
46 eps = constant(dim, eps_);
47 center_lr = center_learning_rate_;
48 decay_coef = decay_coef_;
49 }
50
51 void update(int i, const vec &g) {
52
53 m = (1 - b1) * g + b1 * m; // First moment estimate.
54 v = (1 - b2) * g.array().square().matrix() + b2 * v; // Second moment estimate.
55 vec mhat = m.array()
56 * constant(dim, 1. / (1. - pow(b1, i + 1))).array(); // Bias correction.
57 vec vhat = v.array()
58 * constant(dim, 1. / (1. - pow(b2, i + 1))).array();
59 vec delta = step_size(i)
60 * (mhat.array()
61 * (vhat.cwiseSqrt() + eps).cwiseInverse().array());
62 x -= delta;
63// cout << i << " " << step_size(i) << " "
64// << x.mean() << " " << sdev(x) << " "
65// << m.mean() << " " << sdev(m) << " "
66// << v.mean() << " " << sdev(v) << endl;
67 }
68
69 double step_size(int i) {
70 return center_lr * pow(decay_coef, i);
71 }
72
73 vec x;
74 vec m; // First moment estimate
75 vec v; // Second moment estimate.
76
77private:
78 int dim;
79 double b1;
80 double b2;
81 vec eps;
82 double center_lr;
83 double decay_coef;
84};

90 PGPEOptimizer(long runid_, Fitness *fitfun_, int dim_, int seed_,

93 bool use_ranking_,
94
95 double center_learning_rate_, double stdev_learning_rate_,
96 double stdev_max_change_,
97
98 double b1_, double b2_, double eps_, double decay_coef_) {
99 // runid used to identify a specific run
100 runid = runid_;
101 // fitness function to minimize
102 fitfun = fitfun_;
103 // Number of objective variables/problem dimension
104 dim = dim_;
105 // Population size
106 popsize = popsize_ > 0 ? popsize_ : 4 * dim;
107 // maximal number of evaluations allowed.
108 maxEvaluations = maxEvaluations_ > 0 ? maxEvaluations_ : 50000;
109 // Limit for fitness value.
110 stopfitness = stopfitness_;
111 // Number of iterations already performed.
112 iterations = 0;
113 bestY = DBL_MAX;
114 // stop criteria
115 stop = 0;
116 rs = new pcg64(seed_);
117 optimizer = new ADAM(guess_, b1_, b2_, eps_, center_learning_rate_,
118 decay_coef_);
119 center = fitfun->encode(guess_);
120 stdev = inputSigma_;
121 lr_decay_steps = lr_decay_steps_;
122
123 use_ranking = use_ranking_;
124 center_learning_rate_ = abs(center_learning_rate_);
125 stdev_learning_rate = abs(stdev_learning_rate_);
126 stdev_max_change = abs(stdev_max_change_);
127 init();
128 }

135 vec process_scores(const vec &ys) {
136 //Convert fitness scores to rank if necessary.
137 if (use_ranking) {
138 vec ranks(ys.size());
139 ivec yi = sort_index(ys);
140 for (int i = 0; i < ys.size(); i++)
141 ranks[yi[i]] = ((double) i) / ys.size() - 0.5;
142 return ranks;
143 } else
144 return ys;
145 }
146
147 void compute_reinforce_update(const vec &fitness_scores,
148 const mat &scaled_noises, const vec &stdev) {
149 //Compute the updates for the center and the standard deviation.
150 int n = fitness_scores.size() / 2;
151 mat fit = fitness_scores.reshaped(2, n);
152 vec fit1 = fit.row(0); // fitness for baseline + noise
153 vec fit2 = fit.row(1); // fitness for baseline - noise
154 mat baseline = constant(scaled_noises.rows(), scaled_noises.cols(),
155 fitness_scores.mean());
156 vec all_scores = fit1 - fit2;
157 vec all_avg_scores = 0.5 * (fit1 + fit2);
158 mat stdev_sq = stdev.array().square().matrix();
159 mat total_mu =
160 scaled_noises.array()
161 * all_scores.replicate(1, scaled_noises.rows()).transpose().array()
162 * 0.5;
163 mat total_sigma =
164 (all_avg_scores.replicate(1, scaled_noises.rows()).transpose()
165 - baseline).array()
166 * (scaled_noises.array().square().matrix()
167 - stdev_sq.replicate(1, scaled_noises.cols())).array()
168 * stdev.replicate(1, scaled_noises.cols()).array().inverse();
169 grad_center = total_mu.rowwise().mean();
170 grad_stdev = total_sigma.rowwise().mean();
171 }
172
173 vec update_stdev(const vec &stdev, double lr, const vec &grad,
174 double max_change) {
175 vec allowed_delta = stdev.cwiseAbs() * max_change;
176 vec min_allowed = stdev - allowed_delta;
177 vec max_allowed = stdev + allowed_delta;
178 return (stdev + lr * grad).cwiseMax(min_allowed).cwiseMin(max_allowed);
179 }
180
181 mat ask(const vec &stdev, const vec &center) { // undecoded
182 int n = popsize / 2;
183 scaled_noises = normal(dim, n, *rs).array()
184 * stdev.replicate(1, n).array();
185 mat x = mat(dim, popsize);
186 mat x1 = center.replicate(1, n) + scaled_noises;
187 mat x2 = center.replicate(1, n) - scaled_noises;
188 for (int p = 0; p < n; p++) {
189 x.col(2 * p) = x1.col(p);
190 x.col(2 * p + 1) = x2.col(p);
191 }
192 return x;
193 }
194
195 mat ask() { // undecoded
196 return ask(stdev, center);
197 }
198
199 void init() {
200 popX = mat(dim, popsize);
201 popY = vec(popsize);
202 for (int p = 0; p < popsize; p++) {
203 popY[p] = DBL_MAX; // compute fitness
204 }
205 }
206
207 mat ask_decode() {
208 // generate popsize offspring.
209 mat xs = ask(stdev, center);
210 for (int p = 0; p < popsize; p++)
211 popX.col(p) = fitfun->decode(
212 fitfun->getClosestFeasibleNormed(xs.col(p)));
213 return popX;
214 }
215
216 int tell(vec ys) {
217 popY = process_scores(-ys); // negate values since we minimize
218 double bY = -popY.maxCoeff();
219 if (bestY > bY) {
220 bestY = bY;
221 for (int p = 0; p < popsize; p++) {
222 if (popY[p] == -bestY) {
223 bestX = popX.col(p);
224 break;
225 }
226 }
227// cout << popsize * iterations << ": " << bestY << " "
228// << bestX.transpose() << endl;
229 if (bestY < stopfitness)
230 stop = 1;
231 }
232 compute_reinforce_update(popY, scaled_noises, stdev);
233 optimizer->update(((int) iterations) / lr_decay_steps, -grad_center);
234 iterations += 1;
235 center = optimizer->x;
236 stdev = update_stdev(stdev, stdev_learning_rate, grad_stdev,
237 stdev_max_change);
238 return stop;
239 }

242 return popX;

306void optimizePGPE_C(int64_t runid, callback_parallel func_par, int dim,
307 double *init, double *lower, double *upper, double *sigma, int maxEvals,
308 double stopfitness, int popsize, int64_t seed, int lr_decay_steps,
309 bool use_ranking, double center_learning_rate,
310 double stdev_learning_rate, double stdev_max_change, double b1,
311 double b2, double eps, double decay_coef, bool normalize, double *res) {
312
313 vec guess(dim), lower_limit(dim), upper_limit(dim), inputSigma(dim);
314 for (int i = 0; i < dim; i++) {// guess is mandatory
315 guess[i] = init[i];
316 inputSigma[i] = sigma[i];
317 }
318 if (lower != NULL && upper != NULL) {
319 for (int i = 0; i < dim; i++) {
320 guess[i] = init[i];
321 lower_limit[i] = lower[i];
322 upper_limit[i] = upper[i];
323 }
324 } else {
325 lower_limit.resize(0);
326 upper_limit.resize(0);
327 normalize = false;
328 }
329
330 Fitness fitfun(noop_callback, func_par, dim, 1, lower_limit, upper_limit);
331 fitfun.setNormalize(normalize);
332
333 PGPEOptimizer opt(runid, &fitfun, dim, seed, popsize, guess, inputSigma,
334 maxEvals, stopfitness, lr_decay_steps, use_ranking,
335 center_learning_rate, stdev_learning_rate, stdev_max_change, b1, b2,
336 eps, decay_coef);
337 try {
338 while(fitfun.evaluations() < maxEvals
339 && !fitfun.terminate() && opt.getStop() == 0) {
340 mat xs = opt.ask();
341 vec ys(popsize);
342 fitfun.values(xs, ys);
343 opt.tell(ys);
344 }
345 } catch (std::exception &e) {
346 cout << e.what() << endl;
347 }
348 vec bestX = opt.getBestX();
349 double bestY = opt.getBestValue();
350 for (int i = 0; i < dim; i++)
351 res[i] = bestX[i];
352 res[dim] = bestY;
353 res[dim + 1] = fitfun.evaluations();
354 res[dim + 2] = opt.getIterations();
355 res[dim + 3] = opt.getStop();
356}

361 double stdev_learning_rate, double stdev_max_change, double b1,
362 double b2, double eps, double decay_coef, bool normalize) {
363
364 vec guess(dim), lower_limit(dim), upper_limit(dim), inputSigma(dim);
365 for (int i = 0; i < dim; i++) {// guess is mandatory
366 guess[i] = init[i];
367 inputSigma[i] = sigma[i];
368 }
369 if (lower != NULL && upper != NULL) {
370 for (int i = 0; i < dim; i++) {
371 guess[i] = init[i];
372 lower_limit[i] = lower[i];
373 upper_limit[i] = upper[i];
