53 bounds: Optional[Bounds] = None,

60 """
61 Minimizes a given function using Differential Annealing (DA) algorithm with optional
62 local search. This function is a Python interface to an underlying C implementation.
63
64 Args:
65 fun: The objective function to be minimized. It should accept a 1-D array-like
66 object as input and return a float.
67 bounds: Optional bounds for the variables as an instance of `scipy.optimize.Bounds`.
68 This defines the lower and upper bounds of the search space.
69 x0: Optional initial guess for the solution as a 1-D array-like object.
70 If not provided, it will be generated randomly within the bounds.
71 max_evaluations: Maximum number of function evaluations allowed. Default is 100000.
72 use_local_search: Whether to perform local search after the main optimization
73 (True) or not (False). Default is True.
74 rg: Random number generator instance for reproducibility. Defaults to
75 `numpy.random.Generator(PCG64DXSM())`.
76 runid: Optional identifier for the optimization run. Defaults to 0.
77
78 Returns:
79 OptimizeResult: The optimization result represented as a `scipy.optimize.OptimizeResult` object.
80 This object includes the found solution, function value at the solution, number of
81 function evaluations (nfev), number of iterations (nit), the status of the optimization,
82 and a success flag.
83
84 Raises:
85 Exception: If an unexpected error occurs during the optimization process.
86 """
87
88 lower, upper, guess = _check_bounds(bounds, x0, rg)
89 dim = guess.size
90 array_type = ct.c_double * dim
91 c_callback = call_back_type(callback(fun))
92 seed = int(rg.uniform(0, 2**32 - 1))
93 res = np.empty(dim+4)
94 res_p = res.ctypes.data_as(ct.POINTER(ct.c_double))
95 try:
96 optimizeDA_C(runid, c_callback, dim, seed,
97 array_type(*guess),
98 None if lower is None else array_type(*lower),
99 None if upper is None else array_type(*upper),
100 max_evaluations, use_local_search, res_p)
101 x = res[:dim]
102 val = res[dim]
103 evals = int(res[dim+1])
104 iterations = int(res[dim+2])
105 stop = int(res[dim+3])
106 return OptimizeResult(x=x, fun=val, nfev=evals, nit=iterations, status=stop, success=True)
107 except Exception as ex:
108 return OptimizeResult(x=None, fun=sys.float_info.max, nfev=0, nit=0, status=-1, success=False)
