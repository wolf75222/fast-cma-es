8# The F-8 aircraft control problem is based on a very simple aircraft model.
9# The control problem was introduced by Kaya and Noakes and aims at controlling
10# an aircraft in a time-optimal way from an initial state to a terminal state.
11
12# Uses compiled differential equations based on the Ascent library
13# https://github.com/AnyarInc/Ascent see
14# https://github.com/dietmarwo/fast-cma-es/blob/master/_fcmaescpp/ascent.cpp
15
16# Compare with the F8 results
17# given in http://www.midaco-solver.com/data/pub/The_Oracle_Penalty_Method.pdf
18
19# Tested using https://docs.conda.io/en/main/miniconda.html on Linux Mint 21.2

41def f8(t, y, w):
42 return [
43 -0.877*y[0] + y[2] - 0.088*y[0]*y[2] + 0.47*y[0]**2 - 0.019*y[1]**2 - y[0]**2*y[2] + 3.846*y[0]**3
44 +0.215*ksi - 0.28*y[0]**2*ksi + 0.47*y[0]*ksi**2 - 0.63*ksi**3
45 -(0.215*ksi - 0.28*y[0]**2*ksi - 0.63*ksi**3)*2*w,
46 y[2],
47 -4.208*y[0] - 0.396*y[2] - 0.47*y[0]**2 - 3.564*y[0]**3
48 +20.967*ksi - 6.265*y[0]**2*ksi + 46*y[0]*ksi**2 -61.4*ksi**3
49 -(20.967*ksi - 6.265*y[0]**2*ksi - 61.4*ksi**3)*2*w
50 ]

52def bounds(n):
53 lb = [0]*n
54 ub = [2]*n
55 return Bounds(lb,ub)

57def obj_f(X):
58 try:
59 t = 0.
60 y = [0.4655, 0., 0.]
61 I = ode(f8)
62 I.set_integrator("dopri5", nsteps=10**6, rtol=1E-8, atol=1E-8)
63 I.set_initial_value(y, t)
64 n = len(X)
65 for i in range(n):
66 if X[i] == 0:
67 continue
68 # bang-bang type switches starting with w(t) = 1.
69 w = (i + 1) % 2
70 t += X[i]
71 I.set_f_params(w)
72 with warnings.catch_warnings():
73 warnings.simplefilter("ignore")
74 y = I.integrate(t)
75 val0 = np.sum(X)
76 penalty = np.sum(np.abs(y))
77 # estimated fixed weight for penalty
78 return 0.1*val0 + penalty
79 except Exception:
80 return 1E10 # fail

82def obj_f_c(X):
83 try:
84 y = np.asarray([0.4655, 0., 0.])
85 array_type = ct.c_double * y.size
86 n = len(X)
87 for i in range(n):
88 if X[i] == 0:
89 continue
90 # bang-bang type switches starting with w(t) = 1.
91 w = (i + 1) % 2
92 ry = integrateF8_C(array_type(*y), w, X[i], 0.1)
93 y = np.array(np.fromiter(ry, dtype=np.float64, count=y.size))
94 freemem(ry)
95
96 val0 = np.sum(X)
97 penalty = np.sum(np.abs(y))
98 # estimated fixed weight for penalty
99 return 0.1*val0 + penalty
100 except Exception as e:
101 return 1E10 # fail

106 def __init__(self, dim, f8fun):
107 self.name = "f8"
108 self.dim = dim
109 self.f8fun = f8fun
110 self.bounds = bounds(dim)
111 self.evals = mp.RawValue(ct.c_int, 0)
112 self.best_y = mp.RawValue(ct.c_double, np.inf)
113 self.t0 = time.perf_counter()
114
115 def fun(self, x):
116 self.evals.value += 1
117 y = self.f8fun(x)
118 if y < self.best_y.value:
119 self.best_y.value = y
120 logger.info(str(dtime(self.t0)) + ' ' +
121 str(self.evals.value) + ' ' +
122 str(self.best_y.value) + ' ' +
123 str(list(x)))
124 return y
