145@njit(fastmath=True, cache=True)
146def keplerStart3(e, M):
147 t34 = e * e
148 t35 = e * t34
149 t33 = math.cos(M)
150 return M + (-0.5 * t35 + e + (t34 + 1.5 * t33 * t35) * t33) * math.sin(M)

152@njit(fastmath=True, cache=True)
153def eps3(e, M, x):
