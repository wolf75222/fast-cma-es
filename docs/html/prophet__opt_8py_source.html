197 # Calculate the daily change in predicted prices
198 df['yhat_change'] = df['yhat'].diff()

200 # Set a threshold for generating signals

380def optimize_btc_ask_tell():
381 logger.remove()
382 logger.add(sys.stdout, format="{time:HH:mm:ss.SS} | {level} | {message}", level="INFO")
383 logger.add("log_{time}.txt", format="{time:HH:mm:ss.SS} | {level} | {message}", level="INFO")

385 df = load_data()
386 fit = wrapper(partial(fitness, df))
387 bounds = Bounds([0.001, 300, 3, 0], [1, 1600, 11.999, 1.0])
388 dim = len(bounds.lb)
389 batch_size = 16
390 study_id = 0
391
392 # using ask / tell interface + journal
393 journal = Journal("journal2.log", "BTC_strategy", 1)
394 pfit = evaluator.parallel(fit, workers=batch_size)
395 es = de.DE(dim, bounds = bounds, popsize=batch_size)
396
397 iters = 10000
398 trial_id = 1
399
400 for iter in range(iters):
401 journal.write_starts(study_id, batch_size)
402 xs = es.ask()
403 journal.write_xs(trial_id, xs, bounds)
404 ys = pfit(xs)
405 journal.write_ys(trial_id, ys)
406 trial_id += batch_size
407 stop = es.tell(ys)
408 if stop != 0:
409 break
410 x = es.best_x
411 y = es.best_value
412 btc_show_x(df, x)
