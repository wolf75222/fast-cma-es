46def minimize(problems: ArrayLike,
47 ids: Optional[ArrayLike] = None,
48 retries_inc: Optional[int] = min(256, 8*mp.cpu_count()),
49 num_retries: Optional[int] = 10000,
50 keep: Optional[float] = 0.7,
51 optimizer: Optional[Optimizer] = de_cma(1500),
52 datafile = None) -> List:
53
54 """
55 Minimizes the given set of problems by iteratively applying the optimizer and
56 removing the worst-performing solutions until only one remains. Supports
57 retries and optional data file usage for saving and loading intermediate
58 results.
59
60 Minimization of a list of optimization problems by first applying parallel retry
61 to filter the best ones and then applying coordinated retry to evaluate these further.
62 Can replace mixed integer optimization if the integer variables are narrowly bound.
63 In this case all combinations of these integer values can be enumerated to generate a
64 list of problem instances each representing one combination. See for instance
65 https://www.esa.int/gsp/ACT/projects/gtop/tandem where there is a problem instance for each
66 planet sequence.
67
68 Args:
69 problems (ArrayLike): List or array-like structure containing the problems
70 to be minimized.
71 ids (Optional[ArrayLike]): List or array-like structure containing
72 IDs for the problems. Defaults to None, in which case IDs will
73 be auto-generated as strings in ascending order starting from 1.
74 retries_inc (Optional[int]): Number of initial retries increment. Defaults
75 to min(256, 8 times the number of CPU cores).
76 num_retries (Optional[int]): Maximum number of retries for each problem.
77 Defaults to 10000.
78 keep (Optional[float]): Fraction of best-performing problems to retain in
79 each iteration. Defaults to 0.7.
80 optimizer (Optional[Optimizer]): Optimizer object used to minimize the
81 problems. Defaults to de_cma(1500).
82 datafile: Optional parameter representing the file path to load and save
83 intermediate results. Defaults to None.
84
85 Returns:
86 List: Sorted list of all problem statistics after successful
87 minimization.
88 """
89
90 solver = multiretry()
91 n = len(problems)
92
93 for i in range(n):
94 id = str(i+1) if ids is None else ids[i]
95 solver.add(problem_stats(problems[i], id, i, retries_inc, num_retries))
96
97 if not datafile is None:
98 solver.load(datafile)
99
100 while solver.size() > 1:
101 solver.retry(optimizer)
102 to_remove = int(round((1.0 - keep) * solver.size()))
103 if to_remove == 0 and keep < 1.0:
104 to_remove = 1
105 solver.remove_worst(to_remove)
106 solver.dump()
107 if not datafile is None:
108 solver.save(datafile)
109
110 idx = solver.values_all().argsort()
111 return list(np.asarray(solver.all_stats)[idx])

113class problem_stats:
114 """Represents statistics and operational parameters for a specific problem.
115
116 This class encapsulates the problem's parameters, optimization store, retry mechanism,
117 and manages retries for optimization processes. It is designed to handle optimization
118 problems with adjustable retry increments and constraints.
119
120 Attributes:
121 prob: The problem instance containing the problem's functions and bounds.
122 name (str): The name of the problem.
123 fun: The function representing the objective of the problem.
124 retries_inc (int): The number of retries incrementally added on each retry operation.
125 value (float): The best value obtained during optimization.
126 id: The identifier for the problem instance.
127 index: The index associated with the problem instance.
128 """
129 def __init__(self, prob, id, index, retries_inc = 64, num_retries = 10000):
130 """
131 Initializes an instance of the class with the given problem, identifier, index,
132 optional retries increment value, and number of retries.
133
134 Args:
135 prob: An object representing the problem to solve.
136 id: An identifier associated with the instance.
137 index: An integer specifying the index of the instance.
138 retries_inc: Optional integer specifying the increment for retries (default 64).
139 num_retries: Optional integer specifying the total number of retries (default 10000).
140 """
141 self.store = advretry.Store(prob.fun, prob.bounds, num_retries=num_retries)
142 self.prob = prob
143 self.name = prob.name
144 self.fun = prob.fun
145 self.retries_inc = retries_inc
146 self.value = 0
147 self.id = id
148 self.index = index
149 self.ret = None
150 self.store.num_retries = self.retries_inc
151
152 def retry(self, optimizer):
153 """
154 Retries the optimization process and updates the relevant attributes in the store.
155
156 This method increments the retry count stored in the store. It performs an optimization
157 retry using the provided `optimizer` and updates the best obtained value in the
158 optimization process.
159
160 Args:
161 optimizer: An optimization object used to minimize the function. It must provide
162 a `minimize` method to execute the optimization operations.
163 """
164 self.store.num_retries += self.retries_inc
165 self.ret = advretry.retry(self.store, optimizer.minimize)
166 self.value = self.store.get_y_best()

168class multiretry:
169 """
170 A class for managing and retrying task statistics with extended functionalities like removing
171 worst-performing tasks, saving and loading data, and retrieving metrics.
172
173 The class is designed to facilitate the management of task-related statistics, allowing
174 operations such as retrying tasks with an optimizer, sorting tasks based on performance,
175 and persisting data for external use. It also supports evaluating and managing both primary
176 and aggregated statistics.
177
178 Attributes:
179 problem_stats (list): A list of statistics related to problems currently being managed.
180 all_stats (list): A comprehensive list of all statistics including completed and
181 ongoing problems.
182 """
183 def __init__(self):
184 """
185 Represents a container designed to manage problem statistics and aggregate data.
186
187 Attributes:
188 problem_stats (list): A list designed to hold statistics related specifically to problems.
189 all_stats (list): A list intended to store aggregated or general statistics.
190 """
191 self.problem_stats = []
192 self.all_stats = []
193
194 def add(self, stats):
195 """
196 Adds the provided statistics to the problem and all statistics lists.
197
198 Args:
199 stats: The statistics to be added to the lists.
200 """
201 self.problem_stats.append(stats)
202 self.all_stats.append(stats)

204 def retry(self, optimizer):
205 """
206 Retries optimization for each problem in the problem statistics list.
207
208 This method iterates through the list of problem statistics and attempts to
209 retry the optimization process using the given optimizer.
210
211 Args:
212 optimizer: The optimizer instance to use for retrying the problems.
213 """
214 for ps in self.problem_stats:
215 if is_debug_active():
216 logger.debug("problem " + ps.prob.name + ' ' + str(ps.id))
217 ps.retry(optimizer)

229 def remove_worst(self, n = 1):
230 """
231 Removes the worst-performing items from the current problem statistics list.
232
233 This method removes items from the problem statistics list based on their
234 performance metrics. The items are sorted in ascending order first,
235 and the specified number of worst-performing items (based on the input value)
236 are removed from the end of the list.
237
238 Args:
239 n (int, optional): The number of worst-performing items to remove. Defaults
240 to 1.
241
242 """
243 idx = self.values().argsort()
244 self.problem_stats = list(np.asarray(self.problem_stats)[idx])
245 for _ in range(n):
246 self.problem_stats.pop(-1)

312 def result(self):
313 """
314 Sorts statistics based on values and returns a list of optimization results.
315
316 This method processes the data contained in the `all_stats` attribute by sorting
317 it using the `values_all` method. It extracts optimization-related information,
318 such as the best solution, function value, number of function evaluations, and
319 success status, for each statistics entry.
320
321 Returns:
322 List[List]: A list containing sublists, where each sublist consists of a
323 problem instance and its corresponding optimization outcome
324 encapsulated in an `OptimizeResult` object.
325 """
326 idx = self.values_all().argsort()
327 self.all_stats = list(np.asarray(self.all_stats)[idx])
328 ret = []
329 for i in range(len(self.all_stats)):
330 problem = self.all_stats[i].prob
331 store = self.all_stats[i].store
332 ret.append([problem,
333 OptimizeResult(x=store.get_x_best(), fun=store.get_y_best(),
334 nfev=store.get_count_evals(), success=True)])
