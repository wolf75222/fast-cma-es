6# This example uses the "noisy Travelers Salesman Problem" and applies a machine learning
7# approach to avoid unnecessary function calls. Works only with the Python variant of
8# differential evolution, both single threaded or with parallel function evaluation.

32 def __init__(self, size, interval, filter_prob = 0.9):
33 self.xq = deque(maxlen=size)
34 self.yq = deque(maxlen=size)

40 def add(self, x, y):
41 self.xq.append(x)
42 self.yq.append(y)

46 self.learn()

62 def is_improve(self, x, x_old, y_old):
63 if self.model is None or np.random.random() > self.filter_prob :
64 return True
65 else:
66 try:
67 y = self.model.predict([x, x_old])
68 return y[0] < y[1]

70 print(ex)
