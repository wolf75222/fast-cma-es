35@njit(fastmath=True)
36def _get_observed(n_points, im_ft, chromosome):
37 r, c = im_ft.shape
38 l = 0.01
39 x, y = chromosome[:n_points], chromosome[n_points:]
40
41 lx = (np.expand_dims(x, -1) - x).ravel()
42 ly = (np.expand_dims(y, -1) - y).ravel()
43
44 theta = np.linspace(0, 2*np.pi, 10000)
45 cos_theta = np.cos(theta)
46 sin_theta = np.sin(theta)
47
48 obs_uv_matrix = np.zeros((r, c), dtype=np.int32)
49
50 for i in range(10000):
51 for j in range(len(lx)):
52 full_re_u = math.floor((lx[j] * cos_theta[i] + ly[j] * sin_theta[i] ) / l)
53 full_re_v = math.floor((lx[j] * -sin_theta[i] + ly[j] * cos_theta[i] ) / l)
54 x = math.floor(full_re_u * r / 2**2.5 * l)
55 y = math.floor(full_re_v * r / 2**2.5 * l)
56 obs_uv_matrix[x][y] = 1
57
58 return im_ft * obs_uv_matrix

65class Interferometry():
66 def __init__(self, number_points, image, image_size):
67 self.number_points = number_points

70 #We make sure that it is a power of 2

78 self.im_numpy = np.asarray(img)

81 def get_bounds(self):
82 return ([-1.0]*self.number_points*2, [1.0]*self.number_points*2)

84 def fitness(self, x):
85 observed = _get_observed(self.number_points, self.im_fft, x)
86 im_reconstruct = fft.ifft2(observed).real
87 val = (mean_squared_error(self.im_numpy, im_reconstruct),)
88 with monitor:
89 count.value += 1
90 if val[0] < best.value:
91 best.value = val[0]
92 print(str(count.value) + ' fval = ' + str(val[0]) +
93 " t = " + str(round(1000*(time() - t0))) + " ms" + " x = " + ", ".join(str(xi) for xi in x))
94 return val
