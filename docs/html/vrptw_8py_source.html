28def parse_problem(filename):
29 with open(filename) as csvfile:
30 lines = csvfile.readlines()
31 demand = []
32 coord = []
33 ready = []
34 due = []
35 service = []
36 for line in lines:
37 row = line.split()
38 if len(row) == 2 and row[0][0].isdigit():
39 number = int(row[0])
40 capacity = int(row[1])
41 if len(row) < 5 or not row[0][0].isdigit():
42 continue
43 coord.append(np.array([float(row[1]), float(row[2])]))
44 demand.append(float(row[3]))
45 ready.append(float(row[4]))
46 due.append(float(row[5]))
47 service.append(float(row[6]))
48 n = len(demand)
49 dtimes = np.zeros((n,n))
50 for i in range(n):
51 for j in range(n):
52 dtimes[i,j] = np.linalg.norm(coord[i] - coord[j])
53 return number, capacity, dtimes, np.array(demand), np.array(ready),\
54 np.array(due), np.array(service)

56@njit(fastmath=True)
57def fitness_(seq, capacity, dtime, demands, readys, dues, services):
58 n = len(seq)
59 seq += 1
60 sum_demand = 0
61 sum_dtime = 0
62 time = 0
63 last = 0
64 vehicles = 1
65 for i in range(0, n+1):
66 customer = seq[i] if i < n else 0
67 demand = demands[customer]
68 ready = readys[customer]
69 due = dues[customer]
70 service = services[customer]
71 if sum_demand + demand > capacity or \
72 time + dtime[last, customer] > due:
73 # end vehicle tour, return to base
74 dt = dtime[last, 0]
75 sum_dtime += dt
76 time = 0
77 sum_demand = 0
78 vehicles += 1
79 last = 0
80 # go to customer
81 dt = dtime[last, customer]
82 time += dt
83 if time < ready:
84 time = ready
85 time += service
86 sum_demand += demand
87 sum_dtime += dt
88 last = customer
89 return np.array([float(vehicles), sum_dtime])

91class VRPTW():
92 def __init__(self, problem):
93 self.problem = problem
94 filename = 'problems/' + problem + '.txt'
95 self.vnumber, self.capacity, self.dtime, self.demand, self.ready,\
96 self.due, self.service = parse_problem(filename)
97 self.dim = len(self.demand) - 1
98 self.bounds = Bounds([0]*self.dim, [1]*self.dim)

100 def fitness(self, x):
101 fit = fitness_(np.argsort(x), self.capacity, self.dtime, self.demand, \
102 self.ready, self.due, self.service)
103 return 10*fit[0] + fit[1]

150def optimize(vrptw, opt, num_retries = 64):
151 ret = retry.minimize(wrapper(vrptw.fitness),
152 vrptw.bounds, num_retries = num_retries, optimizer=opt)
153 vrptw.dump(np.argsort(ret.x), ret.fun, vrptw.problem, opt.name.replace(' ','_') + '_')
