109class lambert_problem_stochastic:
110 r"""
111 This class converts a lambert_problem instance - a number of solutions to a multi revolution Lambert problem -
112 to an instance representing a single "good" solution. Criteria is the delta velocity of the incoming
113 velocity v_in to the outgoing solution velocity get_v1(). Can be used as a replacement for lambert_problem
114 in _mga.py and _mga_1dsm.py. The used criteria is not the only one relevant for optimization problems aiming at
115 reducing the total delta velocity. I doesn't consider the incoming solution velocity get_v2() because
116 the corresponding outgoing target velocity is not available yet. By randomly choosing a "good" solution
117 the value of the objective function calling lambert_problem_stochastic is no longer deterministic.
118 Combined with a non derivative optimization algorithm this way search can be replaced by optimization.
119
120 .. note::
121
122 The formula used to randomly skip over superior solutions was tested for _tandem and _messenger and GTOC1.
123 Other problems may require different parameters.
124
125 """
126
127 def __init__(self, v_in, lambert_problem):
128 best_i = 0
129 n = len(lambert_problem.get_v1())
130 if n > 0:
131 best_dv = math.inf
132 for i in range(n):
133 # convert to km/s
134 dv = 0.001 * np.linalg.norm([a - b for a, b in zip(lambert_problem.get_v1()[i], v_in)])
135 if dv < best_dv:
136 # skip improvement randomly if dv is only slightly better
137 if i > 0 and random.random() < 1.0 / (1.0 + 2*(0.001*best_dv - dv)**2):
138 continue
139 best_dv = dv
140 best_i = i
