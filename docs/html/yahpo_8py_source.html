1# Adapted from https://github.com/slds-lmu/qdo_yahpo for application of fcmaes
2#
3# See "A Collection of Quality Diversity Optimization Problems Derived
4# from Hyperparameter Optimization of Machine Learning Models" https://arxiv.org/abs/2204.14061
5#
6# Differences to https://github.com/slds-lmu/qdo_yahpo:
7# - No statistics, just applies the Hyperparameter Optimization and ranger benchmark functions
8# - Parallelism is handled by the optimizer
9# - Configured optimizer is similar to "mixed" in qdo_yahpo, but
10# * Uses CR-FM-NES instead of CMA-ES emitter
11# * CR-FM-NES is initialized with a random solution instead of a niche elite
12# * Uses Voronoi tesselation (see CVT MAP-Elites https://arxiv.org/abs/1610.05729)
13# * Map-Elites is executed in parallel with CR-FM-NES sharing the same QD-archive
14# * Instead of gaussian distribution Map-Elites uses simulated binary crossover + mutation as NSGAII
15#
16# Results cannot be directly compared since different tesselation is used (Grid/Voronoi).
17#
18# Requires yahpo_gym, please follow the installation instuctions at
19# https://github.com/slds-lmu/yahpo_gym/tree/main/yahpo_gym
20
21# Tested using https://docs.conda.io/en/main/miniconda.html on Linux Mint 21.

43def opt_ranger_hardware(task_id, max_evals = 100000):
44 bench = benchmark_set.BenchmarkSet("iaml_ranger", check = False)
45 bench.set_instance(task_id)
46 search_space = bench.get_opt_space()
47
48 params = ["num.trees", "mtry.ratio", "min.node.size", "sample.fraction"]
49 bounds = [(search_space.get_hyperparameter(param).lower, search_space.get_hyperparameter(param).upper) for param in params]
50 lower = [bound[0] for bound in bounds]
51 upper = [bound[1] for bound in bounds]
52
53 if task_id == "41146":
54 qd_bounds = Bounds((1., 0.19), (200., 4.5))
55 elif task_id == "40981":
56 qd_bounds = Bounds((1., 0.10), (40., 0.65))
57 elif task_id == "1489":
58 qd_bounds = Bounds((1., 0.19), (200., 4.5))
59 elif task_id == "1067":
60 qd_bounds = Bounds((1., 0.13), (78., 1.55))
61
62 dim = len(bounds)
63 qd_dim = 2
64 bounds = Bounds([0]*dim, [1]*dim)
65
66 def qd_fun(x): # Only a single fitness is needed. Parallelism is handled by the optimizer.
67 yd = evaluate_ranger_([x], ["rammodel", "timepredict"], params, bench, task_id, lower, upper)
68 return yd[0][0], yd[0][1:]
69
70 niche_num = 4000
71 arch = None
72 opt_params0 = {'solver':'elites', 'popsize':16}
73 opt_params1 = {'solver':'CRMFNES_CPP', 'max_evals':200, 'popsize':16, 'stall_criterion':3}
74
75 archive = diversifier.minimize(
76 mapelites.wrapper(qd_fun, qd_dim, interval=100000),
77 bounds, qd_bounds, opt_params=[opt_params0, opt_params1], max_evals=max_evals, archive = arch,
78 niche_num = niche_num, samples_per_niche = 20)
79
80 name = "range_hard" + task_id
81 print('final archive:', archive.info())
82 archive.save(name)
83 plot(archive, name)

173def evaluate_ranger_(x, features, params, bench, task_id, lower, upper, trafo=True):
174 if trafo:
175 x = np.array([retrafo_(i, lower, upper) for i in x])
176 # YAHPO Gym supports batch predicts if configs are passed as a list of dicts
177 config = [dict(zip(params, x[i])) for i in range(x.shape[0])]
178 # add the other hyperparameters not part of the search space and respect integers
179 for i in range(len(config)):
180 config[i].update({"num.trees":round(config[i]["num.trees"]), "min.node.size":round(config[i]["min.node.size"])})
181 config[i].update({"task_id":task_id, "trainsize":1, "replace":"TRUE", "respect.unordered.factors":"ignore", "splitrule":"gini"})
182 results = bench.objective_function(config)
183 targets = ["mmce"]
184 targets.extend(features)
185 y = pd.DataFrame(results)[targets]
186 y[["mmce"]] = 1 - y[["mmce"]] # pyribs maximizes by default so we turn mmce into acc
187 return y.values # pyribs expects a numpy array as return value

189def evaluate_xgboost_(x, features, params, bench, task_id, lower, upper, trafo=True):
190 if trafo:
191 x = np.array([retrafo_(i, lower, upper) for i in x])
192 # YAHPO Gym supports batch predicts if configs are passed as a list of dicts
193 config = [dict(zip(params, x[i])) for i in range(x.shape[0])]
194 # add the other hyperparameters not part of the search space and respect log trafos and integers
195 for i in range(len(config)):
196 config[i].update({"alpha":np.exp(config[i]["alpha"]), "lambda":np.exp(config[i]["lambda"]), "eta":np.exp(config[i]["eta"]), "gamma":np.exp(config[i]["gamma"]), "min_child_weight":np.exp(config[i]["min_child_weight"])})
197 config[i].update({"nrounds":round(np.exp(config[i]["nrounds"])), "max_depth":round(config[i]["max_depth"])})
198 config[i].update({"task_id":task_id, "trainsize":1, "booster":"gbtree"})
199 results = bench.objective_function(config)
200 targets = ["mmce"]
201 targets.extend(features)
202 y = pd.DataFrame(results)[targets]
203 y[["mmce"]] = 1 - y[["mmce"]] # pyribs maximizes by default so we turn mmce into acc
204 return y.values # pyribs expects a numpy array as return value

206def retrafo_(z, lower, upper):
207 n = len(z)
208 assert(n == len(lower))
209 assert(n == len(upper))
210 return [np.max((lower[i], np.min(((z[i] * (upper[i] - lower[i])) + lower[i], upper[i])))) for i in range(n)]
