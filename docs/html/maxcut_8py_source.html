95 """Returns support level dictionary."""

106 def minimize(
107 self,
108 fun: Callable[[POINT], float],
109 x0: POINT,
110 jac: Optional[Callable[[POINT], POINT]] = None,
111 bounds: Optional[List[Tuple[float, float]]] = None,
112 ) -> OptimizerResult:
113 bnds = self._bounds if bounds is None else Bounds([b[0] for b in bounds], [b[1] for b in bounds])
114 if self._use_wrapper:
115 fun = wrapper(fun) # monitors progress for all parallel processes
116 result = OptimizerResult()
117 if self._max_retries <= 1:
118 x, y, nfev = self._optimizer.minimize(fun, bnds, guess=x0)
119 result.x = x
120 result.fun = y
121 result.nfev = nfev
122 else:
123 ret = retry.minimize(fun, bnds, num_retries=self._max_retries, workers = self._workers,
124 optimizer=self._optimizer)
125 result.x = ret.x
126 result.fun = ret.fun
127 result.nfev = ret.nfev
128 return result

130def maxcut(optimizer, # used quiskit optimizer
131 n = 13, # number of nodes in graph
132 backend_name = "aer_simulator",
133 add_noise=False, # add noise to simulator
134 ):
135 G = nx.dense_gnm_random_graph(n, 2*n, seed=123)
136 for (u, v) in G.edges():
137 G.edges[u,v]['weight'] = 1
138 w = np.zeros([n, n])
139 for i in range(n):
140 for j in range(n):
141 temp = G.get_edge_data(i, j, default=0)
142 if temp != 0:
143 w[i, j] = temp["weight"]
144 print(w)
145 max_cut = Maxcut(w)
146 qp = max_cut.to_quadratic_program()
147 print(qp.prettyprint())
148
149 qubitOp, offset = qp.to_ising()
150 print("Offset:", offset)
151 print("Using Hamiltonian:")
152 print(str(qubitOp))
153
154 ee = NumPyMinimumEigensolver()
155 result = ee.compute_minimum_eigenvalue(qubitOp)
156 x = max_cut.sample_most_likely(result.eigenstate)
157 print("energy:", result.eigenvalue.real)
158 print("max-cut objective:", result.eigenvalue.real + offset)
159 print("solution:", x)
160 print("solution objective:", qp.objective.evaluate(x))
161
162 backend = Aer.get_backend(backend_name)
163
164 if add_noise:
165 device_backend = FakeVigo()
166 device = QasmSimulator.from_backend(device_backend)
167 noise_model = NoiseModel.from_backend(device)
168 print(noise_model)
169 quantum_instance = QuantumInstance(backend, noise_model=noise_model)
170 else:
171 quantum_instance = QuantumInstance(backend)
172
173 backend.set_options(max_parallel_threads=1)
174
175 ry = TwoLocal(qubitOp.num_qubits, "ry", "cz", reps=5, entanglement="linear")
176 vqe = VQE(ry, optimizer=optimizer, quantum_instance=quantum_instance)
177 dim = vqe.ansatz.num_parameters
178 vqe.ansatz.parameter_bounds = [(-2 * np.pi, 2 * np.pi)]*dim
179
180 # run VQE
181 with threadpoolctl.threadpool_limits(limits=1, user_api="blas"):
182 # blas threading restriction, speeds up "aer_simulator_statevector"
183 result = vqe.compute_minimum_eigenvalue(qubitOp)
184
185 # print results
186 x = max_cut.sample_most_likely(result.eigenstate)
187 print("energy:", result.eigenvalue.real)
188 print("time:", result.optimizer_time)
189 print("max-cut objective:", result.eigenvalue.real + offset)
190 print("solution:", x)
191 print("solution objective:", qp.objective.evaluate(x))
