50def minimize(fun: Callable[[ArrayLike], float],
51 bounds: Optional[Bounds] = None,
52 x0: Optional[ArrayLike] = None,
53 input_sigma: Optional[Union[float, ArrayLike, Callable]] = 0.1,
54 popsize: Optional[int] = 32,
55 max_evaluations: Optional[int] = 100000,
56 workers: Optional[int] = None,
57 stop_fitness: Optional[float] = -np.inf,
58 rg: Optional[Generator] = Generator(PCG64DXSM()),
59 runid: Optional[int] = 0,
60 normalize: Optional[bool] = True,
61 lr_decay_steps: Optional[int] = 1000,
62 use_ranking: Optional[bool] = True,
63 center_learning_rate: Optional[float] = 0.15,
64 stdev_learning_rate: Optional[float] = 0.1,
65 stdev_max_change: Optional[float] = 0.2,
66 b1: Optional[float] = 0.9,
67 b2: Optional[float] = 0.999,
68 eps: Optional[float] = 1e-8,
69 decay_coef: Optional[float] = 1.0,
70 ) -> OptimizeResult:

72 """
73 Optimize a given objective function by minimizing its value using the PGPE
74 (Policy Gradient with Parameter-based Exploration) method. This optimization
75 approach uses parallel processing and supports both optional parameter
76 normalization and adaptive learning rates for center and standard deviation.
77
78 Args:
79 fun (Callable[[ArrayLike], float]): Objective function to minimize. Must
80 take a single input of type ArrayLike and return a float.
81 bounds (Optional[Bounds]): Bounds for the variables. Should be specified
82 as a tuple (lower_bounds, upper_bounds) or None for unbounded variables.
83 x0 (Optional[ArrayLike]): Initial guess for the variables. If None,
84 random initialization will be applied within bounds if specified.
85 input_sigma (Optional[Union[float, ArrayLike, Callable]]): Initial
86 standard deviation for parameter sampling. Defaults to 0.1. Can be
87 a scalar, array, or callable returning an array.
88 popsize (Optional[int]): Population size for sampling. Defaults to 32.
89 If not specified or odd, it will be adjusted to the next even number.
90 max_evaluations (Optional[int]): Maximum number of function evaluations
91 to perform. Defaults to 100000.
92 workers (Optional[int]): Number of parallel workers to use for evaluation.
93 Defaults to None (no parallelism).
94 stop_fitness (Optional[float]): Value of fitness to stop early. If the
95 objective reaches this value, the algorithm will terminate early.
96 Defaults to -infinity.
97 rg (Optional[Generator]): Random number generator to use during
98 optimization. Defaults to Generator(PCG64DXSM()).
99 runid (Optional[int]): Unique identifier for the run. Useful for
100 distinguishing runs in logging or debugging.
101 normalize (Optional[bool]): Whether to normalize the input parameters for the
102 optimizer. Defaults to True.
103 lr_decay_steps (Optional[int]): Number of steps for learning rate decay.
104 Defaults to 1000.
105 use_ranking (Optional[bool]): Whether to use ranking over raw fitness
106 when calculating updates. Defaults to True.
107 center_learning_rate (Optional[float]): Learning rate used to update
108 the center. Defaults to 0.15.
109 stdev_learning_rate (Optional[float]): Learning rate used to update the
110 standard deviation. Defaults to 0.1.
111 stdev_max_change (Optional[float]): Maximum allowed change for standard
112 deviation updates. Defaults to 0.2.
113 b1 (Optional[float]): Exponential moving average factor for the first
114 moment estimate during adaptive updates. Defaults to 0.9.
115 b2 (Optional[float]): Exponential moving average factor for the second
116 moment estimate during adaptive updates. Defaults to 0.999.
117 eps (Optional[float]): Small term added to avoid division by zero in
118 adaptive algorithms. Defaults to 1e-8.
119 decay_coef (Optional[float]): Coefficient that controls decay in the
120 learning rate updates. Defaults to 1.0.
121
122 Returns:
123 OptimizeResult: An object containing optimization results, including
124 the best parameters found (`x`), the objective value (`fun`) at
125 those parameters, the number of function evaluations (`nfev`), the
126 number of iterations (`nit`), the stopping status as an integer code
127 (`status`), and a boolean success flag (`success`).
128 """
129
130 lower, upper, guess = _check_bounds(bounds, x0, rg)
131 dim = guess.size
132 if popsize is None:
133 popsize = 32
134 if popsize % 2 == 1: # requires even popsize
135 popsize += 1
136 if lower is None:
137 lower = [0]*dim
138 upper = [0]*dim
139 if callable(input_sigma):
140 input_sigma=input_sigma()
141 if np.ndim(input_sigma) == 0:
142 input_sigma = [input_sigma] * dim
143 parfun = None if (workers is None or workers <= 1) else parallel(fun, workers)
144 array_type = ct.c_double * dim
145 c_callback_par = call_back_par(callback_par(fun, parfun))
146 res = np.empty(dim+4)
147 res_p = res.ctypes.data_as(ct.POINTER(ct.c_double))
148 try:
149 optimizePGPE_C(runid, c_callback_par, dim, array_type(*guess),
150 array_type(*lower), array_type(*upper),
151 array_type(*input_sigma), max_evaluations, stop_fitness,
152 popsize, int(rg.uniform(0, 2**32 - 1)),
153 lr_decay_steps, use_ranking, center_learning_rate,
154 stdev_learning_rate, stdev_max_change, b1, b2, eps, decay_coef,
155 normalize, res_p)
156 x = res[:dim]
157 val = res[dim]
158 evals = int(res[dim+1])
159 iterations = int(res[dim+2])
160 stop = int(res[dim+3])
161 res = OptimizeResult(x=x, fun=val, nfev=evals, nit=iterations, status=stop, success=True)
162 except Exception as ex:
163 res = OptimizeResult(x=None, fun=sys.float_info.max, nfev=0, nit=0, status=-1, success=False)
164 if not parfun is None:
165 parfun.stop()
166 return res

170 Optimization class using a C++ CR-FM-NES implementation via `ctypes`.
171
172 Minimizes a scalar objective function of one or more variables using
173 a population-based evolutionary strategy. This class allows interaction
174 with the underlying C++ implementation to run optimization tasks with
175 high performance and flexibility.
176
177 Attributes:
178 ptr (ctypes.c_void_p): Pointer to the C++ PGPE object.

202 """
203 Initializes an instance of the class with parameters for a PGPE (Policy Gradients
204 with Parameter-based Exploration) optimization algorithm.
205
206 This constructor sets various hyperparameters and configuration options for
207 running the algorithm. These include the problem dimension, bounds, initial
208 guess, population size, learning rates, and other factors that control the
209 optimization process.
210
211 Args:
212 dim (int): Dimensionality of the optimization problem.
213 bounds (Optional[Bounds]): Input bounds defining the feasible region.
214 x0 (Optional[ArrayLike]): Initial guess for the algorithm.
215 input_sigma (Optional[Union[float, ArrayLike, Callable]]): Initial standard
216 deviation for exploration, either as a scalar or array or function returning
217 a value.
218 popsize (Optional[int]): Size of the population. Ensures even number if not
219 already.
220 rg (Optional[Generator]): Random number generator to control randomization in
221 the algorithm. Defaults to a PCG64DXSM generator.
222 runid (Optional[int]): Unique identifier for the optimization run.
223 normalize (Optional[bool]): Whether to normalize the input bounds.
224 lr_decay_steps (Optional[int]): Number of iterations over which the learning
225 rate decays.
226 use_ranking (Optional[bool]): If True, enables ranking-based updates.
227 center_learning_rate (Optional[float]): Learning rate for the mean or center
228 of the distribution.
229 stdev_learning_rate (Optional[float]): Learning rate for the standard
230 deviation of the search distribution.
231 stdev_max_change (Optional[float]): Maximum allowable change for standard
232 deviation in an iteration.
233 b1 (Optional[float]): Exponential decay rate for first moment estimates in
234 adaptive learning.
235 b2 (Optional[float]): Exponential decay rate for second moment estimates in
236 adaptive learning.
237 eps (Optional[float]): Small constant to prevent division by zero in
238 adaptive learning.
239 decay_coef (Optional[float]): Coefficient controlling overall decay in
240 learning adjustments.
241
242 Raises:
243 Exception: Propagates exceptions encountered during the initialization of
244 the algorithm backend.
245 """
246
247 lower, upper, guess = _get_bounds(dim, bounds, x0, rg)
248 if popsize is None:
249 popsize = 32
250 if popsize % 2 == 1: # requires even popsize
251 popsize += 1
252 if lower is None:
253 lower = [0]*dim
254 upper = [0]*dim
255 if callable(input_sigma):
256 input_sigma=input_sigma()
257 if np.ndim(input_sigma) == 0:
258 input_sigma = [input_sigma] * dim
259 array_type = ct.c_double * dim
260 try:
261 self.ptr = initPGPE_C(runid, dim, array_type(*guess),
262 array_type(*lower), array_type(*upper),
263 array_type(*input_sigma), popsize, int(rg.uniform(0, 2**32 - 1)),
264 lr_decay_steps, use_ranking, center_learning_rate,
265 stdev_learning_rate, stdev_max_change, b1, b2, eps, decay_coef,
266 normalize)
267 self.popsize = popsize
268 self.dim = dim
269 except Exception as ex:
270 print (ex)
271 pass

285 def ask(self) -> np.array:
286 """
287 Generates and returns a population of samples based on the current state of the algorithm.
288
289 This method interacts with the C library function `askPGPE_C` to generate a new set of samples
290 for the population. The returned samples are organized into a 2D NumPy array.
291
292 Returns:
293 np.array: A 2D array where each row corresponds to a sample in the population.
294
295 Raises:
296 Exception: If an error occurs during the population generation process with specific details
297 printed to the console.
298 """
299
300 try:
301 lamb = self.popsize
302 n = self.dim
303 res = np.empty(lamb*n)
304 res_p = res.ctypes.data_as(ct.POINTER(ct.c_double))
305 askPGPE_C(self.ptr, res_p)
306 xs = np.empty((lamb, n))
307 for p in range(lamb):
308 xs[p,:] = res[p*n : (p+1)*n]
309 return xs
310 except Exception as ex:
311 print (ex)
312 return None

365 """
366 Fetches the optimization result.
367
368 The method retrieves the solution obtained from the optimization procedure
369 using a C-based backend. The result includes the optimized parameters,
370 objective function value at the solution, the number of function
371 evaluations, the number of iterations, and the exit status of the optimizer.
372
373 In case of any exception during the retrieval process, a default result
374 indicating failure is returned.
375
376 Returns:
377 OptimizeResult: An object containing the optimization result. It includes
378 the following fields:
379 - x: ndarray of the optimized parameters.
380 - fun: float value of the objective function at the solution.
381 - nfev: int count of function evaluations.
382 - nit: int count of iterations performed.
383 - status: int exit status of the optimization.
384 - success: bool indicating the success (True) or failure (False) of the optimization.
385 """

389 resultPGPE_C(self.ptr, res_p)
390 x = res[:self.dim]
391 val = res[self.dim]
392 evals = int(res[self.dim+1])
393 iterations = int(res[self.dim+2])
394 stop = int(res[self.dim+3])
395 res = OptimizeResult(x=x, fun=val, nfev=evals, nit=iterations, status=stop, success=True)
396 except Exception as ex:
397 res = OptimizeResult(x=None, fun=sys.float_info.max, nfev=0, nit=0, status=-1, success=False)
398 return res
