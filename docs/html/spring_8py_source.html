6# This problem consists of minimizing the weight of a tension/compression spring subject to constraints on
7# shear stress, surge frequency and minimum deflection
8# The design variables are:
9
10# - x1: the mean coil diameter
11# - x2: the wire diameter
12# - x3: the number of active coils
13
14# which are subject to four inequality constraints.
15# fcmaes retry used with a penalty for constraint violations 'weight_penalty'
16# works as well as scipy minimize.
17
18# This example is taken from https://www.sciencedirect.com/science/article/abs/pii/S0096300306015098
19
20# Tested using https://docs.conda.io/en/main/miniconda.html on Linux Mint 21.

36bounds = Bounds([0.01, 0.01, 0.01], [20, 20, 20])
37
38def _feasible(x):
39 x = np.array(x)
40 return np.maximum(np.minimum(x, bounds.ub), bounds.lb)
41
42def constraint_ineq(x):
43 return [ x[1]**3 * x[2] / (71785 * x[0]**4) - 1,
44 1 - (4*x[1]**2 - x[0]*x[1]) / (12566*(x[1]*x[0]**3 - x[0]**4)) - 1/(5108*x[0]**2),
45 140.45*x[0]/(x[1]**2 * x[2]) - 1,
46 1 - (x[1] + x[0]) / 1.5]
47
48def penalty(x):
49 return - np.sum(np.minimum(constraint_ineq(x), 0))
50
51def weight(x):
52 return (x[2] + 2) * x[1]*x[0]**2
53
54def weight_penalty(x):
55 try:
56 val = weight(x)
57 return val + 100000*penalty(x)
58 except ZeroDivisionError:
59 return 1E99
60
61def print_result(ret, best, t0, i):
62 x = _feasible(ret.x) # make sure result is _feasible
63 w = weight(x)
64 val = weight_penalty(x) # add penalty for ineq constraint violation
65 if val < best:
66 pen = penalty(x) # check ineq constraint
67 best = min(val, best)
68 print("{0}: time = {1:.1f} best = {2:.8f} f(xmin) = {3:.8f} ineq = {4:.5f}"
69 .format(i+1, dtime(t0), best, w, pen))
70 return best

72def test_minimize_SLSQP(fun, num):
73 ineq_cons = {'type': 'ineq', 'fun' : constraint_ineq}
74
75 best = np.inf
76 t0 = time.perf_counter();

78 guess = random_x(bounds.lb, bounds.ub)
79 ret = minimize(fun, x0 = guess, bounds = bounds,
80 method='SLSQP', constraints=[ineq_cons])
81 if ret.success:
82 best = print_result(ret, best, t0, i)
