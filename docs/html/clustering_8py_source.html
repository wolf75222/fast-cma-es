82@njit(fastmath=True)
83def sum_distance(s):
84 dist = 0
85 g = np.empty((num_groups, group_size), dtype=numba.int32)
86 for i in range(num_groups):
87 g[i] = s[i*group_size:(i+1)*group_size]
88 for i in range(num_groups):
89 for v in range(group_size):
90 for w in range(group_size):
91 dist += distance_matrix[g[i,v],g[i,w]]
