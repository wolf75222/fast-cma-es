15 Description:
16 - Numpy based implementation of an diversifying wrapper / parallel retry mechanism.
17
18 Uses the archive from CVT MAP-Elites [2] to maintain a set of diverse solutions
19 and generalizes ideas from CMA-ME [3]
20 to other wrapped algorithms.
21
22 Both the parallel retry and the archive based modification of the fitness
23 function enhance the diversification of the optimization result.
24 The resulting archive may be stored and can be used to continue the
25 optimization later.
26
27 Requires a QD-fitness function returning both an fitness value and a
28 behavior vector used to determine the corresponding archive niche using
29 Voronoi tesselation.
30
31 Returns an archive of niche-elites containing also for each niche statistics
32 about the associated solutions.

71def minimize(qd_fitness: Callable[[ArrayLike], Tuple[float, np.ndarray]],
72 bounds: Bounds,
73 qd_bounds: Bounds,
74 niche_num: Optional[int] = 10000,
75 samples_per_niche: Optional[int] = 20,
76 max_evals: Optional[int] = None,
77 workers: Optional[int] = mp.cpu_count(),
78 archive: Optional[Archive] = None,
79 opt_params: Optional[Dict] = {},
80 use_stats: Optional[bool] = False,
81 ) -> Archive:
82
83 """
84 Performs parallel optimization to minimize the fitness function with a focus on
85 quality-diversity. The function initializes an archive that stores optimal
86 inputs and their fitness values, organizes the records into defined niches,
87 and carries out optimization using parallel processing.
88
89 Args:
90 qd_fitness: Callable fitness function that takes an input array and returns
91 a tuple containing a scalar objective value and an array of descriptor
92 values.
93 bounds: Bounds object defining the lower and upper bounds of the search
94 space.
95 qd_bounds: Bounds object defining the lower and upper bounds for the
96 quality-diversity archive.
97 niche_num: Optional; The number of niches to partition the archive.
98 Default is 10000.
99 samples_per_niche: Optional; The number of samples to initialize in each
100 niche. Default is 20.
101 max_evals: Optional; The maximum number of function evaluations. If not
102 provided, the default is workers * 50000.
103 workers: Optional; The number of parallel processes to use. Default is the
104 number of CPU cores available.
105 archive: Optional; An existing archive object to initialize the optimization
106 process. If not provided, a new archive is created.
107 opt_params: Optional; A dictionary of parameters used by the optimization
108 function. Default is an empty dictionary.
109 use_stats: Optional; A flag to determine whether to track statistical
110 properties within the archive. Default is False.
111
112 Returns:
113 Archive: An archive object containing the optimized results, including
114 fitness values and descriptors.
115 """
116
117 if max_evals is None:
118 max_evals = workers*50000
119 dim = len(bounds.lb)
120 if archive is None:
121 archive = Archive(dim, qd_bounds, niche_num, use_stats)
122 archive.init_niches(samples_per_niche)
123 # initialize archive with random values
124 archive.xs_view[:] = rng.uniform(bounds.lb, bounds.ub, (niche_num, dim))
125 t0 = perf_counter()
126 qd_fitness.archive = archive # attach archive for logging
127 minimize_parallel_(archive, qd_fitness, bounds, workers, opt_params, max_evals)
128 if is_debug_active():
129 ys = np.sort(archive.get_ys())[:min(100, archive.capacity)] # best fitness values
130 logger.debug(f'best {min(ys):.3f} worst {max(ys):.3f} ' +
131 f'mean {np.mean(ys):.3f} stdev {np.std(ys):.3f} time {dtime(t0)} s')
132 return archive

134def apply_advretry(fitness: Callable[[ArrayLike], float],
135 qd_fitness: Callable[[ArrayLike], Tuple[float, np.ndarray]],
136 bounds: Bounds,
137 archive: Archive,
138 optimizer: Optional[Optimizer] = None,
139 num_retries: Optional[int] = 1000,
140 workers: Optional[int] = mp.cpu_count(),
141 max_eval_fac: Optional[float] = 5.0,
142 xs: Optional[np.ndarray] = None,
143 ys: Optional[np.ndarray] = None,
144 x_conv: Callable[[ArrayLike], ArrayLike] = None):
145
146 """
147 Applies an advanced retry mechanism to optimize solutions for a given fitness function
148 and update an archive with optimized results.
149
150 This function manages an iterative process where previously computed solutions from
151 an archive are refined and optimized using a specified optimizer. Advanced retry logic
152 is utilized to ensure the process effectively minimizes the given fitness function
153 within the defined bounds and constraints.
154
155 Args:
156 fitness: Callable that evaluates the fitness of a solution. Must return a
157 floating-point fitness score for a given input.
158 qd_fitness: Callable that evaluates the quality-diversity of a solution. Returns
159 a tuple where the first item is the fitness value (float) and the second item is
160 a feature descriptor (e.g., NumPy array).
161 bounds: Boundary constraints for the optimization process.
162 archive: Archive object that stores and manages solutions.
163 optimizer: Optional optimizer to be used for the minimization process. If none,
164 defaults to a DE-CMA optimizer with 1500 iterations.
165 num_retries: Optional integer setting the number of retries allowed in
166 advanced retry logic. Defaults to 1000.
167 workers: Optional integer specifying the number of workers for parallelization.
168 Defaults to the number of CPUs available.
169 max_eval_fac: Optional floating-point factor setting the maximum allowed
170 fitness function evaluations per retry. Defaults to 5.0.
171 xs: Optional NumPy array containing a set of previously obtained solutions.
172 If none, solutions are derived from the archive's current entries.
173 ys: Optional NumPy array containing fitness values corresponding to `xs`.
174 If none, fitness values are derived from the archive's entries.
175 x_conv: Optional callable to transform or convert solutions (`xs`) before
176 evaluating their quality-diversity fitness. If none, no transformation is applied.
177 """
178
179 if optimizer is None:
180 optimizer = de_cma(1500)
181 # generate advretry store
182 store = advretry.Store(fitness, bounds, num_retries=num_retries,
183 max_eval_fac=max_eval_fac)
184
185 # select only occupied entries
186 if xs is None:
187 ys = archive.get_ys()
188 valid = (ys < np.inf)
189 ys = ys[valid]
190 xs = archive.xs_view[valid]
191 t0 = perf_counter()
192 # transfer to advretry store
193 for i in range(len(ys)):
194 store.add_result(ys[i], xs[i], 1)
195 # perform parallel retry
196 advretry.retry(store, optimizer.minimize, workers=workers)
197 # transfer back to archive
198 xs = store.xs_view
199 if not x_conv is None:
200 xs = [x_conv(x) for x in xs]
201 yds = [qd_fitness(x) for x in xs]
202 descs = np.array([yd[1] for yd in yds])
203 ys = np.array([yd[0] for yd in yds])
204 niches = archive.index_of_niches(descs)
205 for i in range(len(ys)):
206 archive.set(niches[i], (ys[i], descs[i]), xs[i])
207 archive.argsort()
208 if is_debug_active():
209 ys = np.sort(archive.get_ys())[:min(100, archive.capacity)] # best fitness values
210 logger.debug(f'best {min(ys):.3f} worst {max(ys):.3f} ' +
211 f'mean {np.mean(ys):.3f} stdev {np.std(ys):.3f} time {dtime(t0)} s')

213def minimize_parallel_(archive, fitness, bounds, workers, opt_params, max_evals):
214 """
215 Minimizes a fitness function in parallel using multiple workers.
216
217 This function utilizes multiprocessing to divide the workload of minimizing
218 a fitness function across several worker processes. Each worker operates
219 independently using its own random number generator, and the results are
220 combined to find the minimum value.
221
222 Args:
223 archive: Object or structure used to store and manage state or intermediate
224 results during the optimization process. Details of its structure or
225 behavior depend on the implementation.
226 fitness: Callable that evaluates the fitness or cost function to be
227 minimized. Accepts input variables and returns a numerical value
228 representing the cost/fitness.
229 bounds: Defines the boundaries within which the optimization process is
230 allowed to search. Typically, this can be a list of tuples specifying
231 the lower and upper bounds for each dimension.
232 workers: Integer specifying the number of parallel workers (processes) to
233 execute the fitness function evaluation and optimization tasks.
234 opt_params: Parameters or configuration values required for the optimization
235 method. Includes details specific to the optimization algorithm being
236 employed.
237 max_evals: Integer representing the maximum number of fitness function
238 evaluations allowed across all worker processes in total.
239
240 """
241 sg = SeedSequence()
242 rgs = [Generator(PCG64DXSM(s)) for s in sg.spawn(workers)]
243 evals = mp.RawValue(ct.c_long, 0)
244 proc=[Process(target=run_minimize_,
245 args=(archive, fitness, bounds, rgs[p],
246 opt_params, p, workers, evals, max_evals)) for p in range(workers)]
247 [p.start() for p in proc]
248 [p.join() for p in proc]

250def run_minimize_(archive, fitness, bounds, rg, opt_params, p, workers, evals, max_evals):
251 """
252 Executes the optimization processes involving MAP-Elites or solvers in a sequential or
253 multiple configuration depending on the provided optimization parameters. The method adapts
254 to the given `opt_params`, evaluating various solvers until the maximum number of allowed
255 evaluations (`max_evals`) is reached.
256
257 Args:
258 archive: Archive object used for storing solutions and managing population niches.
259 fitness: A callable fitness function used to evaluate solutions.
260 bounds: Boundary constraints associated with the problem.
261 rg: Random generator for deterministic random processes.
262 opt_params: Optimization parameters, which can be in the form of a dictionary, list,
263 tuple, or NumPy array.
264 p: Integer indicating the number of processing resources available.
265 workers: Integer specifying the number of worker threads or processes allocated for
266 solver execution.
267 evals: A shared counter object tracking the current total number of evaluations performed.
268 max_evals: Maximum allowed number of evaluations over the optimization processes.
269
270 """
271 with threadpoolctl.threadpool_limits(limits=1, user_api="blas"):
272 if isinstance(opt_params, (list, tuple, np.ndarray)):
273 default_workers = int(workers/2) if len(opt_params) > 1 else workers
274 for params in opt_params: # call MAP-Elites
275 if 'elites' == params.get('solver'):
276 elites_workers = params.get('workers', default_workers)
277 if p < elites_workers:
278 run_map_elites_(archive, fitness, bounds, rg, evals, max_evals, params)
279 return
280 while evals.value < max_evals: # call solvers in loop
281 best_x = None
282 if isinstance(opt_params, (list, tuple, np.ndarray)):
283 for params in opt_params: # call in sequence
284 if 'elites' == params.get('solver'):
285 continue # ignore in loop
286 if best_x is None:
287 # selecting a niche elite is no improvement over random x0
288 x0 = None#, _, _ = archive.random_xs_one(select_n, rg)
289 best_x = minimize_(archive, fitness, bounds, rg, evals, max_evals, params,
290 x0 = x0)
291 else:
292 best_x = minimize_(archive, fitness, bounds, rg, evals, max_evals, params, x0 = best_x)
293 else:
294 minimize_(archive, fitness, bounds, rg, evals, max_evals, opt_params)

298def run_map_elites_(archive, fitness, bounds, rg, evals, max_evals, opt_params = {}):
299 """
300 Executes the MAP-Elites algorithm for a given archive and fitness function.
301
302 The function iteratively generates a population of individuals, applies variation
303 operators on them, evaluates their fitness, and updates the archive with the newly
304 generated individuals. It can utilize either simulated binary crossover (SBX) or isolation
305 distribution crossover (ISO/DD) depending on the specified optimization parameters.
306 Additionally, the function includes boundaries for the search space and adjusts the
307 archive to maintain its capacity.
308
309 Args:
310 archive (object): The data structure representing the archive of solutions.
311 fitness (callable): The fitness function to evaluate each solution.
312 bounds (object): The search space bounds containing lower (`lb`) and upper (`ub`)
313 limits for variables.
314 rg (numpy.random.Generator): Random number generator for stochastic operations.
315 evals (object): Object to track the number of evaluations performed.
316 max_evals (int): Maximum number of evaluations allowed for the algorithm.
317 opt_params (dict, optional): A dictionary of optional parameters for optimization.
318 Includes:
319 - popsize (int): Population size for the algorithm (default: 32).
320 - use_sbx (bool): Whether to use simulated binary crossover (default: True).
321 - dis_c (float): Distribution index for simulated binary crossover (SBX) (default: 20).
322 - dis_m (float): Distribution index for mutation (default: 20).
323 - iso_sigma (float): Standard deviation for isotropic distribution (default: 0.01).
324 - line_sigma (float): Standard deviation for line distribution (default: 0.2).
325 """
326 popsize = opt_params.get('popsize', 32)
327 use_sbx = opt_params.get('use_sbx', True)
328 dis_c = opt_params.get('dis_c', 20)
329 dis_m = opt_params.get('dis_m', 20)
330 iso_sigma = opt_params.get('iso_sigma', 0.01)
331 line_sigma = opt_params.get('line_sigma', 0.2)
332 select_n = archive.capacity
333 while evals.value < max_evals:
334 if use_sbx:
335 pop = archive.random_xs(select_n, popsize, rg)
336 xs = variation_(pop, bounds.lb, bounds.ub, rg, dis_c, dis_m)
337 else:
338 x1 = archive.random_xs(select_n, popsize, rg)
339 x2 = archive.random_xs(select_n, popsize, rg)
340 xs = iso_dd_(x1, x2, bounds.lb, bounds.ub, rg, iso_sigma, line_sigma)
341 yds = [fitness(x) for x in xs]
342 evals.value += popsize
343 descs = np.array([yd[1] for yd in yds])
344 niches = archive.index_of_niches(descs)
345 for i in range(len(yds)):
346 archive.set(niches[i], yds[i], xs[i])
347 archive.argsort()
348 select_n = archive.get_occupied()

350def minimize_(archive, fitness, bounds, rg, evals, max_evals, opt_params, x0 = None):
351 """
352 Minimizes a given objective function using an evolutionary algorithm or the B.I.T.E. solver.
353
354 The function determines the solver type based on the provided optimization parameters
355 and runs the optimization process accordingly. It continuously updates an archive
356 of candidate solutions and evaluates their fitness while adhering to the specified
357 evaluation and iteration limits. A stopping condition is also applied based on lack
358 of improvement.
359
360 Args:
361 archive: Archive object for storing solution candidates and their respective
362 fitness evaluations.
363 fitness: Callable representing the fitness function or objective
364 function to be minimized.
365 bounds: Bounds or constraints for the solution search space.
366 rg: Random number generator to ensure reproducibility.
367 evals: A mutable object, typically an integer, tracking the number of
368 evaluations performed.
369 max_evals: Integer specifying the maximum number of fitness evaluations to allow.
370 opt_params: Dictionary containing optimization parameters, such as solver type
371 and stopping criteria.
372 x0: Optional initial guess or starting point for the optimization process.
373
374 Returns:
375 The best found solution, represented as a real-valued array,
376 that optimizes the provided fitness function.
377 """
378 if 'BITE_CPP' == opt_params.get('solver'):
379 return run_bite_(archive, fitness, bounds, rg, evals, max_evals, opt_params, x0 = None)
380 else:
381 es = get_solver_(bounds, opt_params, rg, x0)
382 stall_criterion = opt_params.get('stall_criterion', 20)
383 max_evals_iter = opt_params.get('max_evals', 50000)
384 max_iters = int(max_evals_iter/es.popsize)
385 old_ys = None
386 last_improve = 0
387 best_x = None
388 best_y = np.inf
389 for iter in range(max_iters):
390 xs = es.ask()
391 ys, real_ys = update_archive(archive, xs, fitness)
392 evals.value += es.popsize
393 # update best real fitness
394 yi = np.argmin(real_ys)
395 ybest = real_ys[yi]
396 if ybest < best_y:
397 best_y = ybest
398 best_x = xs[yi]
399 if not old_ys is None:
400 if (np.sort(ys) < old_ys).any():
401 last_improve = iter
402 if last_improve + stall_criterion < iter:
403 break
404 stop = es.tell(ys)
405 if stop != 0 or evals.value >= max_evals:
406 break
407 old_ys = np.sort(ys)
408 return best_x # real best solution

412def run_bite_(archive, fitness, bounds, rg, evals, max_evals, opt_params, x0 = None):
413 """
414 Runs the BiteOpt algorithm to optimize a given fitness function.
415
416 The function utilizes the BiteOpt implementation from bitecpp
417 to minimize the provided fitness function over the given bounds and constraints.
418 It supports dynamic updates to the solution archive and stops execution
419 based on a defined maximum evaluation limit or optimization parameters.
420
421 Args:
422 archive: An archive to keep track of the solution space explored during the optimization process.
423 fitness: A callable that takes an input, evaluates it, and returns a fitness value.
424 bounds: The variable bounds for the optimization problem.
425 rg: A random generator instance to ensure reproducibility in the optimization process.
426 evals: An object containing a mutable integer used to track the number of evaluations performed.
427 max_evals: An integer defining the maximum number of evaluations before the optimization halts.
428 opt_params: A dictionary containing optimization parameters such as 'max_evals' and 'stall_criterion'.
429 x0: Optional starting point for the optimization. Defaults to None.
430
431 Returns:
432 The optimized solution vector obtained from BiteOpt.
433 """
434 # BiteOpt doesn't support ask/tell, so we have to "patch" fitness. Note that Voronoi
435 # tesselation is more expensive if called for single behavior vectors and not for batches.
436
437 def fit(x: Callable[[ArrayLike], float]):
438 """
439 Evaluates a given function with constraints on a maximum number of evaluations.
440
441 This function checks if the given number of evaluations exceeds the permitted
442 maximum before proceeding. If the threshold is not surpassed, it updates the
443 archive with the computed fitness values and returns the fitness of the evaluated
444 input.
445
446 Args:
447 x: A callable function that takes an ArrayLike input and returns a float
448 representing the fitness value.
449
450 Returns:
451 float: The computed fitness value of the input `x`. If the maximum allowed
452 evaluations are reached, returns infinity.
453
454 Raises:
455 ValueError: If the input function (or its return value) does not align with
456 the expected structure or type definitions during processing.
457 """
458 if evals.value >= max_evals:
459 return np.inf
460 evals.value += 1
461 ys, _ = update_archive(archive, [x], fitness)
462 return ys[0]
463
464 max_evals_iter = opt_params.get('max_evals', 50000)
465 stall_criterion = opt_params.get('stall_criterion', 20)
466 #popsize = opt_params.get('popsize', 0)
467 ret = bitecpp.minimize(fit, bounds, x0 = x0, M = 1,
468 stall_criterion = stall_criterion,
469 max_evaluations = max_evals_iter, rg = rg)
470 return ret.x

472def get_solver_(bounds, opt_params, rg, x0 = None):
473 """
474 Selects and initializes the appropriate optimization solver based on the specified
475 parameters. The solver is chosen from a set of predefined options, and it is configured
476 with the given dimensions, bounds, mean, population size, and other solver-specific
477 parameters.
478
479 Args:
480 bounds: Object representing the bounds for the optimization problem. It provides
481 attributes like lower bounds (`lb`) and upper bounds (`ub`).
482 opt_params: Dictionary containing optional solver parameters, such as:
483 - 'popsize': Population size
484 - 'sigma': Step size
485 - 'mean': Initial mean position
486 - 'solver': Name of the solver to use, e.g., 'CMA', 'CMA_CPP', etc.
487 rg: Random number generator for initializing values within the specified bounds
488 or for stochastic components of the solver.
489 x0: Optional starting position for the optimization. Overrides the `mean` parameter
490 if provided.
491
492 Returns:
493 The initialized optimization solver object if the specified solver name is valid.
494 Returns `None` if an invalid solver name is provided.
495 """
496 dim = len(bounds.lb)
497 popsize = opt_params.get('popsize', 31)
498 #sigma = opt_params.get('sigma',rg.uniform(0.03, 0.3)**2)
499 sigma = opt_params.get('sigma',rg.uniform(0.1, 0.5)**2)
500 #sigma = opt_params.get('sigma',rg.uniform(0.2, 0.5)**2)
501 #sigma = opt_params.get('sigma',rg.uniform(0.1, 0.5))
502 mean = opt_params.get('mean', rg.uniform(bounds.lb, bounds.ub)) \
503 if x0 is None else x0
504 name = opt_params.get('solver', 'CMA_CPP')
505 if name == 'CMA':
506 return cmaes.Cmaes(bounds, x0 = mean,
507 popsize = popsize, input_sigma = sigma, rg = rg)
508 elif name == 'CMA_CPP':
509 return cmaescpp.ACMA_C(dim, bounds, x0 = mean, #stop_hist = 0,
510 popsize = popsize, input_sigma = sigma, rg = rg)
511 elif name == 'CRMFNES':
512 return crfmnes.CRFMNES(dim, bounds, x0 = mean,
513 popsize = popsize, input_sigma = sigma, rg = rg)
514 elif name == 'CRMFNES_CPP':
515 return crfmnescpp.CRFMNES_C(dim, bounds, x0 = mean,
516 popsize = popsize, input_sigma = sigma, rg = rg)
517 elif name == 'DE':
518 return de.DE(dim, bounds, popsize = popsize, rg = rg)
519 elif name == 'DE_CPP':
520 return decpp.DE_C(dim, bounds, popsize = popsize, rg = rg)
521 elif name == 'PGPE':
522 return pgpecpp.PGPE_C(dim, bounds, x0 = mean,
523 popsize = popsize, input_sigma = sigma, rg = rg)
524 else:
525 print ("invalid solver")
526 return None
