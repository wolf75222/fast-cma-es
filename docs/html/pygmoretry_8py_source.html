15 Description:
16 - This module provides a retry mechanism for the PYGMO/PAGMO optimization framework.
17 - It allows for parallel retries of optimization problems using the PYGMO/PAGMO library.
18 - It is designed to work with problems that have constraints or multiple objectives,
19 which cannot be handled by the standard fcmaes.retry module.
20 - The retry mechanism uses multiprocessing to perform multiple optimization attempts
21 in parallel, improving efficiency and scalability.

87def retry(store, prob, algo, num_retries, value_limit = np.inf, popsize=1, workers=mp.cpu_count()):
88 """
89 Retries optimization over multiple attempts across parallel workers to find
90 the best result. The function orchestrates the parallel execution, manages
91 random number generators for each worker, and aggregates the results.
92
93 Args:
94 store: An object responsible for storing and managing the optimization
95 results.
96 prob: The optimization problem to be solved.
97 algo: The algorithm used to perform optimization.
98 num_retries: The number of retries to attempt for the optimization process.
99 value_limit: The upper limit for the values considered valid in the
100 optimization result (default is np.inf).
101 popsize: The population size used in the optimization process
102 (default is 1).
103 workers: The number of parallel workers to use for the optimization
104 (default is the number of CPU cores available).
105
106 Returns:
107 OptimizeResult: An object containing the best solution (`x`), the best
108 objective value (`fun`), the number of function evaluations performed
109 (`nfev`), and the success status of the optimization (`success`).
110
111 Raises:
112 ImportError: If the Pygmo library is not installed.
113 """
114 try:
115 import pygmo as pg
116 except ImportError as e:
117 raise ImportError("Please install PYGMO (pip install pygmo) to use PAGMO optimizers") from e
118 sg = SeedSequence()
119 rgs = [Generator(PCG64DXSM(s)) for s in sg.spawn(workers)]
120 proc=[Process(target=_retry_loop,
121 args=(pid, rgs, store, prob, algo, num_retries, value_limit, popsize, pg)) for pid in range(workers)]
122 [p.start() for p in proc]
123 [p.join() for p in proc]
124 store.sort(store.get_xs())
125 store.dump()
126 return OptimizeResult(x=store.get_x_best(), fun=store.get_y_best(),
127 nfev=store.get_count_evals(), success=True)

129def _retry_loop(pid, rgs, store, prob, algo, num_retries, value_limit, popsize, pg):
130 """
131 Executes a retry loop for a given probabilistic algorithm to attempt finding a feasible solution.
132
133 The function runs multiple attempts to generate solutions using a probabilistic algorithm. It retrieves a random seed,
134 evolves a population, and evaluates the best solution. Feasible solutions meeting specified criteria are added to a
135 result store.
136
137 Args:
138 pid (int): Identifier for the process or individual task.
139 rgs (list): List of random generators corresponding to each identifier.
140 store (object): Object responsible for storing results and managing retries.
141 prob (object): Problem object defining the optimization problem.
142 algo (object): Algorithm used to evolve populations.
143 num_retries (int): Maximum number of retries to evolve a solution.
144 value_limit (float): Feasibility threshold for solutions.
145 popsize (int): Size of the population used in each evolutionary attempt.
146 pg (object): External module used for handling evolutionary population and problem definition.
147 """
148 while store.get_runs_compare_incr(num_retries):
149 try:
150 seed = int(rgs[pid].uniform(0, 2**32 - 1))
151 pop = pg.population(prob, popsize, seed=seed)
152 pop = algo.evolve(pop)
153 except Exception:
154 pass # ignore "Maximum number of iteration reached"
155 sol = pop.champion_x
156 y = pop.champion_f
157 evals = pop.problem.get_fevals()
158
159 _feasible = prob.feasibility_x(pop.champion_x)
160 if _feasible:
161 store.add_result(y[0], sol, evals, value_limit)
162 store.dump()
