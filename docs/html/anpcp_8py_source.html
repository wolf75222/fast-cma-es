28@njit(fastmath=True)
29def next_free_(used, p):
30 while used[p]:
31 p = (p + 1) % used.size
32 used[p] = True
33 return p
34
35@njit(fastmath=True)
36def selection_(s, n):
37 disjoined_s = np.zeros(s.size, dtype=numba.int32)
38 used = np.zeros(n, dtype=numba.boolean)
39 for i in range(s.size):
40 disjoined_s[i] = next_free_(used, s[i])
41 return disjoined_s

43@njit(fastmath=True)
44def calc_distance_(users, facilities):
45 distance = np.zeros((len(users), len(facilities)), dtype=numba.int32)
46 for i in range(len(users)):
47 for j in range(len(facilities)): # round to next int
48 dx = users[i,0] - facilities[j,0]
49 dy = users[i,1] - facilities[j,1]
50 distance[i,j] = int(round(np.sqrt(dx*dx + dy*dy), 0))
51 return distance
52
53@njit(fastmath=True)
54def fitness_(selection, distance, alpha):
55 selected = distance[:,selection]
56 partitioned = np.partition(selected, alpha)
57 return max([max(d[:alpha]) for d in partitioned])
58 # variant where instead of the alpha-best we use the sum of the best alpha distances
59 # return max([np.sum(d[:alpha]) for d in partitioned])

78 def init_tsp(self, tsp_file):
79 with open(tsp_file) as csvfile:
80 lines = csvfile.readlines()
81 users = []
82 facilities = []
83 for line in lines:
84 row = line.split()
85 if len(row) != 4 or not row[0][0].isdigit():
86 continue
87 coords = np.array([float(row[1]), float(row[2])])
88 if row[3] == '1':
89 facilities.append(coords)
90 else:
91 users.append(coords)
92 self.users = np.array(users)
93 self.facilities = np.array(facilities)
94 self.unum = len(self.users)
95 self.fnum = len(self.facilities)
96 self.distance = calc_distance_(self.users, self.facilities)
97 self.bounds = Bounds([0]*self.dim, [self.fnum-1E-9]*self.dim)

99 def random_x(self, seed = 123):

110def optimize(anpcp, opt, num_retries = 32):
111 ret = retry.minimize(wrapper(anpcp.fitness),
112 anpcp.bounds, num_retries = num_retries,
113 optimizer=opt)
114 print("selection = ", anpcp.get_selection(ret.x))
