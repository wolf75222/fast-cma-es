316 void operator()(const state_t &pv, state_t &yDot, const double t) {
317
318 // position
319 double x = pv[0];
320 double y = pv[1];
321 double z = pv[2];
322
323 // velocity
324 yDot[0] = pv[3];
325 yDot[1] = pv[4];
326 yDot[2] = pv[5];
327
328 double rq = x * x + y * y + z * z;
329 double r = sqrt(rq);
330 double rq2 = rq * rq;
331 double rq3 = rq2 * rq;
332 double rrr = r*r*r;
333
334 double z15 = 15.0*z*z / rq3;
335 double rq23 = 3.0 / rq2;
336 double muERe2V5C20q2r = muERe2V5C20 / (2*r);
337
338 // earth gravity
339 double fkepX = -muE * x / rrr;
340 double fkepY = -muE * y / rrr;
341 double fkepZ = -muE * z / rrr;
342
343 // J2 perturbation
344 double fJ2X = muERe2V5C20q2r * x * (rq23 - z15);
345 double fJ2Y = muERe2V5C20q2r * y * (rq23 - z15);
346 double fJ2Z = muERe2V5C20q2r * z * (3*rq23 - z15);
347
348 // solar radiation pressure
349 double xsun, ysun, zsun;
350 sunr(t, xsun, ysun, zsun);
351
352 double rsun = sqrt(xsun * xsun + ysun * ysun + zsun * zsun);
353 double rsun3 = rsun*rsun*rsun;
354
355 double xsd = x - xsun;
356 double ysd = y - ysun;
357 double zsd = z - zsun;
358
359 double rsd = sqrt(xsd * xsd + ysd * ysd + zsd * zsd);
360 double rsd3 = rsd*rsd*rsd;
361
362 double fSRPX = _cram * xsd * Psrpas2 / rsd3;
363 double fSRPY = _cram * ysd * Psrpas2 / rsd3;
364 double fSRPZ = _cram * zsd * Psrpas2 / rsd3;
365
366 // C22
367 double tGvEt = thetaG + vE * t;
368 double costGvEt = cos(tGvEt);
369 double sintGvEt = sin(tGvEt);
370
371 double xc = x * costGvEt + y * sintGvEt;
372 double yc = -x * sintGvEt + y * costGvEt;
373 double zc = z;
374
375 double xcq = xc*xc;
376 double ycq = yc*yc;
377 double zcq = zc*zc;
378
379 double rc = xcq + ycq + zcq;
380 double rc5 = rc*rc*rc*rc*rc;
381 double rc5s = sqrt(rc5);
382 double rc7s = sqrt(rc5*rc*rc);
383
384 double fc22x = 5*muERe2V15C22*xc*(ycq-xcq) / (2*rc7s) + muERe2V15C22*x / rc5s;
385 double fc22y = 5*muERe2V15C22*yc*(ycq-xcq) / (2*rc7s) + muERe2V15C22*y / rc5s;
386 double fc22Z = 5*muERe2V15C22*zc*(ycq-xcq) / (2*rc7s);
387 double fc22X = fc22x * costGvEt - fc22y * sintGvEt;
388 double fc22Y = fc22x * sintGvEt - fc22y * costGvEt;
389
390 // S22
391 double fs22x = -5*muERe2V15S22*xcq*y / rc7s + muERe2V15S22*y / rc5s;
392 double fs22y = -5*muERe2V15S22*xc*ycq / rc7s + muERe2V15S22*x / rc5s;
393 double fs22Z = -5*muERe2V15S22*zc*(ycq-xcq) / rc7s;
394 double fs22X = fs22x * costGvEt - fs22y * sintGvEt;
395 double fs22Y = fs22x * sintGvEt - fs22y * costGvEt;
396
397 // solar gravity
398 double fsunX = -muS * (xsd / rsd3 + xsun / rsun3);
399 double fsunY = -muS * (ysd / rsd3 + ysun / rsun3);
400 double fsunZ = -muS * (zsd / rsd3 + zsun / rsun3);
401
402 // lunar gravity
403 // solar radiation pressure
404 double xmoo, ymoo, zmoo;
405 moonr(t, xmoo, ymoo, zmoo);
406
407 double rmoo = sqrt(xmoo * xmoo + ymoo * ymoo + zmoo * zmoo);
408 double rmoo3 = rmoo*rmoo*rmoo;
409
410 double xmd = x - xmoo;
411 double ymd = y - ymoo;
412 double zmd = z - zmoo;
413
414 double rmd = sqrt(xmd * xmd + ymd * ymd + zmd * zmd);
415 double rmd3 = rmd*rmd*rmd;
416
417 double fmooX = -muM * (xmd / rmd3 + xmoo / rmoo3);
418 double fmooY = -muM * (ymd / rmd3 + ymoo / rmoo3);
419 double fmooZ = -muM * (zmd / rmd3 + zmoo / rmoo3);
420
421 // acceleration
422 yDot[3] = fkepX + fJ2X + fSRPX + fc22X + fs22X + fsunX + fmooX;
423 yDot[4] = fkepY + fJ2Y + fSRPY + fc22Y + fs22Y + fsunY + fmooY;
424 yDot[5] = fkepZ + fJ2Z + fSRPZ + fc22Z + fs22Z + fsunZ + fmooZ;
425 }
