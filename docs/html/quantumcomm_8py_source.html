100@njit(fastmath=True)
101def line_of_sight(r1,r2):
102 """Given two position vectors returns the distance of the line of sight to the origin
103
104 Args:
105 r1 (numpy array): first point
106 r2 (numpy array): second point
107 """
108 denom = np.linalg.norm(r2-r1)
109 if denom < 1e-6:
110 # if r1 ~= r2, it will return the norm of r1
111 return np.linalg.norm(r1)
112 else:
113 r21 = (r2-r1) / denom
114 h1 = np.dot(r1,r21)
115 arg = np.linalg.norm(r1)**2 - h1**2
116 # We check for a positive arg in case r1 and r2 are near collinearity
117 return np.sqrt(arg) if arg > 1e-6 else 0.0

119@njit(fastmath=True)
120def zenith_angle(src, dst):
121 """Computes the cosine of the zenith angle (theta_z) of the LOS between source and destination node
122
123 Args:
124 src (numpy array, N_r x 3): rover x, y, z positions
125 dst (numpy array, N_s x 3): mothership x, y, z positions
126
127 Returns:
128 float: cosine of the zenith angle
129 """
130 dpos = dst - src
131 if np.linalg.norm(dpos) < 1e-6:
132 cos_theta_z = 0
133 else:
134 cos_theta_z = np.dot(dpos, src) / (np.linalg.norm(dpos) * np.linalg.norm(src))
135 return cos_theta_z

137@njit(fastmath=True)
138def qkd_metric(idx, src, dst, cos_theta_z, eta, eps_z, n_rovers):
139 """Computes the edge weight according to QKD probabilities
140
141 Args:
142 idx (int): index of node in the graph
143 src (numpy array, 1x3): position of source node
144 dst (numpy array, 1x3): position of destination node
145 cos_theta_z (float): cosine of the zenith angle of qkd link
146 eta (int): satellite quality indicator for corresponding constellation
147
148 Returns:
149 float: edge weight
150 float: communications link distance between src and dst
151 """
152 edge_weight = -np.log(eta) # constellation quality score
153 d_link = np.linalg.norm(src - dst) # distance of communications link
154 edge_weight += 2 * np.log(d_link) # final edge weight
155 if edge_weight < 0:
156 # Safeguard: whenever this happens, the collision-avoidance constraint is
157 # also not satisfied. Nevertheless, we must return a value for the edge weight
158 # to ensure that the fitness does not throw (a negative valued edge would also
159 # imply the non-existence of a shortest path)
160 edge_weight = 1e3
161
162 if idx <= n_rovers:
163 if cos_theta_z >= eps_z: # Apply max zenith angle constraint to mothership-rover link
164 edge_weight += 1.0 / np.sin(np.pi / 2 - np.arccos(cos_theta_z))
165 else:
166 edge_weight = 0
167 return edge_weight, d_link

169@njit(fastmath=True)
170def get_adjmatrix(pos, ep_idx, eta, num_w1_sats, LOS, N, eps_z, n_rovers):
171 adjmatrix = np.zeros((N, N))
172 d_min = np.inf
173 for i in range(N):
174 for j in range(i):
175 # Ensure there is LOS
176 los = line_of_sight(pos[i, ep_idx, :], pos[j, ep_idx, :])
177 cos_theta_z = zenith_angle(pos[i, ep_idx, :], pos[j, ep_idx, :])
178 if los >= LOS or cos_theta_z > 0:
179 # Eta based on j because it is the destination satellite in the link
180 eta_j = eta[0] if j < num_w1_sats else eta[1]
181 adjmatrix[i,j], d_link = \
182 qkd_metric(N-i, pos[i, ep_idx, :], pos[j, ep_idx, :], cos_theta_z, eta_j, eps_z, n_rovers)
183 if d_link < d_min:
184 d_min = d_link
185 adjmatrix[j,i] = adjmatrix[i,j]
186 return adjmatrix, d_min

188@njit(fastmath=True)
189def construct_rover_pos(lambda0, phi0, n_rovers, jds, R_p, w_p):
190 """Computes the position of the rovers at time t based on the initial latitude and longitude
191
192 Args:
193 lambda0 (float, N_r x 1): initial latitudes of the rovers
194 phi0 (float, N_r x 1): initial longitudes of the rovers
195
196 Returns:
197 float, n_rovers x n_epochs x 3: rover x, y, z positions
198 """
199 pos_r = np.zeros((n_rovers, jds.shape[0], 3))
200 time_range = (jds - jds[0]) * 24 * 3600 # in seconds
201 for i, t in enumerate(time_range):
202 pos_r[:, i, 0] = R_p * np.cos(lambda0) * np.cos(phi0 + w_p * t) # x
203 pos_r[:, i, 1] = R_p * np.cos(lambda0) * np.sin(phi0 + w_p * t) # y
204 pos_r[:, i, 2] = R_p * np.sin(lambda0) # z
205 return pos_r

207class constellation_udp:
208 """A Pygmo compatible UDP (User Defined Problem) representing the constellation design problem for SpOC 2023.
209
210 Two Walker constellations are defined in a mixint chromosome:
211 x = [a1,ei,i1,w1,eta1] + [a2,e2,i2,w2,eta2] + [S1,P1,F1] + [S2,P2,F2] + [r1,r2,r3,r4]
212
213 The constellations must relay information between 7 motherships in orbit and 4 rovers on the surface of New Mars
214 """
215 def __init__(self):
216 """Constructor"""
217
218 # Define the time grid over which to optimize the communications network
219 self._t0 = 10000 # starting epoch in mjd2000
220 self.n_epochs = 11 # number of epochs in time grid to consider
221 self._duration = 10 # difference between the first and last epoch considered in years
222 jd0, fr = pk.epoch(self._t0, 'mjd2000').jd, 0.0 # reference Julian date
223 self.jds = np.linspace(jd0, jd0 + self._duration * 365.25, self.n_epochs) # time grid in JD
224 self.frs = self.jds * fr # date fractions (defaults to 0)
225
226 # Reference epoch for SGP4 is 1949 December 31 00:00 UT
227 self.ep_ref = pk.epoch_from_iso_string("19491231T000000")
228
229 # SGP4-ready mothership satellites
230 mothership_satellites = get_mothership_satellites()
231 self.pos_m = self.construct_mothership_pos(SatrecArray(mothership_satellites))
232 self.n_motherships = len(mothership_satellites)
233
234 # Latitudes and longitudes of rovers
235 rovers_db_path = os.path.join(".", "data", "spoc2", "constellations", "rovers.txt")
236 self.rovers_db = np.loadtxt(rovers_db_path)
237 self.lambdas = self.rovers_db[:, 0] # latitudes
238 self.phis = self.rovers_db[:, 1] # longitudes
239 self._min_rover_dist = 3000 # minimum inter-rover distance (km)
240 self.n_rovers = 4
241
242 # Minimum line-of-sight parameter (km)
243 # Radius amplification factor: 1.05
244 self.LOS = 1.05 * pk.EARTH_RADIUS / 1000.0
245 # Radius of the New-Mars planet (km)
246 self.R_p = pk.EARTH_RADIUS / 1000.0
247 # Angular velocity of New Mars (rad/s)
248 self.w_p = 7.29e-5 # 2 * pi / (23 hours 56 minutes 4 seconds)
249 # Threshold zenith angle constraint for rover-sat link (rad)
250 self._zenith_angle = np.pi / 3
251 self.eps_z = np.cos(self._zenith_angle)
252 # Minimum inter-satellite distance (km)
253 self._min_sat_dist = 50

295 def get_rover_constraint(self, lambda0, phi0):
296 """Evaluate the rover constraint (minimum distance between any two rovers)
297
298 Args:
299 lambda0 (float, N_r x 1): latitudes of the rovers
300 phi0 (float, N_r x 1): longitudes of the rovers
301
302 Returns
303 float: the difference between the actual and allowable minimum distance between rovers
304 """
305 # Compute rover positions on the planet
306 pos = np.zeros((self.n_rovers, 3))
307 pos[:, 0] = np.sin(lambda0) * np.cos(phi0)
308 pos[:, 1] = np.cos(lambda0) * np.cos(phi0)
309 pos[:, 2] = np.sin(phi0)
310 def safe_arccos(u, v):
311 inner_product = np.dot(u, v)
312 if inner_product > 1:
313 return 0
314 if inner_product < -1:
315 return np.pi
316 return np.arccos(inner_product)
317 d = scipy.spatial.distance.cdist(pos, pos, lambda u, v: pk.EARTH_RADIUS/1000 * safe_arccos(u, v))
318 d = d + np.diag([np.inf]*4)
319 min_d = np.min(d)
320 # Will be negative if min(d) is larger than the min allowable inter-rover distance
321 return self._min_rover_dist - min_d, min_d

323 def get_sat_constraint(self, d_min):
324 """Evaluate the satellite constraint (minimum distance between any two satellites)
325
326 Args:
327 d_min (float): the minimum distance between any two satellites at any epoch
328
329 Returns:
330 float: the difference between the actual and allowable minimum distance between satellites
331 """
332 # Will be negative if d_min is larger than the min allowable inter-satellite distance
333 return self._min_sat_dist - d_min

336 def generate_walker(self, S,P,F,a,e,incl,w,t0):
337 """Generates a Walker constallation as a SatrecArray
338
339 Args:
340 S (int): number of satellites per plane
341 P (int): number of planes
342 F (int): spacing parameter (i.e. if 2 phasing repeats each 2 planes)
343 a (float): semi-major axis
344 e (float): eccentricity
345 incl (float): inclination
346 w (float): argument of perigee
347 t0 (float): epoch
348
349 Returns:
350 SatrecArray: satellites ready to be SGP4 propagated
351 """
352 walker_l = []
353 mean_motion = np.sqrt(pk.MU_EARTH/a**3/pk.EARTH_RADIUS**3)
354 # planes
355 for i in range(P):
356 #satellites
357 for j in range(S):
358 satellite = Satrec()
359 satellite.sgp4init(
360 WGS72, # gravity model
361 'i', # 'a' = old AFSPC mode, 'i' = improved mode
362 j + i*S, # satnum: Satellite number
363 t0-self.ep_ref.mjd2000, # epoch: days since 1949 December 31 00:00 UT
364 0.0, # bstar: drag coefficient (1/earth radii) - 3.8792e-05
365 0.0, # ndot: ballistic coefficient (revs/day)
366 0.0, # nddot: mean motion 2nd derivative (revs/day^3)
367 e, # ecco: eccentricity
368 w, # argpo: argument of perigee (radians)
369 incl, # inclo: inclination (radians)
370 2*np.pi/P/S*F*i+2.*np.pi/S*j, # mo: mean anomaly (radians)
371 mean_motion*60, # no_kozai: mean motion (radians/minute)
372 2.*np.pi/P*i # nodeo: R.A. of ascending node (radians)
373 )
374 walker_l.append(satellite)
375 # Creating the vectorized list
376 return SatrecArray(walker_l)

378 def build_graph(self, ep_idx, pos, num_w1_sats, eta):
379 """Builds a networkx graph from the satellite positions. Links are weighted via a "QKD-inspired metric
380 and only exist when motherships/constellation satellites/rovers have line-of-sight
381
382 Args:
383 ep_idx (int): idx of the epoch in the time grid
384 pos (numpy array 3xN): position vector of the satellites
385 num_w1_sats (int): number of satellites in the first Walker constellation
386 eta (tuple): satellite quality indicator for each Walker constellation
387
388 Returns:
389 igraph graph: nodes are motherships/Walker satellites/rovers; links are distances when there is LOS
390 """
391 N = pos[:, ep_idx, :].shape[0] # number of vertices
392 adjmatrix, d_min = get_adjmatrix(pos, ep_idx, eta, num_w1_sats, self.LOS, N, self.eps_z, self.n_rovers)
393 g = ig.Graph.Adjacency((adjmatrix > 0).tolist())
394 g.es['weight'] = adjmatrix[adjmatrix.nonzero()]
395 return g, adjmatrix, d_min

397 def average_shortest_path(self, G, src, dst):
398 """Computes the average shortest path length between the source and destination *partitions* of nodes in the graph *G*
399 (the source is assumed to be the motherships and the destination the rovers)
400
401 Args:
402 G (networkx graph): The graph
403 src (int): the number of motherships (to be used as a negative index in G)
404 dst (int): the number of rovers (to be used as a negative index in G)
405
406 Returns:
407 float: average shortest path
408 """
409 n_nodes = G.vcount()
410 src0 = n_nodes - src - dst
411 dst0 = n_nodes - dst
412 sp = np.array(G.distances(list(range(src0, src0+src)), \
413 list(range(dst0, dst0+dst)), weights=G.es["weight"]))
414 return np.mean(sp)

416 def construct_walkers(self, x):
417 """Generates two Walker constellations according to specifications
418
419 Args:
420 x (list): chromosome describing the New Mars communications infrastructure
421
422 Returns:

427 a1,e1,i1,w1,_,a2,e2,i2,w2,_,S1,P1,F1,S2,P2,F2,_,_,_,_ = x
428 # Construct the 1st walker constellation as a SatrecArray

434 def construct_mothership_pos(self, motherships):
435 """Computes the position of the motherships over a predefined time grid
436
437 Args:
438 motherships (SatrecArray): mothership satellites ready to be SGP4 propagated
439
440 Returns:
441 float, n_motherships x n_epochs x 3: mothership x, y, z positions

450 def construct_pos(self, walker1, walker2, pos_r):
451 """Construct cumulative position of Walker satellites, motherships and rovers
452
453 Args:
454 walker1 (SatrecArray): Walker1 constellation satellites ready to be SGP4 propagated
455 walker2 (SatrecArray): Walker2 constellation satellites ready to be SGP4 propagated
456 pos_r (float, n_rovers x n_epochs x 3): rover x, y, z positions
457
458 Returns:
459 float, (S1xP1 + S2xP2 + n_motherships + n_rovers) x n_epochs x 3: overall position vector
460 """
461 # Compute ephemerides for Walker1 satellites at all epochs)
462 err_w1, pos_w1, _ = walker1.sgp4(self.jds, self.frs)
463 # Compute ephemerides for Walker2 satellites at all epochs)
464 err_w2, pos_w2, _ = walker2.sgp4(self.jds, self.frs)
465 # Check propagation went well
466 if not (np.all(err_w1 == 0) and np.all(err_w2 == 0)):
467 raise ValueError("The walker constellations cannot be propagated successfully on the defined time grid")
468 # Position vector for Walker constellation satellites, motherships and rovers)
469 cum_pos = np.concatenate((pos_w1,pos_w2, self.pos_m, pos_r))
470 return cum_pos

472 def fitness(self, x, verbose=False):
473 """Evaluate the fitness of the decision variables.
474
475 Args:
476 x (list): chromosome describing the New Mars communications infrastructure
477 verbose (bool): If True, print some info.
478
479 Returns:
480 float: fitness for average shortest path
481 float: fitness for total number of satellites
482 float: constraint for rover positioning
483 """
484 # Construct the Walker constellations based on input chromosome
485 walker1, walker2 = self.construct_walkers(x)
486 # Extract the quality factors and the number of satellites in the Walkers
487 _,_,_,_,eta1,_,_,_,_,eta2,S1,P1,_,S2,P2,_,_,_,_,_ = x
488 N1 = S1 * P1
489 N2 = S2 * P2
490 # Extract the rover indices from the input chromosome
491 rovers_idx = np.array(x[-4:]).astype(int)
492 # Look up latitude and longitudes corresponding to rover indices
493 lambda0 = self.lambdas[rovers_idx]
494 phi0 = self.phis[rovers_idx]
495 # Construct the rover positions
496 rovers = construct_rover_pos(lambda0, phi0, self.n_rovers, self.jds, self.R_p, self.w_p)
497 # Concatenate the position of the Walkers, motherships and rover
498 cum_pos = self.construct_pos(walker1, walker2, rovers)
499
500 # Evaluating the fitness function
501 if verbose:
502 print("FITNESS EVALUATION:")
503
504 # First objective (minimize):
505 # Compute the average shortest path between any mothership-rover pair
506 # Iterate over epochs
507 f1 = 0
508 nf1 = 34 # f1 normalization factor
509 d_sat_min_ep = np.inf
510 for ep_idx in range(1, self.n_epochs):
511 # Constructs the graph:
512 # Nodes: Walker sats + motherships + rovers
513 # Edges: LOS communication
514 G, _, d_sat_min = self.build_graph(ep_idx, cum_pos, N1, (eta1, eta2))
515 if d_sat_min < d_sat_min_ep:
516 d_sat_min_ep = d_sat_min
517 f1 += self.average_shortest_path(G, self.n_motherships, self.n_rovers)
518
519 # Average over the number of epochs
520 f1 /= (self.n_epochs - 1)
521
522 # Second objective (minimize):
523 # Compute the total number of satellites (weighted by the quality factors)
524 f2 = eta1 * N1 + eta2 * N2
525 nf2 = 100000 # f2 normalization factor
526
527 # Constraints:
528 # The minimum distance between any two rovers needs to be at least 3000km
529 # to ensure good coverage of the surface of New Mars
530 min_rover_d, d_rover_min = self.get_rover_constraint(lambda0, phi0)
531 # The minimum distance between any two nodes of the graph across all epochs
532 # needs to be at least 50km to ensure a collision-free communications network
533 min_sat_d = self.get_sat_constraint(d_sat_min_ep)
534
535 # Additional information on the fitness of the input chromosome
536 if verbose:
537 print(100 * "-")
538 print("RESULTS:")
539 print("Total number of satellites (W1: {}, W2: {}): {}".format(N1, N2, N1+N2))
540 print("OBJECTIVE #1 - Average communications cost: {}".format(f1/nf1))
541 print("OBJECTIVE #2 - Cost of infrastructure: {}".format(f2/nf2))
542 print("CONSTRAINT - Minimum distance between rovers ({}): {} km".format("NOK" if min_rover_d > 0 else "OK", d_rover_min))
543 print("CONSTRAINT - Minimum distance between sats ({}): {} km".format("NOK" if min_sat_d > 0 else "OK", d_sat_min_ep))
544 print(100 * "-")
545 return [f1/nf1, f2/nf2, min_rover_d, min_sat_d]

547 def pretty(self, x):
548 """A verbose evaluation of the fitness functions
549
550 Args:
551 x (list): chromosome describing the New Mars communications infrastructure
552
553 Returns:
554 float: fitness for average shortest path
555 float: fitness for total number of satellites
556 float: constraint for rover positioning
557 float: constraint for satellite positioning

600 def compute_orbit_walker(self, walker, ep0, sma):
601 """Compute one full-orbit of the Walker constellation planes (for plots)
602
603 Args:
604 walker (sgp4.SatrecArray): the array of Walker satellites to plot
605 ep0 (float): Julian date denoting starting epoch
606 sma (float): semi-major axis of orbit
607
608 Returns:
609 pos (numpy array, P x N x 3): N orbital x, y, z positions for P planes
610 """
611
612 # Extract mean motion
613 mean_motion = np.sqrt(pk.MU_EARTH / sma**3 / pk.EARTH_RADIUS**3) * 24 * 60 * 60 / (2 * np.pi)
614 # Compute time range for one full orbit
615 jds = np.linspace(ep0, ep0 + 1 / mean_motion, 100)
616 frs = jds * 0.0
617 # Propagate using SGP4
618 err, pos, _ = walker.sgp4(jds, frs)
619 if not np.all(err == 0):
620 raise ValueError("The satellite cannot be propagated successfully on the defined time grid")
621
622 return pos
