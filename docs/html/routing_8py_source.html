19@njit(fastmath=True)
20def fitness_(seq, distance, demands, capacity):
21 n = len(seq)
22 seq += 1
23 sum_demand = 0
24 sum_dist = 0
25 last = 0
26 vehicles = 1
27 for i in range(n+1):
28 customer = seq[i] if i < n else 0
29 demand = demands[customer]
30 if sum_demand + demand > capacity:
31 # end vehicle tour, return to base
32 sum_dist += distance[last, 0]
33 sum_demand = 0
34 vehicles += 1
35 last = 0
36 # go to customer
37 sum_demand += demand
38 sum_dist += distance[last, customer]
39 last = customer
40 return np.array([-float(vehicles), sum_dist])

42def parse(filename):
43 with open(filename) as csvfile:
44 lines = csvfile.readlines()
45 demands = []
46 coords = []
47 for line in lines:
48 row = line.split()
49 if len(row) < 5 or not row[0][0].isdigit():
50 continue
51 demands.append(float(row[3]))
52 coords.append(np.array([float(row[1]), float(row[2])]))
53 n = len(demands)
54 distances = np.zeros((n,n))
55 for i in range(n):
56 for j in range(n):
57 distances[i,j] = np.linalg.norm(coords[i] - coords[j])
58 return np.array(demands), distances

60class Routing():
61 def __init__(self, filename, capacity):
62 self.capacity = capacity
63 self.demands, self.distance = parse(filename)
64 self.dim = len(self.demands) - 1
65 self.bounds = Bounds([0]*self.dim, [1]*self.dim)
66 self.bounds_capacity = Bounds([40] + [0]*(self.dim), [500] + [1]*self.dim)

68 def fitness(self, x):
69 return fitness_(np.argsort(x), self.distance, self.demands, self.capacity)

71 def fitness_capacity(self, x):
72 y = fitness_(np.argsort(x[1:]), self.distance, self.demands, x[0])
73 return np.array([x[0], y[1]])

103 last = customer
104 print ("vehicle ", vehicles, "tour", tour, "demands", sum_demand, "distance", sum_dist)
