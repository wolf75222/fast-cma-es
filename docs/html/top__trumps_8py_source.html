6# The gbea TopTrump benchmark is a carefully designed real world benchmark.
7# Both its single objective and multi-objective fitness functions reflect the requirements of
8# a real world TopTrump card game designer. Its simulation based tests are efficiently implemented,
9# so that it is possible to compare optimization algorithms investing limited CPU time, specially
10# if parallelization is applied. To do so the socket based interface is replaced by a much simpler
11# ctypes based interface.
12
13# https://github.com/ttusar/coco-gbea
14# https://doi.org/10.5281/zenodo.2594848
15# https://github.com/ttusar/coco-gbea/blob/main/code-experiments/rw-problems/GBEA.md

51def objectives_rw(name, numObjs, function, instance, rep, x):
52 try:
53 x = [round(xi, 0) for xi in x]
54 x = np.array(x)
55 y = np.empty(numObjs)
56 x_p = x.ctypes.data_as(ct.POINTER(ct.c_double))
57 y_p = y.ctypes.data_as(ct.POINTER(ct.c_double))
58
59 evaluate_rw_top_trumps(rep, ct.create_string_buffer(name.encode('utf-8')),
60 numObjs, function, instance, len(x), x_p, y_p)
61
62 return y
63 except Exception as ex:
64 return None

66def bounds_rw(dim, instance):
67 try:
68 lb = np.empty(4)
69 ub = np.empty(4)
70 lb_p = lb.ctypes.data_as(ct.POINTER(ct.c_double))
71 ub_p = ub.ctypes.data_as(ct.POINTER(ct.c_double))
72 rw_top_trumps_bounds(instance, 4, lb_p, ub_p)
73 lb[:] -= (0.5 - 1e-6)
74 ub[:] += (0.5 - 1e-6)
75 lower = np.empty(dim)
76 upper = np.empty(dim)
77 for i in range(dim):
78 lower[i] = lb[i % 4]
79 upper[i] = ub[i % 4]
80 return lower, upper
81 except Exception as ex:
82 return None
