58class fitness(object):
59
60 def __init__(self, dim):
61 self.dim = dim
62 self.best_y = mp.RawValue(ct.c_double, np.inf)
63 self.bounds = Bounds([0]*dim, [1]*dim)
64 self.qd_dim = 2
65 self.qd_bounds = Bounds([10, 0.5], [40, 1.5])
66
67 def __call__(self, X):
68 n = int(self.dim/2)
69 dt = 2*max_time/n
70 dts = X[:n]*dt
71 alphas = X[n:]*2*max_alpha - max_alpha
72 y = np.array([1,0])
73 for i in range(n):
74 y = integrate_C(y, dts[i], alphas[i], 0.1)
75 return abs(y[0])+abs(y[1])

77 def qd_fit(self, x):
78 y = self(x)
79 n = int(self.dim/2)
80 dt = 2*max_time/n
81 dts = x[:n]*dt
82 alphas = x[n:]*2*max_alpha - max_alpha
83 dtsum = np.sum(dts)
84 energy = np.sum(np.multiply(dts, abs(alphas)))
85 b = np.array([dtsum, energy])
86 if y < self.best_y.value:
87 self.best_y.value = y
88 print(f'{y:.3f} { list(b) }')
89 return y, b

109def plot_archive(problem, archive = None):
110 name = 'damp_nd'
111 if archive is None:

113 si = archive.argsort()
