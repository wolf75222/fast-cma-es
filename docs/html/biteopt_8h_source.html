144 void init( CBiteRnd& rnd, const double* const InitParams = NULL,
145 const double InitRadius = 1.0 )
146 {
147 initCommonVars( rnd );
148
149 // Initialize solution vectors randomly, calculate objective function
150 // values of these solutions.
151
152 const double sd = 0.25 * InitRadius;
153 int i;
154 int j;
155
156 if( InitParams == NULL )
157 {
158 for( j = 0; j < PopSize; j++ )
159 {
160 ptype* const p = PopParams[ j ];
161
162 for( i = 0; i < ParamCount; i++ )
163 {
164 p[ i ] = wrapParam( rnd, getGaussianInt(
165 rnd, sd, IntMantMult >> 1 ));
166 }
167 }
168 }
169 else
170 {
171 ptype* const p0 = PopParams[ 0 ];
172
173 for( i = 0; i < ParamCount; i++ )
174 {
175 p0[ i ] = wrapParam( rnd,
176 (ptype) (( InitParams[ i ] - MinValues[ i ]) /
177 DiffValues[ i ]));
178 }
179
180 for( j = 1; j < PopSize; j++ )
181 {
182 ptype* const p = PopParams[ j ];
183
184 for( i = 0; i < ParamCount; i++ )
185 {
186 p[ i ] = wrapParam( rnd,
187 getGaussianInt( rnd, sd, p0[ i ]));
188 }
189 }
190 }
191
192 ParOpt.init( rnd, InitParams, InitRadius );
193 ParOpt2.init( rnd, InitParams, InitRadius );
194 UseParOpt = 0;
195
196 ParOptPop.resetCurPopPos();
197 ParOpt2Pop.resetCurPopPos();
198 OldPop.resetCurPopPos();
199
200 DoInitEvals = true;
201 }

221 int optimize( CBiteRnd& rnd, CBiteOpt* const PushOpt = NULL )
222 {
223 int i;
224
225 if( DoInitEvals )
226 {
227 const ptype* const Params = getCurParams();
228
229 for( i = 0; i < ParamCount; i++ )
230 {
231 NewValues[ i ] = getRealValue( Params, i );
232 }
233
234 NewCost = optcost( NewValues );
235 updateBestCost( NewCost, NewValues,
236 updatePop( NewCost, Params, false ));
237
238 if( CurPopPos == PopSize )
239 {
240 updateCentroid();
241
242 for( i = 0; i < ParPopCount; i++ )
243 {
244 ParPops[ i ] -> copy( *this );
245 }
246
247 DoInitEvals = false;
248 }
249
250 return( 0 );
251 }
252
253 DoEval = true;
254
255 const int SelMethod = select( MethodSel, rnd );
256
257 if( SelMethod == 0 )
258 {
259 generateSol2( rnd );
260 }
261 else
262 if( SelMethod == 1 )
263 {
264 const int SelM1 = select( M1Sel, rnd );
265
266 if( SelM1 == 0 )
267 {
268 const int SelM1A = select( M1ASel, rnd );
269
270 if( SelM1A == 0 )
271 {
272 generateSol2b( rnd );
273 }
274 else
275 if( SelM1A == 1 )
276 {
277 generateSol2c( rnd );
278 }
279 else
280 {
281 generateSol2d( rnd );
282 }
283 }
284 else
285 if( SelM1 == 1 )
286 {
287 if( select( M1BSel, rnd ))
288 {
289 generateSol4( rnd );
290 }
291 else
292 {
293 generateSol5b( rnd );
294 }
295 }
296 else
297 if( SelM1 == 2 )
298 {
299 if( select( M1CSel, rnd ))
300 {
301 generateSol5( rnd );
302 }
303 else
304 {
305 generateSol10( rnd );
306 }
307 }
308 else
309 {
310 generateSol6( rnd );
311 }
312 }
313 else
314 if( SelMethod == 2 )
315 {
316 if( select( M2Sel, rnd ))
317 {
318 generateSol1( rnd );
319 }
320 else
321 {
322 const int SelM2B = select( M2BSel, rnd );
323
324 if( SelM2B == 0 )
325 {
326 generateSol3( rnd );
327 }
328 else
329 if( SelM2B == 1 )
330 {
331 generateSol7( rnd );
332 }
333 else
334 if( SelM2B == 2 )
335 {
336 generateSol8( rnd );
337 }
338 else
339 {
340 generateSol9( rnd );
341 }
342 }
343 }
344 else
345 {
346 generateSolPar( rnd );
347 }
348
349 if( DoEval )
350 {
351 // Evaluate objective function with new parameters, if the
352 // solution was not provided by the parallel optimizer.
353 // Wrap parameter values so that they stay in the [0; 1] range.
354
355 for( i = 0; i < ParamCount; i++ )
356 {
357 TmpParams[ i ] = wrapParam( rnd, TmpParams[ i ]);
358 NewValues[ i ] = getRealValue( TmpParams, i );
359 }
360
361 NewCost = optcost( NewValues );
362 }
363
364 const int p = updatePop( NewCost, TmpParams, true );
365
366 if( p > CurPopSize1 )
367 {
368 // Upper bound cost constraint check failed, reject this solution.
369
370 applySelsDecr( rnd );
371
372 StallCount++;
373
374 if( DoEval && CurPopSize < PopSize )
375 {
376 if( select( PopChangeIncrSel, rnd ))
377 {
378 // Increase population size on fail.
379
380 incrCurPopSize();
381 }
382 }
383 }
384 else
385 {
386 updateBestCost( NewCost, NewValues, p );
387 applySelsIncr( rnd, 1.0 - p * CurPopSizeI );
388
389 StallCount = 0;
390
391 if( rnd.get() < ParamCountI )
392 {
393 OldPop.updatePop( *getObjPtr( PopParams[ CurPopSize1 ]),
394 PopParams[ CurPopSize1 ], false );
395 }
396
397 if( PushOpt != NULL && PushOpt != this &&
398 !PushOpt -> DoInitEvals && p > 0 )
399 {
400 PushOpt -> updatePop( NewCost, TmpParams, true );
401 PushOpt -> updateParPop( NewCost, TmpParams );
402 }
403
404 if( DoEval && CurPopSize > PopSize / 2 )
405 {
406 if( select( PopChangeDecrSel, rnd ))
407 {
408 // Decrease population size on success.
409
410 decrCurPopSize();
411 }
412 }
413 }
414
415 // "Diverging populations" technique.
416
417 updateParPop( NewCost, TmpParams );
418
419 return( StallCount );
420 }

598 void generateSol1( CBiteRnd& rnd )
599 {
600 ptype* const Params = TmpParams;
601
602 const CBitePop& ParPop = selectParPop( 0, rnd );
603
604 copyParams( Params, ParPop.getParamsOrdered(
605 getMinSolIndex( 0, rnd, ParPop.getCurPopSize() )));
606
607 // Select a single random parameter or all parameters for further
608 // operations.
609
610 int a;
611 int b;
612 bool DoAllp = false;
613
614 if( rnd.get() < 1.8 * ParamCountI )
615 {
616 if( select( Gen1AllpSel, rnd ))
617 {
618 DoAllp = true;
619 }
620 }
621
622 if( DoAllp )
623 {
624 a = 0;
625 b = ParamCount;
626 }
627 else
628 {
629 a = rnd.getInt( ParamCount );
630 b = a + 1;
631 }
632
633 // Bitmask inversion operation, works as the main "driver" of
634 // optimization process.
635
636 const double r1 = rnd.get();
637 const double r12 = r1 * r1;
638 const int ims = (int) ( r12 * r12 * 48.0 );
639 const ptype imask = (ptype) ( IntMantMask >> ims );
640
641 const int im2s = rnd.getSqrInt( 96 );
642 const ptype imask2 = (ptype) ( im2s > 63 ? 0 : IntMantMask >> im2s );
643
644 const int si1 = (int) ( r1 * r12 * ParPop.getCurPopSize() );
645 const ptype* const rp1 = ParPop.getParamsOrdered( si1 );
646 int i;
647
648 for( i = a; i < b; i++ )
649 {
650 Params[ i ] = (( Params[ i ] ^ imask ) +
651 ( rp1[ i ] ^ imask2 )) >> 1;
652 }
653
654 if( rnd.get() < 1.0 - ParamCountI )
655 {
656 const ptype* const rp2 = getParamsOrdered(
657 rnd.getSqrInt( CurPopSize ));
658
659 if( rnd.get() < sqrt( ParamCountI ))
660 {
661 if( select( Gen1MoveAsyncSel, rnd ))
662 {
663 a = 0;
664 b = ParamCount;
665 }
666 }
667
668 // Random move around a random previous solution vector.
669
670 static const double SpanMults[ 4 ] = { 0.5, 1.5, 2.0, 2.5 };
671
672 const double m = SpanMults[ select( Gen1MoveSpanSel, rnd )];
673 const double m1 = rnd.getTPDF() * m;
674 const double m2 = rnd.getTPDF() * m;
675
676 for( i = a; i < b; i++ )
677 {
678 Params[ i ] += (ptype) (( rp2[ i ] - Params[ i ]) * m1 );
679 Params[ i ] += (ptype) (( rp2[ i ] - Params[ i ]) * m2 );
680 }
681 }

690 void generateSol2( CBiteRnd& rnd )
691 {
692 ptype* const Params = TmpParams;
693
694 const int si1 = getMinSolIndex( 1, rnd, CurPopSize );
695 const ptype* const rp1 = getParamsOrdered( si1 );
696 const ptype* const rp3 = getParamsOrdered( CurPopSize1 - si1 );
697
698 const int si2 = 1 + rnd.getInt( CurPopSize1 );
699 const ptype* const rp2 = getParamsOrdered( si2 );
700
701 const int si4 = rnd.getSqrInt( CurPopSize );
702 const ptype* const rp4 = getParamsOrdered( si4 );
703 const ptype* const rp5 = getParamsOrdered( CurPopSize1 - si4 );
704
705 // The "step in the right direction" (Differential Evolution
706 // "mutation") operation towards the best (minimal) and away from
707 // the worst (maximal) parameter vector, plus a difference of two
708 // random vectors.
709
710 const int Mode = select( Gen2ModeSel, rnd );
711 int i;
712
713 if( Mode == 0 )
714 {
715 for( i = 0; i < ParamCount; i++ )
716 {
717 Params[ i ] = rp1[ i ] + ((( rp2[ i ] - rp3[ i ]) +
718 ( rp4[ i ] - rp5[ i ])) >> 1 );
719 }
720 }
721 else
722 {
723 const ptype* const rp1b = getParamsOrdered(
724 rnd.getSqrInt( CurPopSize ));
725
726 for( i = 0; i < ParamCount; i++ )
727 {
728 Params[ i ] = (( rp1[ i ] + rp1b[ i ]) +
729 ( rp2[ i ] - rp3[ i ]) + ( rp4[ i ] - rp5[ i ])) >> 1;
730 }
731 }

786 void generateSol2c( CBiteRnd& rnd )
787 {
788 ptype* const Params = TmpParams;
789 zeroParams( Params );
790
791 const int si1 = rnd.getPowInt( 4.0, CurPopSize / 2 );
792 const ptype* const rp1 = getParamsOrdered( si1 );
793
794 const int PairCount = 3;
795 const int pc = 1 + 2 * PairCount;
796 int PopIdx[ pc ];
797 PopIdx[ 0 ] = si1;
798
799 int pp = 1;
800 int i;
801 int j;
802
803 if( CurPopSize1 <= pc )
804 {
805 while( pp < pc )
806 {
807 PopIdx[ pp ] = rnd.getInt( CurPopSize );
808 pp++;
809 }
810 }
811 else
812 {
813 while( pp < pc )
814 {
815 const int sii = rnd.getInt( CurPopSize );
816
817 for( j = 0; j < pp; j++ )
818 {
819 if( PopIdx[ j ] == sii )
820 {
821 break;
822 }
823 }
824
825 if( j == pp )
826 {
827 PopIdx[ pp ] = sii;
828 pp++;
829 }
830 }
831 }
832
833 const ptype* const rp2 = getParamsOrdered( PopIdx[ 1 ]);
834 const ptype* const rp3 = getParamsOrdered( PopIdx[ 2 ]);
835 const ptype* const rp4 = getParamsOrdered( PopIdx[ 3 ]);
836 const ptype* const rp5 = getParamsOrdered( PopIdx[ 4 ]);
837 const ptype* const rp6 = getParamsOrdered( PopIdx[ 5 ]);
838 const ptype* const rp7 = getParamsOrdered( PopIdx[ 6 ]);
839
840 for( i = 0; i < ParamCount; i++ )
841 {
842 Params[ i ] = ( rp2[ i ] - rp3[ i ]) + ( rp4[ i ] - rp5[ i ]) +
843 ( rp6[ i ] - rp7[ i ]);
844 }
845
846 if( rnd.getBit() && rnd.getBit() )
847 {
848 const int k = rnd.getInt( ParamCount );
849
850 // Produce sparsely-random bit-strings.
851
852 const ptype v1 = (ptype) ( rnd.getRaw() & rnd.getRaw() &
853 rnd.getRaw() & rnd.getRaw() & rnd.getRaw() & IntMantMask );
854
855 const ptype v2 = (ptype) ( rnd.getRaw() & rnd.getRaw() &
856 rnd.getRaw() & rnd.getRaw() & rnd.getRaw() & IntMantMask );
857
858 Params[ k ] += v1 - v2; // Apply in TPDF manner.
859 }
860
861 const int Mode = select( Gen2cModeSel, rnd );
862
863 if( Mode == 0 )
864 {
865 int si2 = si1 + rnd.getBit() * 2 - 1;
866
867 if( si2 < 0 )
868 {
869 si2 = 1;
870 }
871
872 const ptype* const rp1b = getParamsOrdered( si2 );
873
874 for( i = 0; i < ParamCount; i++ )
875 {
876 Params[ i ] = ( rp1[ i ] + rp1b[ i ] + Params[ i ]) >> 1;
877 }
878 }
879 else
880 {
881 for( i = 0; i < ParamCount; i++ )
882 {
883 Params[ i ] = rp1[ i ] + ( Params[ i ] >> 1 );
884 }
885 }
886 }
