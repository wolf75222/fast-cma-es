142 int optimize( CBiteRnd& rnd, double* const OutCost = NULL,
143 double* const OutValues = NULL )
144 {
145 double* const Params = getCurParams();
146 int i;
147
148 if( DoCentEval )
149 {
150 DoCentEval = false;
151
152 for( i = 0; i < ParamCount; i++ )
153 {
154 Params[ i ] = CentParams[ i ];
155 NewValues[ i ] = getRealValue( CentParams, i );
156 }
157 }
158 else
159 {
160 double s2 = 1e-300;
161
162 for( i = 0; i < ParamCount; i++ )
163 {
164 Params[ i ] = rnd.get() - 0.5;
165 s2 += Params[ i ] * Params[ i ];
166 }
167
168 const double d = Radius / sqrt( s2 );
169
170 if( ParamCount > 4 )
171 {
172 for( i = 0; i < ParamCount; i++ )
173 {
174 Params[ i ] = wrapParam( rnd,
175 CentParams[ i ] + Params[ i ] * d );
176
177 NewValues[ i ] = getRealValue( Params, i );
178 }
179 }
180 else
181 {
182 for( i = 0; i < ParamCount; i++ )
183 {
184 const double m = JitOffs + rnd.get() * JitMult;
185
186 Params[ i ] = wrapParam( rnd,
187 CentParams[ i ] + Params[ i ] * d * m );
188
189 NewValues[ i ] = getRealValue( Params, i );
190 }
191 }
192 }
193
194 const double NewCost = optcost( NewValues );
195
196 if( OutCost != NULL )
197 {
198 *OutCost = NewCost;
199 }
200
201 if( OutValues != NULL )
202 {
203 copyValues( OutValues, NewValues );
204 }
205
206 updatePop( NewCost, Params, false );
207 updateBestCost( NewCost, NewValues );
208
209 AvgCost += NewCost;
210 cure++;
211
212 if( cure >= curem )
213 {
214 AvgCost /= cure;
215
216 if( AvgCost < HiBound )
217 {
218 HiBound = AvgCost;
219 StallCount = 0;
220
221 applySelsIncr( rnd );
222 }
223 else
224 {
225 StallCount += cure;
226
227 applySelsDecr( rnd );
228 }
229
230 resetCurPopPos();
231 AvgCost = 0.0;
232 cure = 0;
233
234 update( rnd );
235
236 curem = (int) ceil( CurPopSize * EvalFac );
237 }
238
239 return( StallCount );
240 }

281 void update( CBiteRnd& rnd )
282 {
283 static const double WCent[ 4 ] = { 4.5, 6.0, 7.5, 10.0 };
284 static const double WRad[ 4 ] = { 14.0, 16.0, 18.0, 20.0 };
285 static const double EvalFacs[ 3 ] = { 2.1, 2.0, 1.9 };
286
287 const double CentFac = WCent[ select( CentPowSel, rnd )];
288 const double RadFac = WRad[ select( RadPowSel, rnd )];
289 EvalFac = EvalFacs[ select( EvalFacSel, rnd )];
290
291 const double lm = 1.0 / curem;
292 double s1 = 0.0;
293 double s2 = 0.0;
294 int i;
295
296 for( i = 0; i < CurPopSize; i++ )
297 {
298 const double l = 1.0 - i * lm;
299
300 const double v1 = pow( l, CentFac );
301 WPopCent[ i ] = v1;
302 s1 += v1;
303
304 const double v2 = pow( l, RadFac );
305 WPopRad[ i ] = v2;
306 s2 += v2;
307 }
308
309 s1 = 1.0 / s1;
310 s2 = 1.0 / s2;
311
312 const double* ip = PopParams[ 0 ];
313 double* const cp = CentParams;
314 const double* const wc = WPopCent;
315 double w = wc[ 0 ] * s1;
316
317 for( i = 0; i < ParamCount; i++ )
318 {
319 cp[ i ] = ip[ i ] * w;
320 }
321
322 int j;
323
324 for( j = 1; j < CurPopSize; j++ )
325 {
326 ip = PopParams[ j ];
327 w = wc[ j ] * s1;
328
329 for( i = 0; i < ParamCount; i++ )
330 {
331 cp[ i ] += ip[ i ] * w;
332 }
333 }
334
335 const double* const rc = WPopRad;
336 Radius = 0.0;
337
338 for( j = 0; j < CurPopSize; j++ )
339 {
340 ip = PopParams[ j ];
341 double s = 0.0;
342
343 for( i = 0; i < ParamCount; i++ )
344 {
345 const double d = ip[ i ] - cp[ i ];
346 s += d * d;
347 }
348
349 Radius += s * rc[ j ];
350 }
351
352 Radius = sqrt( Radius * s2 );
353 }
354};
