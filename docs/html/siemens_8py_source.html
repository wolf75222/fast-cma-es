30class Machine:
31
32 def __init__(self, type, dtime):
33 self.product = -1
34 self.last_product = -1
35 self.time = 0
36 self.type = type
37 self.dtime = dtime

39 def process(self, time, product):
40 #print(f'process p{product} {self}')
41 time = max(time, self.time)
42 if product != self.last_product:
43 time += setup_time
44 self.product = product
45 self.time = time + self.dtime[self.type, product]

47 def take(self):
48 #print(f'take {self}')
49 self.last_product = self.product
50 self.product = -1
51 return self.last_product

53 def finished(self, time):
54 return time >= self.time and self.product >= 0

56 def available(self, time):
57 return time >= self.time and self.product < 0

59 def __str__(self):

96 def reset(self):
97 self.machine1 = Machine(0, self.dtime)
98 self.machine2 = Machine(1, self.dtime)
99 self.buffer = Buffer()
100 self.count_finished = 0
101 self.time = 0
102 self.product = -1
103 self.to_process = 0 # switch to product 1 at the beginning

105 def __str__(self):
106 return f'{self.time} {self.machine1} {self.machine2} {self.buffer} {self.count_finished}'
107
108 def next_product(self):
109 #return 1
110 if self.to_process > 0:
111 self.to_process -= 1
112 else:
113 # switch product
114 self.product = (self.product + 1) % self.dim
115 self.to_process = self.batch_size[self.product]
116
117 return self.product

119 def tick(self):
120 #print(self)
121 # from machine 1 to buffer
122 if self.machine1.finished(self.time):
123 if self.buffer.can_add():
124 self.buffer.add(self.machine1.take())
125
126 # from batch to machine1
127 if self.machine1.available(self.time):
128 self.machine1.process(self.time, self.next_product())
129
130 # take from machine 2 if ready
131 if self.machine2.finished(self.time):
132 product = self.machine2.take()
133 self.count_finished += 1
134 #print(f'finished {product} {self.count_finished}')
135
136 # from buffer to machine2
137 if self.machine2.available(self.time) and self.buffer.can_retrieve():
138 self.machine2.process(self.time, self.buffer.retrieve())
139
140 self.time += 1

142 def simulate(self, all_time, batch_size):
143 self.reset()
144 self.batch_size = batch_size
145 while self.time < all_time:
146 self.tick()
147 return self.count_finished
