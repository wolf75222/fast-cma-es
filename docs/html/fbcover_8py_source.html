52def circles_arr(circs):
53 max_circ = max([len(c) for c in circs])
54 acircs = np.empty((len(circs), max_circ), dtype=int)
55 for c in range(len(circs)):
56 circ = circs[c]
57 for i in range(len(circ)):
58 acircs[c, i] = circ[i]
59 return acircs

75def parse_fb(edge_file, circle_file):
76 lines = open(edge_file, 'r').readlines()
77 source = []
78 target = []
79 vmap = {}
80 for line in lines:
81 vs = line.split()
82 source.append(add_vertice(vs[0], vmap))
83 target.append(add_vertice(vs[1], vmap))
84 lines = open(circle_file, 'r').readlines()
85 circs = []
86 for line in lines:
87 vs = line.split()[1:]
88 if len(vs) > 1: #drop circles of 1
89 for v in vs: # add circ nodes to vmap
90 add_vertice(v, vmap)
91 circs.append([vmap[v] for v in vs])
92 return fb_graph(vmap, np.array(source, dtype=int), np.array(target, dtype=int),
93 circs)

95@njit(fastmath=True)
96def fb_covered(source, target, circles, circle_size, nds):
97 num = 0
98 nedges = len(source)
99 for i in range(nedges): # check all edges
100 if nds[source[i]] or nds[target[i]]:
101 num += 1
102 for i in range(len(circle_size)):
103 circle = circles[i] # check all circle pairs
104 size = circle_size[i]
105 for j in range(size-1):
106 for k in range(j+1, size): # add if both are covered
107 if nds[circle[j]] and nds[circle[k]]:
108 num += 1.0/math.sqrt(size)
109 return num

118class problem_fb():
119
120 def __init__(self, g):
121 self.dim = g.nnodes
122 self.bounds = Bounds([0]*self.dim, [1.99999]*self.dim)
123 self.g = g
124 self.best_y = mp.RawValue(ct.c_double, np.inf)
125 self.max_cost, self.max_cov = self.cost(np.array([1]*self.dim))
126
127 def cost(self, x):
128 nds = nodes(x.astype(int))
129 cov = fb_covered(self.g.source, self.g.target, self.g.acircles,
130 self.g.circle_size, nds)
131 cost = sum_weights(nds, self.g.weights)
132 return cost, cov
133
134 def fitness(self, x):
135 cost, cov = self.cost(x)
136 cost /= self.max_cost # to be minimized
137 cov /= -self.max_cov # to be maximized
138 return [cost, cov]
139
140 def fitness_so(self, x):
141 cost, cov = self.cost(x)
142 cost /= self.max_cost # to be minimized
143 cov /= -self.max_cov # to be maximized
144 cost = max(0.3, cost) # target 30% cost
145 y = 2*cost + cov
146 if y < self.best_y.value:
147 self.best_y.value = y
148 nds = nodes(x.astype(int))
149 print("n,cov", cost, cov, num_true(nds), len(nds))
150 return y
