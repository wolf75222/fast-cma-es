41class Wrapper(object):
42 """Wrapper for parallel computation with shared state.
43
44 This class acts as a wrapper to provide thread/process-safe management
45 of shared variables during multiprocessing operations. It enables
46 evaluation of a function and maintains the best result (minimum value
47 of function output) along with a counter of evaluations.
48
49 Attributes:
50 func (Callable): The function to be evaluated.
51 best_x (mp.RawArray): Shared memory array to store the inputs
52 corresponding to the best function output.
53 best_y (mp.RawValue): Shared memory value to store the best function
54 output observed.
55 count (mp.RawValue): Shared memory counter to track the number
56 of function evaluations.
57 """
58
59 def __init__(self, fun, dim):
60 """
61 Initializes the instance with function to optimize and dimension of the problem.
62
63 Args:
64 fun: Callable function to optimize.
65 dim: int. The dimensionality of the optimization problem.
66
67 Attributes:
68 func: Callable. The function to optimize.
69 statMutex: multiprocessing.Lock. A lock to ensure thread safety.
70 best_x: multiprocessing.Array. Shared array storing the best optimization
71 solution found so far.
72 best_y: multiprocessing.Value. Shared value storing the best objective
73 value found so far.
74 count: multiprocessing.Value. Shared value counting the number of function
75 evaluations.
76 """
77 self.func = fun
78 self.statMutex = mp.Lock()
79 self.best_x = mp.RawArray(ct.c_double, dim)
80 self.best_y = mp.RawValue(ct.c_double, sys.float_info.max)
81 self.count = mp.RawValue(ct.c_int, 0)

83 def eval(self, x):
84 """
85 Evaluates a given input `x` using the function `func`, updates the best observed
86 value and corresponding input if applicable, and increments the evaluation
87 counter.
88
89 Args:
90 x: Input data to be evaluated by the function `func`.
91
92 Returns:
93 The result of evaluating `x` using the function `func`.
94 """
95 y = self.func(x)
96 with self.statMutex:
97 if y < self.best_y.value:
98 self.best_y.value = y
99 self.best_x[:] = x[:]
100 #print(str(self.count.value) + " " + str(y) + " " + str(x))
101 self.count.value += 1
102 return y

324def _rosen(xs, alpha=1e2):
325 """
326 Computes the Rosenbrock function for the given input.
327
328 The Rosenbrock function is a non-convex mathematical function often used
329 as a performance test problem for optimization algorithms. This function
330 takes a list of numerical inputs and evaluates the Rosenbrock polynomial
331 for those inputs. It also supports handling scalars and will return outputs
332 as either a scalar or a list depending on the input.
333
334 Args:
335 xs: A numerical iterable or scalar. The input values for which the
336 Rosenbrock function is to be computed. If the first element of `xs`
337 is scalar, it is treated as a single value.
338 alpha: An optional numerical value. Represents the parameter to scale
339 the quadratic term in the Rosenbrock function. Defaults to 1e2.
340
341 Returns:
342 float or list: The evaluated Rosenbrock function result. If the input
343 is a scalar or a single list, it returns a scalar result. If the input
344 is a list of multiple numerical iterables, it returns a list of computed
345 results for each input iterable.
346 """
347 xs = [xs] if np.isscalar(xs[0]) else xs
348 xs = np.asarray(xs)
349 f = [sum(alpha * (x[:-1]**2 - x[1:])**2 + (1. - x[:-1])**2) for x in xs]
350 return f if len(f) > 1 else f[0] # 1-element-list into scalar

352def _rastrigin(x):
353 """
354 Computes the Rastrigin function value for a given input vector.
355
356 The Rastrigin function is commonly used as a non-convex test function for
357 optimization algorithms. It is highly multimodal, meaning it has many local
358 minima, which makes it a challenging benchmark for optimization techniques.
359
360 Args:
361 x (array-like): A vector of input values on which the Rastrigin function
362 is evaluated. Must be convertible to a NumPy array.
363
364 Returns:
365 float: The computed value of the Rastrigin function for the input vector.
366 """
367 dim = len(x)
368 x = np.asarray(x)
369 return 10.0*dim + sum(x*x - 10.0*np.cos(2.0*math.pi*x))

371def _cigar(x):
372 """

374
375 The "cigar" function is a common benchmark function in optimization studies.
376 It is designed to simulate an elliptical-shaped function with one dominant
377 dimension. It is used to evaluate optimization algorithms in high-dimensional

384 Returns:
385 float: Calculated "cigar" function value for the given input array.

408def _elli(x):
409 """
410 Calculates the elliptic function value for the given input.
411
412 The elliptic function is commonly used in optimization problems as a
413 benchmark function. It is computed using the input array and a factor
414 that exponentially grows based on the dimension of the input.
415
416 Args:
417 x (array-like): Input array representing the variables for which
418 the elliptic function is calculated.
419
420 Returns:
421 float: The computed value of the elliptic function for the input.
422 """
423 dim = len(x)
424 x = np.asarray(x)
425 factor = 1E6
426 f = 0
427 for i in range(dim):
428 f += factor ** (i / (dim - 1.)) * x[i] * x[i]
429 return f

450def _rastrigin_mean(x, n):
451 """
452 Computes the mean output of the Rastrigin function over `n` modifications of
453 input `x`. This function applies a slight modification to the input value `x`
454 before passing it to `_rastrigin` and accumulates the results to compute
455 the mean.
456
457 Args:
458 x: The initial input value for the Rastrigin function that will be slightly
459 modified for each iteration.
460 n: The number of iterations or modifications of input `x` for computing
461 the average.
462
463 Returns:
464 The mean result of the Rastrigin function after applying it to `n`
465 modified versions of the input `x`.
466 """
467 delta = 0.001
468 sumy = 0
469 for i in range(n):
470 sumy += _rastrigin(_modify(x, delta))
471 return sumy / n
