67def initial_solution(g):
68 solution = list(g.nodes())
69 # sort nodes for degree, low degree has better chance not to uncover an edge
70 for _, node in \
71 sorted(list(zip(list(dict(g.degree(solution)).values()), solution))):
72 remove = True
73 for neigbor in g.neighbors(node): # all neighbors covered?
74 if neigbor not in solution:
75 remove = False # bad luck, would uncover an edge
76 if remove:
77 solution.remove(node)
78 return solution

80def remove_node(g, solution, mean, start_time, max_time):
81 solution = solution.copy()
82 uncovered = []
83 while len(uncovered) == 0:
84 to_delete = random.choice(solution)
85 for neighbor in g.neighbors(to_delete):
86 if neighbor not in solution:
87 uncovered.append(neighbor)
88 uncovered.append(to_delete)
89 solution.remove(to_delete)
90 i = 0
91 max_i = mean * 10
92 while len(uncovered) > 0 and i < max_i and \
93 time.time() - start_time < max_time:
94 i += 1
95 # delete node from solution
96 next_solution = solution.copy()
97 next_uncovered = uncovered.copy()
98 to_delete = random.choice(solution)
99 solution.remove(to_delete)
100 for neighbor in g.neighbors(to_delete):
101 if neighbor not in solution:
102 uncovered.append(neighbor)
103 uncovered.append(to_delete)
104 # add node to solution
105 to_add = random.choice(uncovered)
106 solution.append(to_add)
107 for neighbor in g.neighbors(to_add):
108 if neighbor not in solution:
109 uncovered.remove(neighbor)
110 uncovered.remove(to_add)
111 # update solution if uncovered shrink
112 if len(next_uncovered) < len(uncovered) or \
113 (len(next_uncovered) == len(uncovered) and \
114 i > mean and random.random() < 1.0/mean):
115 solution = next_solution.copy()
116 uncovered = next_uncovered.copy()
117 return solution, uncovered, i

119def solve_greedy(g, seed, max_time):
120 print("seed", seed)
121 random.seed(seed)
122 start_time = time.time()
123 solution = initial_solution(g)
124 iters = []
125 mean = 10000
126 while time.time() - start_time < max_time:
127 next_solution, uncovered, i = remove_node(g, solution, mean, start_time, max_time)
128 iters.append(i)
129 mean = np.mean(iters)
130 if len(uncovered) == 0: # all covered ?
131 solution = next_solution
132 print(round(time.time()-start_time,3), len(solution), i, int(mean))
133
134 print(round(time.time()-start_time,3), len(solution))
135 print('Solution: ({}) {}'.format(
136 len(solution), solution))
137 return solution

168@njit(fastmath=True)
169def num_true(nodes):
170 n = 0
171 for b in nodes:
172 if b:
173 n += 1
174 return n
175
176@njit(fastmath=True)
177def numba_covered(source, target, nodes):
178 num = 0
179 nedges = len(source)
180 for i in range(nedges):
181 if nodes[source[i]] or nodes[target[i]]:
182 num += 1
183 return nedges - num
184
185def not_covered(g, nodes):
186 return numba_covered(g.source, g.target, nodes)
187
188@njit(fastmath=True)
189def nodes(x):
190 nds = np.empty(len(x), dtype=numba.boolean)
191 for i in range(len(x)):
192 nds[i] = True if x[i] > 0.5 else False
193 return nds

202class problem():
203
204 def __init__(self, g):
205 self.dim = len(g.nodes())
206 self.bounds = Bounds([0]*self.dim, [1.999999999999]*self.dim)
207 self.g = graph(g)
208 self.best_n = mp.RawValue(ct.c_double, np.inf)

210 def fitness(self, x):
211 nds = nodes(x.astype(int))
212 ncov = not_covered(self.g, nds)
213 n = num_true(nds)
214 return n + 2*ncov
