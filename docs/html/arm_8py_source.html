6"""
7Basic example for quality diversity optimization:
8
9A planar robotic arm consists of dim+1 = 21 segments with dim=20 joints.
10
11For each reachable position minimize either:
12
13- the standard deviation of the joint-angles
14- the sum of the joint-angles
15- the maximal joint-angle
16
17Since all results are stored in the ND archive finally we know for all target positions
18how to adjust all the joint-angles to meet the objective without any further optimization.
19
20On a modern 16 core CPU the whole optimization takes about 30 seconds.
21
22Play around with the parameters of the experiment:
23
24- select different opt_params (optimization algorithms + their configurations)
25- select different objectives
26- change the dimension dim - the number of joint-angles
27- in mapelites.wrapper set optional parameter workers=1 to switch to single thread execution
28
29and check the resulting plot.
30"""

48def forward_kinematics(q):
49 """
50 Calculates links positions given joint angles
51 Parameters
52 ----------
53 q : numpy.ndarray
54 (n_links,) array of angles in radians
55
56 adapted from https://github.com/tsitsimis/planar-robot/blob/master/planarobot/planar_arm.py
57 """
58 n_links = len(q) + 1
59 base_pos = (0,0)
60 links = np.full(n_links, 1)
61 pos = np.zeros((n_links, 2))
62 pos[0, :] = base_pos + np.array([[links[0] * np.cos(q[0]), links[0] * np.sin(q[0])]])
63 for i in range(1, n_links):
64 delta_pos = np.array([links[i] * np.cos(np.sum(q[0:i+1])),
65 links[i] * np.sin(np.sum(q[0:i+1]))])
66 pos[i, :] = pos[i - 1, :] + delta_pos
67 return pos

69class fitness(object):
70
71 def __init__(self, dim):
72 self.dim = dim
73 self.best_y = mp.RawValue(ct.c_double, np.inf)
74 self.bounds = Bounds([-np.pi]*dim, [np.pi]*dim)
75 self.qd_dim = 2
76 self.qd_bounds = Bounds([-dim-1, -dim-1], [dim+1, dim+1])
77
78 def __call__(self, X):
79 pos = forward_kinematics(X)
80 #y = np.amax(np.abs(X))
81 #y = np.sum(np.abs(X))
82 y = np.std(X)
83 b = pos[-1]
84 if y < self.best_y.value:
85 self.best_y.value = y
86 print(f'{y:.3f} { list(b) }')
87 return y, b

104def plot_archive(problem, archive = None):
105 name = 'arm_nd'
106 if archive is None:
107 archive = mapelites.load_archive(name, problem.bounds, problem.qd_bounds)
108 si = archive.argsort()
109 ysp = []
110 beh = archive.get_ds()[si]
111 ys = archive.get_ys()[si]
112 lb = problem.qd_bounds.lb
113 ub = problem.qd_bounds.ub
114 for i in range(len(si)):
115 # if ys[i] < 5.0: # use for y = np.sum(np.abs(X))
116 if ys[i] < 0.2: # use for y = np.std(X)
117 b = beh[i]
118 if np.any(np.greater(b, ub)) or np.any(np.greater(lb, b)):
119 continue
120 ysp.append([b[0], b[1], ys[i]])
121 ysp = np.array(ysp)
122 plot3d(ysp, name, 'x', 'y', 'objective')

124def optimize_qd(dim):
125 problem = fitness(dim)
126 name = 'arm_nd'
127 opt_params0 = {'solver':'elites', 'popsize':512}
128 opt_params1 = {'solver':'CRMFNES_CPP', 'max_evals':4000, 'popsize':32, 'stall_criterion':3}
129 #opt_params1 = {'solver':'CRMFNES_CPP', 'max_evals':4000, 'popsize':32, 'sigma':0.2, 'stall_criterion':3}
130 opt_params2 = {'solver':'DE_CPP', 'max_evals':4000, 'popsize':32, 'stall_criterion':3}
131 opt_params3 = {'solver':'CMA_CPP', 'max_evals':4000, 'popsize':32, 'stall_criterion':3}
132 archive = diversifier.minimize(
133 mapelites.wrapper(problem, problem.qd_dim, interval=200000, save_interval=12000000),
134 problem.bounds, problem.qd_bounds, opt_params=[opt_params0, opt_params3], max_evals=3000000)
135 print('final archive:', archive.info())
136 archive.save(name)
137 plot_archive(problem, archive)
