28static vec sequence(double start, double end, double step) {
29 int size = (int) ((end - start) / step + 1);
30 vec d(size);
31 double value = start;
32 for (int r = 0; r < size; r++) {
33 d(r) = value;
34 value += step;
35 }
36 return d;
37}

43 CrfmnesOptimizer(int64_t runid_, Fitness* fitfun_, int dim_, vec m_, double sigma_, int lamb_,
44 int maxEvaluations_, double stopfitness_,
45 double penalty_coef_, bool use_constraint_violation_, int64_t seed) {
46
47 runid = runid_;
48 fitfun = fitfun_;
49 dim = dim_;
50 m = fitfun->encode(m_);
51 sigma = sigma_;
52 lamb = lamb_;
53 mu = lamb_/2;
54 maxEvaluations = maxEvaluations_;
55 stopfitness = stopfitness_;
56 penalty_coef = penalty_coef_ > 0 ? penalty_coef_ : 1e5;
57 use_constraint_violation = use_constraint_violation_;
58 rs = new pcg64(seed);
59
60 stop = 0;
61 v = normalVec(dim, *rs) / sqrt(dim);
62 D = constant(dim, 1);
63 w_rank_hat = ((log(sequence(1, lamb, 1).array()) * -1.) + log(mu + 1)).cwiseMax(0);
64 w_rank = (w_rank_hat / w_rank_hat.sum()).array() - (1. / lamb);
65 vec wlamb = w_rank.array() + (1. / lamb);
66 mueff = 1. / (wlamb.transpose() * wlamb)(0,0);
67 cs = (mueff + 2.) / (dim + mueff + 5.);
68 cc = (4. + mueff / dim) / (dim + 4. + 2. * mueff / dim);
69 c1_cma = 2. / (pow(dim + 1.3, 2) + mueff);
70 // initialization
71 chiN = sqrt(dim) * (1. - 1. / (4. * dim) + 1. / (21. * dim * dim));
72 pc = zeros(dim);
73 ps = zeros(dim);
74 // distance weight parameter
75 h_inv = get_h_inv(dim);
76 // learning rate
77 eta_m = 1.0;
78 eta_move_sigma = 1.;
79
80 g = 0;
81 no_of_evals = 0;
82 z = mat(dim, lamb);
83 f_best = INFINITY;
84 x_best = vec(dim);
85 };

91 void doOptimize() {
92 // -------------------- Generation Loop --------------------------------
93 for (iterations = 1; fitfun->evaluations() < maxEvaluations && !fitfun->terminate();
94 iterations++) {
95 // generate and evaluate lamb offspring
96 try {
97 mat xs = ask();
98 vec evs(lamb);
99 fitfun->values(xs, evs);
100 tell(evs);
101 } catch (std::exception &e) {
102 stop = -1;
103 }
104 if (stop != 0)
105 return;
106 }

141 mat ask() {
142 mat zhalf = normal(dim, mu, *rs);
143 for (int i = 0; i < lamb; i++) {
144 if (i < mu) z.col(i) = zhalf.col(i);
145 else z.col(i) = -zhalf.col(i-mu);
146 }
147 normv = v.norm();
148 normv2 = normv*normv;
149 vbar = v / normv;
150 y = z + ((sqrt(1 + normv2) - 1.) * (vbar * (vbar.transpose() * z)));
151 x = ((sigma * y.array()).colwise() * D.array()).colwise() + m.array();
152 return x;
153 }
154
155 void tell(vec &evs) {
156 evals_no_sort = vec(evs);
157 for (int k = 0; k < lamb; k++) {
158 if (!isfinite(evals_no_sort[k]))
159 evals_no_sort[k] = DBL_MAX;
160 }
161 xs_no_sort = mat(x);
162 ivec sorted_indices;
163 if (use_constraint_violation) {
164 vec violations = fitfun->violations(x, penalty_coef);
165 sorted_indices = sort_indices_by(evals_no_sort + violations, z);
166 } else
167 sorted_indices = sort_indices_by(evals_no_sort, z);
168 int best_eval_id = sorted_indices[0];
169 double f_best_ = evals_no_sort[best_eval_id];
170
171 z = mat(z)(Eigen::indexing::all, sorted_indices);
172 y = mat(y)(Eigen::indexing::all, sorted_indices);
173 x = mat(x)(Eigen::indexing::all, sorted_indices);
174 no_of_evals += lamb;
175 g += 1;
176 if (f_best_ < f_best) {
177 f_best = f_best_;
178 x_best = fitfun->decode(xs_no_sort.col(best_eval_id));
179 if (f_best < stopfitness)
180 stop = 1;
181 //cout << f_best << endl;
182 }
183 // This operation assumes that if the solution is infeasible, infinity comes in as input.
184 double lambF = 0;
185 for (int k = 0; k < lamb; k++)
186 if (evals_no_sort[k] < DBL_MAX) lambF++;
187 // evolution path p_sigma
188 ps = (1 - cs) * ps + sqrt(cs * (2. - cs) * mueff) * (z * w_rank);
189 double ps_norm = ps.norm();
190 // distance weight
191 vec w_tmp(lamb);
192 for (int k = 0; k < lamb; k++)
193 w_tmp[k] = w_rank_hat[k] * w_dist_hat(z.col(k), lambF);
194 vec weights_dist = (w_tmp / w_tmp.sum()).array() - 1. / lamb;
195 // switching weights and learning rate
196 vec weights = ps_norm >= chiN ? weights_dist : w_rank;
197 double eta_sigma = ps_norm >= chiN ? eta_move_sigma :
198 (ps_norm >= 0.1 * chiN ? eta_stag_sigma(lambF) : eta_conv_sigma(lambF));
199 // update pc, m
200 vec wxm = (x.array().colwise() - m.array()).matrix() * weights;
201 pc = (1. - cc) * pc + sqrt(cc * (2. - cc) * mueff) * wxm / sigma;
202 m += eta_m * wxm;
203 // calculate s, t
204 // step1
205 double normv4 = normv2 * normv2;
206 mat exY(dim, lamb+1);
207 for (int k = 0; k < lamb; k++)
208 exY.col(k) = y.col(k);
209 exY.col(lamb) = pc.array() / D.array();
210 mat yy = exY.array() * exY.array(); // dim x lamb+1
211 vec ip_yvbar = vbar.transpose() * exY;
212 mat yvbar = exY.array().colwise() * vbar.array(); // dim x lamb+1. exYvbar
213 double gammav = 1. + normv2;
214 vec vbarbar = vbar.array() * vbar.array();
215 double alphavd = min(1., sqrt(normv4 + (2 * gammav - sqrt(gammav)) / vbarbar.maxCoeff()) / (2. + normv2)); // scalar
216 mat vbar_bc = zeros(dim, lamb+1).colwise() + vbar; // broadcasting vbar
217 vec ibg = (ip_yvbar.array()*ip_yvbar.array()) + gammav;
218 mat t = (exY.array().rowwise() * ip_yvbar.transpose().array()) -
219 (vbar_bc.array().rowwise() * ibg.transpose().array()) / 2.;
220 double b = -(1 - alphavd * alphavd) * normv4 / gammav + 2 * alphavd * alphavd;
221 vec H = constant(dim, 2.) - (b + 2 * alphavd * alphavd) * vbarbar; // dim x 1
222 vec invH = 1. / H.array();
223 mat s_step1 = yy.array() - normv2 / gammav * (yvbar.array().rowwise() * ip_yvbar.transpose().array()).array()
224 - constant(dim, lamb+1, 1.).array(); // dim x lamb+1
225 vec ip_vbart = vbar.transpose() * t; // 1 x lamb+1
226 mat s_step2 = s_step1.array()
227 - (alphavd / gammav * ((2 + normv2) * (t.array().colwise() * vbar.array()).array()
228 - (normv2 * (vbarbar * ip_vbart.transpose())).array())); // dim x lamb+1
229 vec invHvbarbar = invH.array() * vbarbar.array();
230 vec ip_s_step2invHvbarbar = invHvbarbar.transpose() * s_step2; // 1 x lamb+1
231
232 double div = 1 + b * (vbarbar.transpose() * invHvbarbar)(0,0);
233 if (div == 0)
234// div = 1E-13;int
235 throw std::invalid_argument( "division by 0" );
236 mat s = (s_step2.array().colwise() * invH.array()).array()
237 - ((b / div) * (invHvbarbar * ip_s_step2invHvbarbar.transpose())).array(); // dim x lamb+1
238
239 vec ip_svbarbar = vbarbar.transpose() * s; // 1 x lamb+1
240 t = t.array() - alphavd * ((2 + normv2) * (s.array().colwise() * vbar.array()).array()
241 - (vbar * ip_svbarbar.transpose()).array()); // dim x lamb+1
242 // update v, D
243 vec exw(lamb+1);
244 for (int k = 0; k < lamb; k++)
245 exw[k] = eta_B(lambF) * weights[k];
246 exw[lamb] = c1(lambF);
247
248 v = v.array() + (t * exw).array() / normv;
249 D = D.array() + (s * exw).array() * D.array();
250
251 // calculate detA
252 if (D.minCoeff() < 0) {
253 //throw std::invalid_argument( "D < 0" );
254 stop = -1;
255 return;
256 }
257 double nthrootdetA = cexp(D.array().log().sum() / dim + log(1 + (v.transpose() * v)(0,0)) / (2 * dim));
258 D = D.array() / nthrootdetA;
259 // update sigma
260 double G_s = (((z.array() * z.array()).array() - constant(dim, lamb, 1.).array()).matrix() * weights).sum() / dim;
261 sigma = sigma * cexp(eta_sigma / 2 * G_s);
262 }
263
264private:
265
266 double cexp(double a) { return exp(min(a, 100.0)); } // avoid overflow
267 double c1(double lambF) { return c1_cma * (dim - 5) / 6 * (lambF / lamb); }
268 double eta_B(double lambF) { return tanh((min(0.02 * lambF, 3 * log(dim)) + 5) / (0.23 * dim + 25)); }
269 double alpha_dist(double lambF) { return h_inv * min(1., sqrt(((double)lamb) / dim)) * sqrt(((double)lambF) / lamb); }
270 double w_dist_hat(mat z, double lambF) { return cexp(alpha_dist(lambF) * z.norm()); }
271 double eta_stag_sigma(double lambF) { return tanh((0.024 * lambF + 0.7 * dim + 20.) / (dim + 12.)); }
272 double eta_conv_sigma(double lambF) { return 2. * tanh((0.025 * lambF + 0.75 * dim + 10.) / (dim + 4.)); }
273 double f(double a) { return ((1. + a * a) * cexp(a * a / 2.) / 0.24) - 10. - dim; }
274 double f_prime(double a) { return (1. / 0.24) * a * cexp(a * a / 2.) * (3. + a * a); }
275
276 double get_h_inv(int dim) {
277 double h_inv = 1.0;
278 while (abs(f(h_inv)) > 1e-10)
279 h_inv = h_inv - 0.5 * (f(h_inv) / f_prime(h_inv));
280 return h_inv;
281 }

283 int num_feasible(const vec &evals) {
284 int n = 0;
285 for (int i = 0; i < evals.size(); i++)
286 if (evals[i] != INFINITY) n++;
287 return n;
288 }
289
290 ivec sort_indices_by(const vec &evals, mat z) {
291 int lam = evals.size();
292 ivec sorted_indices = sort_index(evals);
293 vec sorted_evals = evals(sorted_indices);
294 int no_of_feasible_solutions = num_feasible(sorted_evals);
295 if (no_of_feasible_solutions != lam) {
296 vec distances(lam - no_of_feasible_solutions);
297 int n = 0;
298 for (int i = 0; i < lam; i++)
299 if (evals[i] == INFINITY) {
300 distances[n] = z.col(i).squaredNorm();
301 n++;
302 }
303 ivec indices_sorted_by_distance = sort_index(distances);
304 for (int i = no_of_feasible_solutions; i < lam; i++)
305 sorted_indices[i] = sorted_indices[no_of_feasible_solutions
306 + indices_sorted_by_distance[i-no_of_feasible_solutions]];
307 }
308 return sorted_indices;
309 }

365void optimizeCRFMNES_C(int64_t runid, callback_parallel func_par, int dim,
366 double *init, double *lower, double *upper, double sigma,
367 int maxEvals, double stopfitness, int popsize,
368 int64_t seed, double penalty_coef, bool use_constraint_violation, bool normalize, double* res) {
369
370 vec guess(dim), lower_limit(dim), upper_limit(dim);
371 for (int i = 0; i < dim; i++) // guess is mandatory
372 guess[i] = init[i];
373 if (lower != NULL && upper != NULL) {
374 for (int i = 0; i < dim; i++) {
375 guess[i] = init[i];
376 lower_limit[i] = lower[i];
377 upper_limit[i] = upper[i];
378 }
379 } else {
380 lower_limit.resize(0);
381 upper_limit.resize(0);
382 normalize = false;
383 }
384
385 Fitness fitfun(noop_callback, func_par, dim, 1, lower_limit, upper_limit);
386 fitfun.setNormalize(normalize);
387
388 CrfmnesOptimizer opt(runid, &fitfun, dim, guess, sigma, popsize,
389 maxEvals, stopfitness, penalty_coef, use_constraint_violation, seed);
390 try {
391 opt.doOptimize();
392 } catch (std::exception &e) {
393 cout << e.what() << endl;
394 }
395 vec bestX = opt.getBestX();
396 double bestY = opt.getBestValue();
397 for (int i = 0; i < dim; i++)
398 res[i] = bestX[i];
399 res[dim] = bestY;
400 res[dim + 1] = fitfun.evaluations();
401 res[dim + 2] = opt.getIterations();
402 res[dim + 3] = opt.getStop();
403}
