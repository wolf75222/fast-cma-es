60dim = len(bounds)
61lb = np.array([b[0] for b in bounds])
62ub = np.array([b[1] for b in bounds])

64class hbv(object):
65
66 def __init__(self):
67 self.bounds = Bounds(lb, ub)
68 self.hbv = None
69 self.best_y = mp.RawValue(ct.c_double, np.inf)
70
71 def __call__(self, x):
72 x = np.array(x)
73 y = np.empty(nobj)
74 x_p = x.ctypes.data_as(ct.POINTER(ct.c_double))
75 y_p = y.ctypes.data_as(ct.POINTER(ct.c_double))
76 if self.hbv is None:
77 self.hbv = hbv_C()
78 fitness_hbv_C(self.hbv, x_p, y_p)
79 return np.array(y)
80
81 def qd_fitness(self, x):
82 y = self.__call__(x)
83 b = y.copy()
84 y = (y - self.qd_bounds.lb) / (self.qd_bounds.ub - self.qd_bounds.lb)
85 ws = sum(y)
86 if ws < self.best_y.value:
87 self.best_y.value = ws
88 print(f'{ws:.3f} { list(b) }')
89 return ws, b
