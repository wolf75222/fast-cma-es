46class lrgv(object):
47
48 def __init__(self):
49 self.bounds = Bounds(lb, ub)
50 self.lrgv_p = 0
51 self.name = "lrgvDE"
52 self.best_y = mp.RawValue(ct.c_double, np.inf)
53
54 def __call__(self, x):
55 y = np.empty(nobj) # C fitness call returns 6 objectives
56 c = np.empty(ncon)
57 x_p = x.ctypes.data_as(ct.POINTER(ct.c_double))
58 y_p = y.ctypes.data_as(ct.POINTER(ct.c_double))
59 c_p = c.ctypes.data_as(ct.POINTER(ct.c_double))
60 try:
61 if self.lrgv_p == 0:
62 args = ["-m","std-io","-b","AllDecAll","-c","ten-year"]
63 arguments = [bytes(argsi, 'utf-8') for argsi in args]
64 argv = (ct.c_char_p * len(arguments))()
65 argv[:] = arguments
66 argc = len(arguments)
67 self.lrgv_p = lrgv_C(argc, argv)
68 fitness_lrgv_C(self.lrgv_p, x_p, y_p, c_p)
69 except Exception as ex:
70 print(str(ex))
71 return np.array(list(y) + list(np.array(c)*-1.0)) # negate constraints
72
73 def qd_fitness(self, x):
74 y = self.__call__(x)
75 b = y[:nobj].copy()
76 constr = np.maximum(y[nobj:], 0) # we are only interested in constraint violations
77 c = np.amax(constr)
78 if c > 0.001: c += 10
79 y = (y[:nobj] - self.qd_bounds.lb) / (self.qd_bounds.ub - self.qd_bounds.lb)
80 ws = sum(y) + c
81 if ws < self.best_y.value:
82 self.best_y.value = ws
83 print(f'{ws:.3f} {sum(constr):.3f} { list(b) }')
84 return ws, b
