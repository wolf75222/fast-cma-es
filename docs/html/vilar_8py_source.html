95def get_bounds(model, scale):
96 lower = []
97 upper = []
98 for _, param in model.listOfParameters.items():
99 lower.append(param.value/scale)
100 upper.append(param.value*scale)
101 return Bounds(lower, upper)

109 # multi processing result list
110 results = mp.Manager().list()

112 class fcmaes_problem():
113
114 def __init__(self):
115 self.bounds = get_bounds(VilarOscillator(), 100)
116 self.dim = len(self.bounds.ub)
117
118 def fitness(self, x):
119 model = VilarOscillator()
120 set_params(model, x)
121 res = model.run(algorithm = "SSA")
122 # store params, result tuple
123 results.append((x, res))
124 R = res['R'] # time series for R
125 r_mean = np.mean(R)
126 r_over = np.array(np.fromiter((r for r in R if r > r_mean), dtype=float))
127 ilocs_max = argrelextrema(r_over, np.greater_equal, order=3)[0]
128 freq = len(ilocs_max) / len(R)
129 peak_dists = np.array(np.fromiter((ilocs_max[i] - ilocs_max[i-1] for i in range(1, len(ilocs_max))), dtype=float))
130 sdev_peak_dist = np.std(peak_dists)
131 peaks = (r_over - r_mean)[ilocs_max]
132 sdev_amp = np.std(peaks)
133 # maximize sdev_peak_dist and sdev_amp
134 return [-sdev_peak_dist, -sdev_amp, freq]
135
136 problem = fcmaes_problem()
137 popsize = 64 # population size of the evolutionary algorithm
138 max_evaluations = popsize*16 # maximum number of evaluation
139 # popsize = 256 # population size of the evolutionary algorithm
140 # max_evaluations = popsize*96 # maximum number of evaluation
141 nobj = 3 # number of objectives
142 ncon = 0 # number of constraints
143 # stores all values; if capacity is reached, content is replaced by the pareto front
144 store = mode.store(problem.dim, nobj, capacity=max_evaluations)
145 # perform the parameter sweep by multi objective optimization
146 xs, ys = mode.minimize(mode.wrapper(problem.fitness, 3, interval=64, store=store),
147 nobj, ncon,
148 problem.bounds, popsize = popsize,
149 max_evaluations = max_evaluations,
150 nsga_update=False, workers=min(popsize, mp.cpu_count()))
151 # save last population of the evolutionary algorithm
152 np.savez_compressed("sweep", xs=xs, ys=ys)
153 # save all evaluations
154 xs, ys = store.get_xs(), store.get_ys()
155 np.savez_compressed("sweep_all", xs=xs, ys=ys)
156 # show results
157 for x, res in results[:10]:
158 print(list(x), list(res['R']))
159 moretry.plot("sweep_all", ncon, xs, ys) # plot 2d
160 plot3d(xs, ys, "sweep_3d") # plot 3d
