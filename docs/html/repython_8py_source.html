91@njit(fastmath=True)
92def re_4(x):
93 x1, x2, x3, x4 = x[0], x[1], x[2], x[3]
94
95 P = 6000
96 L = 14
97 E = 30 * 1e6
98 G = 12 * 1e6
99 tauMax = 13600
100 sigmaMax = 30000
101
102 f1 = (1.10471 * x1 * x1 * x2) + (0.04811 * x3 * x4) * (14.0 + x2)
103 f2 = div(4 * P * L * L * L, E * x4 * x3 * x3 * x3)
104
105 M = P * (L + (x2 / 2))
106 tmpVar = ((x2 * x2) / 4.0) + np.power((x1 + x3) / 2.0, 2)
107 R = np.sqrt(tmpVar)
108 tmpVar = ((x2 * x2) / 12.0) + np.power((x1 + x3) / 2.0, 2)
109 J = 2 * np.sqrt(2) * x1 * x2 * tmpVar
110
111 tauDashDash = div(M * R, J)
112 tauDash = div(P, np.sqrt(2) * x1 * x2)
113 tmpVar = tauDash * tauDash + div((2 * tauDash * tauDashDash * x2), (2 * R)) + (tauDashDash * tauDashDash)
114 tau = np.sqrt(tmpVar)
115 sigma = div(6 * P * L, x4 * x3 * x3)
116 tmpVar = 4.013 * E * np.sqrt((x3 * x3 * x4 * x4 * x4 * x4 * x4 * x4) / 36.0) / (L * L)
117 tmpVar2 = (x3 / (2 * L)) * np.sqrt(E / (4 * G))
118 PC = tmpVar * (1 - tmpVar2)
119
120 g = np.array([
121 tauMax - tau,
122 sigmaMax - sigma,
123 x4 - x1,
124 PC - P
125 ])
126 g = np.maximum(g*-1, 0)
127
128 return np.array([f1,f2] + list(g))
