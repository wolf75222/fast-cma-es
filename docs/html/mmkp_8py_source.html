26@njit(fastmath=True)
27def fitness_(x, n, l, avail, values, resources):
28 vsum = 0
29 rsum = np.zeros(l, dtype=numba.int32)
30 for i in range(n):
31 vsum += values[i][x[i]]
32 rsum += resources[i][x[i]]
33 rsum = np.maximum(rsum - avail, np.zeros(l, dtype=numba.int32))
34 pen = np.sum(rsum)
35 return vsum, pen

37def parse(filename):
38 n = 0
39 with open(filename) as csvfile:
40 lines = csvfile.readlines()
41 avail = None
42 values = []
43 resources = []
44 for line in lines:
45 row = line.split()
46 if len(row) < 1:
47 continue
48 if n == 0:
49 n = int(row[0])
50 l = int(row[1])
51 m = int(row[2])
52 else:
53 if avail is None:
54 avail = [int(r) for r in row]
55 else:
56 if row[0] == 'Solutions':
57 break
58 if len(row) > 1:
59 values.append(float(row[0]))
60 resources.append([int(r) for r in row[1:]])
61 bsol = lines[-1].split()
62 bval = float(bsol[-1] )
63 bsol = np.array([int(r) for r in bsol[:-1]], dtype=int)
64 return n, l, m, bval, bsol, \
65 np.array(avail, dtype=int), \
66 np.reshape(np.array(values), (n,l)),\
67 np.reshape(np.array(resources, dtype=int), (n,l,m))

69class MMKP():
70 def __init__(self, problem):
71 self.problem = problem
72 filename = 'problems/' + problem
73 self.n, self.l, self.m, self.best_val, self.best_sol,\
74 self.avail, self.values, self.resources = parse(filename)
75 self.dim = self.n
76 self.bounds = Bounds([0]*self.dim, [self.l-1E-12]*self.dim)
77
78 def deviation(self, vsum):
79 return 100 * ((self.best_val - vsum) / self.best_val)
80
81 def fitness_pen(self, x):
82 return fitness_(x.astype(int), self.n, self.l, self.avail, \
83 self.values, self.resources)
84
85 def fitness(self, x):
86 vsum, penalty = self.fitness_pen(x)
87 if penalty > 0:
88 penalty += 100
89 return self.deviation(vsum) + penalty

91 def dump(self, x):
92 x = x.astype(int)
93 vsum, _ = self.fitness_pen(x)
94 lines = []
95 lines.append('Instance Name : ' + self.problem + '\n')
96 lines.append('Date : ' + str(datetime.today().date()) + '\n')
97 lines.append('Score = ' + str(vsum) +
98 ' Deviation = ' + str(round(self.deviation(vsum),2)) + " %\n")
99 lines.append('Solution\n')
100 lines.append(' '.join([str(xi) for xi in x]))
101 filename = 'solutions_co/' + self.problem + '.txt'
102 with open(filename, 'w') as f:
103 f.writelines(lines)
104 logger.info(''.join(lines))

106def optimize(mmkp, opt, num_retries = 32):
107 ret = retry.minimize(wrapper(mmkp.fitness),
108 mmkp.bounds, num_retries = num_retries,
109 stop_fitness = 1E-12, optimizer=opt)
110 x = ret.x.astype(int)
111 mmkp.dump(ret.x)
