55 // Project the vector x to the bound constraint set
56 static void force_bounds(Vector& x, const Vector& lb, const Vector& ub)
57 {
58 x.noalias() = x.cwiseMax(lb).cwiseMin(ub);
59 }
60
61 // Norm of the projected gradient
62 // ||P(x-g, l, u) - x||_inf
63 static Scalar proj_grad_norm(const Vector& x, const Vector& g, const Vector& lb, const Vector& ub)
64 {
65 return ((x - g).cwiseMax(lb).cwiseMin(ub) - x).cwiseAbs().maxCoeff();
66 }
67
68 // The maximum step size alpha such that x0 + alpha * d stays within the bounds
69 static Scalar max_step_size(const Vector& x0, const Vector& drt, const Vector& lb, const Vector& ub)
70 {
71 const int n = x0.size();
72 Scalar step = std::numeric_limits<Scalar>::infinity();
73
74 for(int i = 0; i < n; i++)
75 {
76 if(drt[i] > Scalar(0))
77 {
78 step = std::min(step, (ub[i] - x0[i]) / drt[i]);
79 } else if(drt[i] < Scalar(0)) {
80 step = std::min(step, (lb[i] - x0[i]) / drt[i]);
81 }
82 }
83
84 return step;
85 }

115 template <typename Foo>
116 inline int minimize(Foo& f, Vector& x, Scalar& fx, const Vector& lb, const Vector& ub)
117 {
118 using std::abs;
119
120 // Dimension of the vector
121 const int n = x.size();
122 if(lb.size() != n || ub.size() != n)
123 throw std::invalid_argument("'lb' and 'ub' must have the same size as 'x'");
124
125 // Check whether the initial vector is within the bounds
126 // If not, project to the feasible set
127 force_bounds(x, lb, ub);
128
129 // Initialization
130 reset(n);
131
132 // The length of lag for objective function value to test convergence
133 const int fpast = m_param.past;
134
135 // Evaluate function and compute gradient
136 fx = f(x, m_grad);
137 Scalar projgnorm = proj_grad_norm(x, m_grad, lb, ub);
138 if(fpast > 0)
139 m_fx[0] = fx;
140
141 // std::cout << "x0 = " << x.transpose() << std::endl;
142 // std::cout << "f(x0) = " << fx << ", ||proj_grad|| = " << projgnorm << std::endl << std::endl;
143
144 // Early exit if the initial x is already a minimizer
145 if(projgnorm <= m_param.epsilon || projgnorm <= m_param.epsilon_rel * x.norm())
146 {
147 return 1;
148 }
149
150 // Compute generalized Cauchy point
151 Vector xcp(n), vecc;
152 IndexSet newact_set, fv_set;
153 Cauchy<Scalar>::get_cauchy_point(m_bfgs, x, m_grad, lb, ub, xcp, vecc, newact_set, fv_set);
154
155 /* Vector gcp(n);
156 Scalar fcp = f(xcp, gcp);
157 Scalar projgcpnorm = proj_grad_norm(xcp, gcp, lb, ub);
158 std::cout << "xcp = " << xcp.transpose() << std::endl;
159 std::cout << "f(xcp) = " << fcp << ", ||proj_grad|| = " << projgcpnorm << std::endl << std::endl; */
160
161 // Initial direction
162 m_drt.noalias() = xcp - x;
163 m_drt.normalize();
164 // Tolerance for s'y >= eps * (y'y)
165 const Scalar eps = std::numeric_limits<Scalar>::epsilon();
166 // s and y vectors
167 Vector vecs(n), vecy(n);
168 // Number of iterations used
169 int k = 1;
170 for( ; ; )
171 {
172 // Save the curent x and gradient
173 m_xp.noalias() = x;
174 m_gradp.noalias() = m_grad;
175
176 // Line search to update x, fx and gradient
177 Scalar step_max = max_step_size(x, m_drt, lb, ub);
178 step_max = std::min(m_param.max_step, step_max);
179 Scalar step = Scalar(1);
180 step = std::min(step, step_max);
181 LineSearch<Scalar>::LineSearch(f, fx, x, m_grad, step, step_max, m_drt, m_xp, m_param);
182
183 // New projected gradient norm
184 projgnorm = proj_grad_norm(x, m_grad, lb, ub);
185
186 /* std::cout << "** Iteration " << k << std::endl;
187 std::cout << " x = " << x.transpose() << std::endl;
188 std::cout << " f(x) = " << fx << ", ||proj_grad|| = " << projgnorm << std::endl << std::endl; */
189
190 // Convergence test -- gradient
191 if(projgnorm <= m_param.epsilon || projgnorm <= m_param.epsilon_rel * x.norm())
192 {
193 return k;
194 }
195 // Convergence test -- objective function value
196 if(fpast > 0)
197 {
198 const Scalar fxd = m_fx[k % fpast];
199 if(k >= fpast && abs(fxd - fx) <= m_param.delta * std::max(std::max(abs(fx), abs(fxd)), Scalar(1)))
200 return k;
201
202 m_fx[k % fpast] = fx;
203 }
204 // Maximum number of iterations
205 if(m_param.max_iterations != 0 && k >= m_param.max_iterations)
206 {
207 return k;
208 }
209
210 // Update s and y
211 // s_{k+1} = x_{k+1} - x_k
212 // y_{k+1} = g_{k+1} - g_k
213 vecs.noalias() = x - m_xp;
214 vecy.noalias() = m_grad - m_gradp;
215 if(vecs.dot(vecy) > eps * vecy.squaredNorm())
216 m_bfgs.add_correction(vecs, vecy);
217
218 force_bounds(x, lb, ub);
219 Cauchy<Scalar>::get_cauchy_point(m_bfgs, x, m_grad, lb, ub, xcp, vecc, newact_set, fv_set);
220
221 /*Vector gcp(n);
222 Scalar fcp = f(xcp, gcp);
223 Scalar projgcpnorm = proj_grad_norm(xcp, gcp, lb, ub);
224 std::cout << "xcp = " << xcp.transpose() << std::endl;
225 std::cout << "f(xcp) = " << fcp << ", ||proj_grad|| = " << projgcpnorm << std::endl << std::endl;*/
226
227 SubspaceMin<Scalar>::subspace_minimize(m_bfgs, x, xcp, m_grad, lb, ub,
228 vecc, newact_set, fv_set, m_param.max_submin, m_drt);
229
230 /*Vector gsm(n);
231 Scalar fsm = f(x + m_drt, gsm);
232 Scalar projgsmnorm = proj_grad_norm(x + m_drt, gsm, lb, ub);
233 std::cout << "xsm = " << (x + m_drt).transpose() << std::endl;
234 std::cout << "f(xsm) = " << fsm << ", ||proj_grad|| = " << projgsmnorm << std::endl << std::endl;*/
235
236 k++;
237 }
238
239 return k;
240 }
