94class Cma_pyg(Optimizer):
95 """CMA_ES pagmo implementation."""
96
97 def __init__(self, max_evaluations=50000, popsize = 31, guess=None, stop_fitness = -np.inf):
98 Optimizer.__init__(self, max_evaluations, 'cma pagmo')
99 self.popsize = popsize
100 self.guess = guess
101 self.stop_fitness = stop_fitness
102
103 def minimize(self, fun, bounds, guess=None, sdevs=0.3, rg=Generator(MT19937()), store=None):
104 gen = int(self.max_eval_num(store) / self.popsize + 1)
105 algo = pg.algorithm(pg.cmaes(gen=gen, force_bounds = True,
106 sigma0 = np.mean(sdevs), seed = int(rg.uniform(0, 2**32 - 1))))
107 udp = pygmo_udp(fun, bounds)
108 prob = pg.problem(udp)
109 pop = pg.population(prob, self.popsize)
110 if not guess is None:
111 scale = np.multiply(0.5 * (bounds.ub - bounds.lb), sdevs)
112 for i in range(self.popsize):
113 xi = np.random.normal(guess, scale)
114 xi = np.maximum(np.minimum(xi, bounds.ub), bounds.lb)
115 pop.set_x(i, xi)
116 pop = algo.evolve(pop)
117 return pop.champion_x, pop.champion_f, pop.problem.get_fevals()

145def _test_archipelago(algo, problem, num = 10000, stop_val = -1E99, log = logger()):
146 udp = pygmo_udp(problem.fun, problem.bounds)
147 prob = pg.problem(udp)
148 best_y = np.inf
149 best_x = None
150 t0 = time.perf_counter()
151
152 for _ in range(num):
153 archi = pg.archipelago(n=mp.cpu_count(), algo=algo, prob=prob, pop_size=64)
154 archi.evolve()
155 archi.wait()
156 ys = archi.get_champions_f()
157 if not ys is None and len(ys) > 0:
158 sort = np.argsort([y[0] for y in ys])
159 y = ys[sort[0]][0]
160 if y < best_y:
161 best_y = y
162 best_x = archi.get_champions_x()[sort[0]]
163 message = '{0} {1} {2} {3!s}'.format(
164 problem.name, dtime(t0), best_y, list(best_x))
165 logger.info(message)
166 if best_y < stop_val:
167 break
168 return OptimizeResult(x=best_x, fun=best_y, success=True)
