32template<typename T>
33class blocking_queue {
34
35private:
36 size_t _capacity;
37 std::queue<T> _queue;
38 std::mutex _mutex;
39 std::condition_variable _not_full;
40 std::condition_variable _not_empty;
41
42public:
43 inline blocking_queue(size_t capacity) :
44 _capacity(capacity) {
45 }
46
47 inline size_t size() {
48 std::unique_lock<std::mutex> lock(_mutex);
49 return _queue.size();
50 }
51
52 //Inserts the specified element into this queue,
53 // waiting if necessary for space to become available.
54 inline void put(const T &elem) {
55 {
56 std::unique_lock<std::mutex> lock(_mutex);
57 while (_queue.size() >= _capacity)
58 _not_full.wait(lock);
59 _queue.push(elem);
60 }
61 _not_empty.notify_one();
62 }
63
64 // Retrieves and removes the head of this queue,
65 // waiting if necessary until an element becomes available.
66 inline const T& take() {
67 std::unique_lock<std::mutex> lock(_mutex);
68 while (_queue.size() == 0)
69 _not_empty.wait(lock);
70 T &front = _queue.front();
71 _queue.pop();
72 _not_full.notify_one();
73 return front;
74 }
75};

164static ivec sort_index(const vec &x) {
165 int size = x.size();
166 IndexVal ivals[size];
167 for (int i = 0; i < size; i++) {
168 ivals[i].index = i;
169 ivals[i].val = x[i];
170 }
171 std::sort(ivals, ivals + size, compareIndexVal);

355 void values(const mat &popX, vec &ys) {
356 int popsize = popX.cols();
357 int n = popX.rows();
358 double pargs[popsize * n];
359 double res[popsize];
360 for (int p = 0; p < popsize; p++) {
361 vec x = getClosestFeasible(decode(popX.col(p)));
362 for (int i = 0; i < n; i++)
363 pargs[p * n + i] = x(i);

372 vec violations = zeros(X.cols());
373 if (_lower.size() > 0) {
374 for (int i = 0; i < X.cols(); i++) {

376 violations[i] = penalty_coef

378 + (x - _upper).cwiseMax(0).sum());

419class evaluator {
420public:
421
422 evaluator(Fitness *fit, int nobj, int workers) :
423 _fit(fit), _nobj(nobj), _workers(workers), _stop(false) {
424 _requests = new blocking_queue<vec_id*>(2 * workers);
425 _evaled = new blocking_queue<vec_id*>(2 * workers);
426 _t0 = Clock::now();
427 if (_workers <= 0)
428 _workers = std::thread::hardware_concurrency();
429 for (int thread_id = 0; thread_id < _workers; thread_id++) {
430 _jobs.push_back(evaluator_job(thread_id, this));
431 }
432 }
433
434 ~evaluator() {
435 join();
436 delete _requests;
437 delete _evaled;
438 }
439
440 void evaluate(vec &x, int id) {
441 _requests->put(new vec_id(x, id));
442 }
443
444 // needs to be deleted
445 vec_id* result() {
446 return _evaled->take();
447 }
448
449 void execute(int thread_id) {
450 while (!_stop) {
451 vec_id *vid = _requests->take();
452 if (!_stop) {
453 try {
454 vec x = _fit->getClosestFeasible(_fit->decode(vid->_v));
455 vid->_v = _fit->eval(x);
456 } catch (std::exception &e) {
457 std::cout << e.what() << std::endl;
458 vid->_v = constant(_nobj, DBL_MAX);
459 }
460 _evaled->put(vid);
461 } else
462 delete vid;
463 }
464 }
465
466 void join() {
467 _stop = true;
468 vec x(0);
469 // to release all locks
470 for (auto &job : _jobs) {
471 _requests->put(new vec_id(x, 0));
472 }
473 for (auto &job : _jobs) {
474 job.join();
475 }
476 }

488 if (_thread.joinable())
489 _thread.join();
