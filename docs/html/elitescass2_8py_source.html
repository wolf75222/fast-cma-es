80 def __init__(self, prob):
81 self.problem = prob
82 self.dim = len(prob.bounds.lb)
83 self.qd_dim = 2
84 self.bounds = prob.bounds
85
86 min_tof = tof(prob.bounds.lb)
87 max_tof = tof(prob.bounds.ub)
88 min_launch = launch(prob.bounds.lb)
89 max_launch = launch(prob.bounds.ub)
90
91 self.qd_bounds = Bounds([min_tof, min_launch], [max_tof, max_launch])
92
93 def qd_fitness(self, x):
94 return self.problem.fun(x), np.array([tof(x), launch(x)])

96 def fitness(self, x):
97 return self.problem.fun(x)
98
99 def descriptors(self, x):

102def cma_elite(problem, archive, num=300):
103 ''' applies CMA-ES to the best num niches'''
104 si = archive.argsort()
105 for i in range(1, num+1):
106 try:
107 j = si[i]
108 print (j, archive.get_count(j))
109 print (archive.get_x_mean(j))
110 print (archive.get_x_min(j))
111 print (archive.get_x_max(j))
112 print (list(archive.get_x_stdev(j)))
113
114 guess = archive.get_x(j)
115 fun = archive.in_niche_filter(problem.qd_fitness, j)
116 print (archive.get_y(j), fun(guess))
117
118 lb = np.nan_to_num(archive.get_x_min(j), nan=-np.inf)
119 ub = np.nan_to_num(archive.get_x_max(j), nan=np.inf)
120 bounds = Bounds(np.maximum(problem.bounds.lb, lb),
121 np.minimum(problem.bounds.ub, ub))
122 from fcmaes import retry
123 from fcmaes.optimizer import Cma_cpp
124 res = retry.minimize(fun, bounds, num_retries=24*8,
125 optimizer=Cma_cpp(guess=guess, sdevs=0.001, workers=24)
126 )
127 y, d = problem.qd_fitness(res.x)
128 print (j, res.fun, fun(res.x), y, d)
129 archive.set(j, [y,d], res.x)
130 if i % 50 == 0:
131 archive.save("cass2archCma" + str(i))
132 except Exception as ex:
133 pass
134 archive.save("cass2archCma")

143def run_diversifier():
144 name = 'cass2div'
145 problem = Cassini2_me(Cassini2())
146 opt_params0 = {'solver':'elites', 'popsize':640}
147 opt_params1 = {'solver':'DE_CPP', 'max_evals':20000, 'popsize':32, 'stall_criterion':3}
148 opt_params2 = {'solver':'CMA_CPP', 'max_evals':50000, 'popsize':32, 'stall_criterion':3}
149 archive = diversifier.minimize(
150 mapelites.wrapper(problem.qd_fitness, 2), problem.bounds, problem.qd_bounds,
151 opt_params=[opt_params0, opt_params1, opt_params2], max_evals=640000000,
152 niche_num=160*160, samples_per_niche=12)
153 diversifier.apply_advretry(wrapper(problem.fitness), problem.descriptors,
154 problem.bounds, archive, num_retries=40000)
155 print('final archive:', archive.info())
156 archive.save(name)
157 plot_archive(archive)
