38@njit(fastmath=True)
39def evaluate_tsp(x, W, d, noise_factor, iter_num):
40 robust_total_route_length = 0
41 order = np.argsort(x) + 1
42 for _ in range(iter_num):
43 total_route_length = 0
44 total_route_length += W[0, order[0]] + np.random.random() * noise_factor
45 total_route_length += W[order[d-1], 0] + np.random.random() * noise_factor
46 for i in range(d-1):
47 total_route_length += W[order[i], order[i+1]] + np.random.random() * noise_factor
48 robust_total_route_length = max(total_route_length, robust_total_route_length)
49 return robust_total_route_length

53 def __init__(self, name, W, noise_factor, iter_num):
54 self.name = name
55 self.d = W.shape[0] - 1
56 self.W = W
57 self.noise_factor = noise_factor
58 self.iter_num = iter_num
59 self.evals = mp.RawValue(ct.c_long, 0) # writable across python processes
60 self.best_y = mp.RawValue(ct.c_double, np.inf)
61 self.t0 = time.perf_counter()
62
63 def __call__(self, x):
64 y = evaluate_tsp(x, self.W, self.d, self.noise_factor, self.iter_num)
65 self.evals.value += 1
66 if y < self.best_y.value:
67 self.best_y.value = y
68 logger.info("evals = {0}: time = {1:.1f} y = {2:.5f} x= {3:s}"
69 .format(self.evals.value, dtime(self.t0), y,
70 '[' + ", ".join([f"{xi:.16f}" for xi in x]) + ']'
71 ))
72 return y
