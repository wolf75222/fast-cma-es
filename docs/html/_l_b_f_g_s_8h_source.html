78 template <typename Foo>
79 inline int minimize(Foo& f, Vector& x, Scalar& fx)
80 {
81 using std::abs;
82
83 // Dimension of the vector
84 const int n = x.size();
85 reset(n);
86
87 // The length of lag for objective function value to test convergence
88 const int fpast = m_param.past;
89
90 // Evaluate function and compute gradient
91 fx = f(x, m_grad);
92 Scalar gnorm = m_grad.norm();
93 if(fpast > 0)
94 m_fx[0] = fx;
95
96 // Early exit if the initial x is already a minimizer
97 if(gnorm <= m_param.epsilon || gnorm <= m_param.epsilon_rel * x.norm())
98 {
99 return 1;
100 }
101
102 // Initial direction
103 m_drt.noalias() = -m_grad;
104 // Initial step size
105 Scalar step = Scalar(1) / m_drt.norm();
106
107 // Number of iterations used
108 int k = 1;
109 for( ; ; )
110 {
111 // Save the curent x and gradient
112 m_xp.noalias() = x;
113 m_gradp.noalias() = m_grad;
114
115 // Line search to update x, fx and gradient
116 LineSearch<Scalar>::LineSearch(f, fx, x, m_grad, step, m_drt, m_xp, m_param);
117
118 // New gradient norm
119 gnorm = m_grad.norm();
120
121 // Convergence test -- gradient
122 if(gnorm <= m_param.epsilon || gnorm <= m_param.epsilon_rel * x.norm())
123 {
124 return k;
125 }
126 // Convergence test -- objective function value
127 if(fpast > 0)
128 {
129 const Scalar fxd = m_fx[k % fpast];
130 if(k >= fpast && abs(fxd - fx) <= m_param.delta * std::max(std::max(abs(fx), abs(fxd)), Scalar(1)))
131 return k;
132
133 m_fx[k % fpast] = fx;
134 }
135 // Maximum number of iterations
136 if(m_param.max_iterations != 0 && k >= m_param.max_iterations)
137 {
138 return k;
139 }
140
141 // Update s and y
142 // s_{k+1} = x_{k+1} - x_k
143 // y_{k+1} = g_{k+1} - g_k
144 m_bfgs.add_correction(x - m_xp, m_grad - m_gradp);
145
146 // Recursive formula to compute d = -H * g
147 m_bfgs.apply_Hv(m_grad, -Scalar(1), m_drt);
148
149 // Reset step = 1.0 as initial guess for the next line search
150 step = Scalar(1);
151 k++;
152 }
153
154 return k;
155 }
156};
