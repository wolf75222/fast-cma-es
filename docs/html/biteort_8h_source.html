119 void updateWeights( const int UsePopSize )
120 {
121 // Calculate weights for centroid and covariance calculation.
122
123 const double lm = 1.0 / (int) ceil( UsePopSize * EvalFac );
124 double s = 0.0;
125 int i;
126
127 for( i = 0; i < UsePopSize; i++ )
128 {
129 const double l = 1.0 - i * lm;
130 WPopCent[ i ] = pow( l, CentPow );
131 s += WPopCent[ i ];
132 }
133
134 s = 1.0 / s;
135
136 for( i = 0; i < UsePopSize; i++ )
137 {
138 WPopCent[ i ] *= s;
139 WPopCov[ i ] = sqrt( WPopCent[ i ]);
140 }
141 }

180 void update( const CBitePop& ExtPop )
181 {
182 const int UsePopSize = ExtPop.getCurPopSize();
183 const double** const ExtParams = ExtPop.getPopParams();
184
185 updateWeights( UsePopSize );
186
187 // Prepare PopParams (vector of per-parameter population deviations),
188 // later used to calculate weighted covariances, use current centroid
189 // vector.
190
191 int i;
192 int j;
193
194 for( i = 0; i < ParamCount; i++ )
195 {
196 double* const op = PopParams[ i ];
197 const double c = CentParams[ i ];
198
199 for( j = 0; j < UsePopSize; j++ )
200 {
201 op[ j ] = ( ExtParams[ j ][ i ] - c ) * WPopCov[ j ];
202 }
203 }
204
205 // Store older centroid and calculate new centroid.
206
207 copyParams( PrevCentParams, CentParams );
208 calcCent( ExtParams, UsePopSize );
209
210 // Update covariance matrix, the left-handed triangle only. Uses leaky
211 // integrator averaging filter. "avgc" selects corner frequency
212 // of the filter.
213
214 // Select covariance update rate based on geometry parameter,
215 // depends on sphericity.
216
217 const double spc2 = spc * spc;
218 const double spc3 = spc2 * spc;
219 const double spc4 = spc2 * spc2;
220
221 const double avgc = 1.0 - pow( 0.01,
222 ( CovUpdFast * spc4 + CovUpdSlow * ( 1.0 - spc4 )) /
223 ( UsePopSize * EvalFac ));
224
225 // Perform covariance matrix update using a leaky integrator filter.
226
227 for( j = 0; j < ParamCount; j++ )
228 {
229 double* const cp = CovParams[ j ];
230
231 for( i = 0; i <= j; i++ )
232 {
233 const double cov = dotp( PopParams[ j ], PopParams[ i ],
234 UsePopSize );
235
236 cp[ i ] += ( cov - cp[ i ]) * avgc;
237 }
238 }
239
240 // Calculate rotation matrix and standard deviations.
241
242 eigen();
243
244 // Calculate distribution's geometry parameter.
245
246 double maxd = 1e-100; // Maximal deviation among all parameters.
247
248 for( i = 0; i < ParamCount; i++ )
249 {
250 if( DParams[ i ] > maxd )
251 {
252 maxd = DParams[ i ];
253 }
254 }
255
256 maxd = 1.0 / maxd;
257 spc = 0.0;
258
259 for( i = 0; i < ParamCount; i++ )
260 {
261 TmpParams[ i ] = DParams[ i ] * maxd;
262 spc += TmpParams[ i ];
263 }
264
265 spc /= ParamCount;
266
267 // Apply extension or contraction to positive and negative halfs of
268 // standard deviations, depending on centroid step size.
269
270 for( i = 0; i < ParamCount; i++ )
271 {
272 PrevCentParams[ i ] = CentParams[ i ] - PrevCentParams[ i ];
273 }
274
275 ortnc( PrevCentParams, DParamsN ); // Orthogonalize centroid step.
276
277 for( i = 0; i < ParamCount; i++ )
278 {
279 const double d = DParamsN[ i ];
280
281 DParamsN[ i ] = DParams[ i ] - d;
282 DParams[ i ] += d;
283 }
284
285 // Modify standard deviations, apply reduction, or extension to
286 // overly shrunk dimensions. Depends on sphericity.
287
288 for( i = 0; i < ParamCount; i++ )
289 {
290 const double c = 1.0 - TmpParams[ i ];
291 const double m = ( SigmaMulBase2 +
292 ( SigmaMulExp2 - SigmaMulBase2 ) * c * c * c ) *
293 ( 1.0 - spc3 ) + SigmaMulBase1 * spc3;
294
295 DParams[ i ] *= m;
296 DParamsN[ i ] *= m;
297 }
298 }

493 static void tred2( const int n, double* d, double** V, double* e )
494 {
495 // This is derived from the Algol procedures tred2 by
496 // Bowdler, Martin, Reinsch, and Wilkinson, Handbook for
497 // Auto. Comp., Vol.ii-Linear Algebra, and the corresponding
498 // Fortran subroutine in EISPACK.
499
500 for (int j = 0; j < n; j++) {
501 d[j] = V[n-1][j];
502 }
503
504 // Householder reduction to tridiagonal form.
505
506 for (int i = n-1; i > 0; i--) {
507
508 // Scale to avoid under/overflow.
509
510 double scale = 0.0;
511 double h = 0.0;
512 for (int k = 0; k < i; k++) {
513 scale = scale + abs(d[k]);
514 }
515 if (scale == 0.0) {
516 e[i] = d[i-1];
517 for (int j = 0; j < i; j++) {
518 d[j] = V[i-1][j];
519 V[i][j] = 0.0;
520 V[j][i] = 0.0;
521 }
522 } else {
523
524 // Generate Householder vector.
525
526 for (int k = 0; k < i; k++) {
527 d[k] /= scale;
528 h += d[k] * d[k];
529 }
530 double f = d[i-1];
531 double g = sqrt(h);
532 if (f > 0) {
533 g = -g;
534 }
535 e[i] = scale * g;
536 h = h - f * g;
537 d[i-1] = f - g;
538 for (int j = 0; j < i; j++) {
539 e[j] = 0.0;
540 }
541
542 // Apply similarity transformation to remaining columns.
543
544 for (int j = 0; j < i; j++) {
545 f = d[j];
546 V[j][i] = f;
547 g = e[j] + V[j][j] * f;
548 for (int k = j+1; k <= i-1; k++) {
549 g += V[k][j] * d[k];
550 e[k] += V[k][j] * f;
551 }
552 e[j] = g;
553 }
554 f = 0.0;
555 for (int j = 0; j < i; j++) {
556 e[j] /= h;
557 f += e[j] * d[j];
558 }
559 double hh = f / (h + h);
560 for (int j = 0; j < i; j++) {
561 e[j] -= hh * d[j];
562 }
563 for (int j = 0; j < i; j++) {
564 f = d[j];
565 g = e[j];
566 for (int k = j; k <= i-1; k++) {
567 V[k][j] -= (f * e[k] + g * d[k]);
568 }
569 d[j] = V[i-1][j];
570 V[i][j] = 0.0;
571 }
572 }
573 d[i] = h;
574 }
575
576 // Accumulate transformations.
577
578 for (int i = 0; i < n-1; i++) {
579 V[n-1][i] = V[i][i];
580 V[i][i] = 1.0;
581 double h = d[i+1];
582 if (h != 0.0) {
583 for (int k = 0; k <= i; k++) {
584 d[k] = V[k][i+1] / h;
585 }
586 for (int j = 0; j <= i; j++) {
587 double g = 0.0;
588 for (int k = 0; k <= i; k++) {
589 g += V[k][i+1] * V[k][j];
590 }
591 for (int k = 0; k <= i; k++) {
592 V[k][j] -= g * d[k];
593 }
594 }
595 }
596 for (int k = 0; k <= i; k++) {
597 V[k][i+1] = 0.0;
598 }
599 }
600 for (int j = 0; j < n; j++) {
601 d[j] = V[n-1][j];
602 V[n-1][j] = 0.0;
603 }
604 V[n-1][n-1] = 1.0;
605 e[0] = 0.0;
606 }

612 static double hypot_( double a, double b )
613 {
614 const double fa = fabs(a);
615 const double fb = fabs(b);
616 double r;
617 if (fa > fb)
618 {
619 r = b/a;
620 r = fa*sqrt(1.0+r*r);
621 }
622 else
623 if (b != 0.0)
624 {
625 r = a/b;
626 r = fb*sqrt(1.0+r*r);
627 }
628 else
629 {
630 r = 0.0;
631 }
632 return r;
633 }

649 static void tql2( const int n, double* d, double** V, double* e )
650 {
651 // This is derived from the Algol procedures tql2, by
652 // Bowdler, Martin, Reinsch, and Wilkinson, Handbook for
653 // Auto. Comp., Vol.ii-Linear Algebra, and the corresponding
654 // Fortran subroutine in EISPACK.
655
656 for (int i = 1; i < n; i++) {
657 e[i-1] = e[i];
658 }
659 e[n-1] = 0.0;
660
661 double f = 0.0;
662 double tst1 = 0.0;
663 double eps = pow(2.0,-52.0);
664 for (int l = 0; l < n; l++) {
665
666 // Find small subdiagonal element
667
668 tst1 = max_(tst1,abs(d[l]) + abs(e[l]));
669 int m = l;
670 while (m < n) {
671 if (abs(e[m]) <= eps*tst1) {
672 break;
673 }
674 m++;
675 }
676
677 // If m == l, d[l] is an eigenvalue,
678 // otherwise, iterate.
679
680 if (m > l) {
681 do {
682 // Compute implicit shift
683
684 double g = d[l];
685 double p = (d[l+1] - g) / (2.0 * e[l]);
686 double r = hypot_(p,1.0);
687 if (p < 0) {
688 r = -r;
689 }
690 d[l] = e[l] / (p + r);
691 d[l+1] = e[l] * (p + r);
692 double dl1 = d[l+1];
693 double h = g - d[l];
694 for (int i = l+2; i < n; i++) {
695 d[i] -= h;
696 }
697 f = f + h;
698
699 // Implicit QL transformation.
700
701 p = d[m];
702 double c = 1.0;
703 double c2 = c;
704 double c3 = c;
705 double el1 = e[l+1];
706 double s = 0.0;
707 double s2 = 0.0;
708 for (int i = m-1; i >= l; i--) {
709 c3 = c2;
710 c2 = c;
711 s2 = s;
712 g = c * e[i];
713 h = c * p;
714 r = hypot_(p,e[i]);
715 e[i+1] = s * r;
716 s = e[i] / r;
717 c = p / r;
718 p = c * d[i] - s * g;
719 d[i+1] = h + s * (c * g + s * d[i]);
720
721 // Accumulate transformation.
722
723 for (int k = 0; k < n; k++) {
724 h = V[k][i+1];
725 V[k][i+1] = s * V[k][i] + c * h;
726 V[k][i] = c * V[k][i] - s * h;
727 }
728 }
729 p = -s * s2 * c3 * el1 * e[l] / dl1;
730 e[l] = s * p;
731 d[l] = c * p;
732
733 // Check for convergence.
734
735 } while (abs(e[l]) > eps*tst1);
736 }
737 d[l] = d[l] + f;
738 e[l] = 0.0;
739 }
740 }

747 void eigen()
748 {
749 // Replicate left-handed triangular matrix.
750
751 int i;
752 int j;
753
754 for( j = 0; j < ParamCount; j++ )
755 {
756 for( i = 0; i <= j; i++ )
757 {
758 BParams[ j ][ i ] = BParams[ i ][ j ] = CovParams[ j ][ i ];
759 }
760 }
761
762 // Calculate eigen-pairs.
763
764 tred2( ParamCount, DParams, BParams, TmpParams );
765 tql2( ParamCount, DParams, BParams, TmpParams );
766
767 for( i = 0; i < ParamCount; i++ )
768 {
769 if( DParams[ i ] < 1e-60 )
770 {
771 DParams[ i ] = 1e-30;
772 }
773 else
774 {
775 DParams[ i ] = sqrt( DParams[ i ]);
776 }
777 }
