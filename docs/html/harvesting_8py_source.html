6# This code implements a variant of the multi objective flexible job shop problem
7# and is derived from https://github.com/dietmarwo/fast-cma-es/blob/master/examples/jobshop.py
8
9# It supports both a multiobjective and the single objective variant utilizing Numba and the optimizers
10# https://github.com/dietmarwo/fast-cma-es/blob/master/_fcmaescpp/modeoptimizer.cpp (MO) and
11# https://github.com/dietmarwo/fast-cma-es/blob/master/_fcmaescpp/include/biteopt.h (SO).
12
13# The FJSP variant implemented here is related to "asteroid harvesting":
14
15# N movable identical factories are deployed on N asteroids to perform operations associated to m jobs.
16# As in FJSP the operations need to be executed in the order specified by the job.
17# The equivalent to a machine in FJSP is a factory deployment to a specific asteroid. Its resources
18# determine its capability to execute job operations. Therefore - using this simplified
19# model - asteroid harvesting can be viewed as a FJSP with two constraints:
20
21# a) Moving factories is expensive, therefore a factory can only be deployed once on an asteroid,
22# it is active for a single consecutive time window.
23# b) The upper limit of active machines (factory deployments) is determined by N, the number
24# of factories.
25
26# See https://github.com/dietmarwo/fast-cma-es/blob/master/tutorials/JobShop.adoc for a detailed description.
27
28# Tested using https://docs.conda.io/en/main/miniconda.html on Linux Mint 21.2

53def scheduling(tasks, n_jobs, n_machines, max_active, start, duration):
54 success, start, stop = adjust_timing(start, duration, max_active)
55 if not success:
56 logger.info("timing error")
57 machine_time = start # we initialize with the machine startup times
58 job_time = np.zeros(n_jobs)
59 fails = 0
60 solution = {'machine': [], 'start': [], 'end': [], 'job': [], 'task':[]}
61 for task in tasks:
62 job = int(task[0])
63 machine = int(task[2])
64 time = task[3]
65 start = max(machine_time[machine], job_time[job])
66 end = start + time
67 if end > stop[machine]: # machine already shut down
68 fails += 1 # failure to execute task at all
69 continue
70 machine_time[machine] = end
71 job_time[job] = end
72 solution['machine'].append(int(machine))
73 solution['start'].append(int(start))
74 solution['end'].append(int(end))
75 solution['job'].append(int(job))
76 solution['task'].append(int(task[1]))
77 logger.info('fails = ' + str(fails))
78 return solution

85@njit(fastmath=True)
86def adjust_timing(start, duration, max_active):
87 inf = 10000000
88 start = start.copy()
89 i_sort = np.argsort(start)
90 for k in range(max_active):
91 start[i_sort[k]] = 0 # first max_active available from start
92 stop = start + duration
93 j_sort = np.argsort(stop)
94 for k in range(max_active):
95 stop[j_sort[-1 - k]] = inf # last max_active available forever
96 n = len(start)
97 moved = True
98 while moved:
99 active = 0
100 i = 0
101 j = 0
102 moved = False
103 while i < n and j < n:
104 ci = i_sort[i] # current start
105 cj = j_sort[j] # current stop
106 if start[ci] >= stop[cj]:
107 active -= 1
108 j += 1
109 else:
110 if active < max_active:
111 active += 1
112 i += 1
113 else:
114 if (ci != cj): # move start after next machine stops
115 for i in range(n):
116 if start[i] == stop[ci]: # other machine already waiting for stop[cj]
117 return False, start, stop # otherwise can lead to an infinite loop
118 start[ci] = stop[cj]
119 stop[ci] = start[ci] + duration[ci]
120 moved = True
121 break;
122 else:
123 j += 1
124 if moved:
125 i_sort = np.argsort(start) # sort again
126 j_sort = np.argsort(stop)
127 return True, start, stop

129@njit(fastmath=True)
130def exec_tasks(tasks, n_jobs, n_machines, max_active, start, duration):
131 success, start, stop = adjust_timing(start, duration, max_active)
132 if not success:
133 return None, None, None, None
134 machine_time = start # we initialize with the machine startup times
135 machine_work_time = np.zeros(n_machines)
136 job_time = np.zeros(n_jobs)
137 fails = 0
138 for task in tasks:
139 job = int(task[0])
140 machine = int(task[2])
141 time = task[3]
142 # previous task needs to be finished and machine needs to be available
143 end_time = max(machine_time[machine], job_time[job]) + time
144 if end_time > stop[machine]: # machine already shut down
145 fails += 1 # failure to execute task at all
146 continue
147 machine_time[machine] = end_time
148 job_time[job] = end_time
149 machine_work_time[machine] += time
150 return machine_time, job_time, machine_work_time, fails

152@njit(fastmath=True)
153def filtered_tasks(x, task_data, n_operations, n_machines, job_indices, job_ids):
154 operations = filter_tasks(x, task_data, n_operations, n_machines)
155 order = np.argsort(x[n_operations:2*n_operations])
156 tasks = reorder(operations, order, n_operations, job_ids, job_indices)
157 return tasks

161 def __init__(self, task_data, bounds, n_jobs, n_operations, n_machines, max_active, name):
162 self.evals = mp.RawValue(ct.c_long, 0) # writable across python processes
163 self.best_y = mp.RawValue(ct.c_double, np.inf) # writable across python processes
164 self.t0 = time.perf_counter()
165 self.task_data = task_data
166 self.n_jobs = n_jobs

171 self.name = name
172 self.nobj = 3

185 def fun(self, x): # multi objective function
186 tasks = filtered_tasks(x, self.task_data, self.n_operations, self.n_machines,
187 self.job_indices, self.job_ids)
188 max_time = x[-1]
189 start = x[-self.n_machines-1:-1]*max_time
190 duration = x[-2*self.n_machines-1:-self.n_machines-1]*max_time
191 machine_time, job_time, machine_work_time, fails = \
192 exec_tasks(tasks, self.n_jobs, self.n_machines, self.max_active, start, duration)
193 if fails is None: # timing error
194 return np.array([0, 0, 0, 10000])
195 span = np.amax(machine_time)
196 work = np.sum(machine_work_time)
197 wmax = np.amax(machine_work_time)
198 ys = np.array([span, work, wmax, fails])
199 y = sum(self.weights*ys) # weighted sum
200 self.evals.value += 1
201 if y < self.best_y.value:
202 self.best_y.value = y
203 logger.info("evals = {0}: time = {1:.1f} t = {2:.0f} f = {3:.0f} y = {4:.2f} s = {5:.2f} w = {6:.0f} m = {7:.0f} m= {8:s} j= {9:s} w= {10:s}"
204 .format(self.evals.value, dtime(self.t0), max_time, fails, y, span, work, wmax,
205 str([int(si) for si in machine_time]),
206 str([int(oi) for oi in job_time]),
207 str([int(oi) for oi in machine_work_time]),
208 ))
209 return ys

211 def __call__(self, x): # single objective function
212 ys = self.fun(x)
213 return sum(self.weights*ys) # weighted sum

215def retry_modecpp(fit, retry_num = 32, popsize = 48, max_eval = 500000,

217 xf, yf = modecpp.retry(fit.fun, fit.nobj, fit.ncon, fit.bounds, None, retry_num, popsize,
218 max_evaluations = 960000, nsga_update = nsga_update, workers=workers)

230def optimize(bi, max_active, multi_objective = True):
231 name = "BrandimarteMk" + str(bi)
232 tasks, n_jobs, n_machines, n_operations, sum_time = read_fjs("data/1_Brandimarte/" + name + ".fjs")
233
234 dim = 2*n_operations + 2*n_machines + 1
235 lower_bound = np.zeros(dim)
236 lower_bound[:] = 0.0000001
237 upper_bound = np.zeros(dim)
238 upper_bound[:] = 0.9999999
239 upper_bound[-1] = sum_time

252 retry.retry(store, Bite_cpp(960000,M=1).minimize, num_retries=256)
