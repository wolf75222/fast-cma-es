133def cv_score(X):
134 X = X[0]
135 score = cross_val_score(
136 XGBRegressor(colsample_bytree=X[0],
137 gamma=X[1],
138 min_child_weight=X[2],
139 learning_rate=X[3],
140 max_depth=int(X[4]),
141 n_estimators=10000,
142 reg_alpha=X[5],
143 reg_lambda=X[6],
144 subsample=X[7],
145 n_jobs=1 # required for cmaes with multiple workers
146 ),
147 train_x, train_y, scoring='neg_mean_squared_error').mean()
148
149 return np.array(score)
150
151def obj_f(X):
152 return -cv_score([X])
153
154class cv_problem(object):
155
156 def __init__(self, pfun, bounds):
157 self.name = "cv_score"
158 self.dim = len(bounds.lb)
159 self.pfun = pfun
160 self.bounds = bounds
161 self.evals = mp.RawValue(ct.c_int, 0)
162 self.best_y = mp.RawValue(ct.c_double, np.inf)
163 self.t0 = time.perf_counter()
164
165 def fun(self, x):
166 self.evals.value += 1
167 y = self.pfun(x)
168 if y < self.best_y.value:
169 self.best_y.value = y
170 logger.info(str(dtime(self.t0)) + ' ' +
171 str(self.evals.value) + ' ' +
172 str(self.best_y.value) + ' ' +
173 str(list(x)))
174 return y
