1"""Find the optimal subset fulfilling any property.
2
3Single objective variant of https://github.com/dietmarwo/fast-cma-es/blob/master/examples/subset_mo.py
4
5As example we use transactions where a subset is to be matched to a list of payments.
6For transactions: [2,4,5,1,3] and payments: [4, 4] the sum of payments is 8.
7Subsets of transactions that minimize the difference between payments and
8transactions are for instance [5,3], or [4,3,1].
9
10In general we have to define a mapping 'selection_value' which maps a
11specific selection/subset represented as boolean array to a value to be minimized.
12
13See https://github.com/dietmarwo/fast-cma-es/blob/master/tutorials/Subset.adoc for a detailed description.

31class transaction_value():
32
33 def __init__(self, transactions, payments):
34 self.transactions = transactions
35 self.sum_payments = sum(payments)
36
37 def __call__(self, selection):
38 return abs(sum(self.transactions[selection]) - self.sum_payments)

40class fitness():
41
42 def __init__(self, selection_value, dim):
43 self.selection_value = selection_value
44 self.bounds = Bounds([0]*dim, [1.99999999]*dim)
45
46 def selected(self, x):
47 return x.astype(int)
48
49 # all decision variables are in the [0,2[ interval and mapped to a boolean array.
50 def __call__(self, x):
51 return self.selection_value(x.astype(int).astype(bool))

53# Multiple optimizations are executed in parallel and all results are collected
54def optimize(fitness, opt, num_retries = 32):
55 store = retry.Store(wrapper(fitness), fitness.bounds)
56 retry.retry(store, opt.minimize, num_retries)
57 xs = store.get_xs()
58 ys = store.get_ys()
59 # show the best results
60 for i in range(len(xs)):
61 if ys[i] > 0.001: break
62 print(i+1, ") Optimal Objective value: ", ys[i])
63 print(fitness.selected(xs[i]))
