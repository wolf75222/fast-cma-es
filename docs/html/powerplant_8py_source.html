157 def calculate_efficiency(self, x):
158 # set extraction pressure
159 self.nw.get_conn('extraction1').set_attr(p=x[0])
160 self.nw.get_conn('extraction2').set_attr(p=x[1])
161
162 self.nw.solve('design')
163
164 # components are saved in a DataFrame, column 'object' holds the
165 # component instances
166 for cp in self.nw.comps['object']:
167 if isinstance(cp, Condenser) or isinstance(cp, Desuperheater):
168 if cp.Q.val > 0:
169 return np.nan
170 elif isinstance(cp, Pump):
171 if cp.P.val < 0:
172 return np.nan
173 elif isinstance(cp, Turbine):
174 if cp.P.val > 0:
175 return np.nan
176
177 if self.nw.res[-1] > 1e-3 or self.nw.lin_dep:
178 return np.nan
179 else:
180 return self.nw.busses['power'].P.val / self.nw.busses['heat'].P.val

182 def calculate_qd(self, x):
183 y = self.calculate_efficiency(x)
184 desc = [self.nw.busses['power'].P.val, self.nw.busses['heat'].P.val]

211 def efficiency(self, x):
212 try:
213 with threadpoolctl.threadpool_limits(limits=1, user_api="blas"):
214 eff = self.get_model().calculate_efficiency(x)
215 if not np.isfinite(eff): # model gets corrupted in case of an error
216 self.create_model() # we need to recreate the model
217 return 0
218 return eff
219 except Exception as ex:
220 return 0

222 def fitness(self, x):
223 y = -self.efficiency(x)
224 c = -x[0] + x[1]
225 return [y, c]

228 if x[1] > x[0]: # handle constraint

291 class qd_problem():
292
293 def __init__(self):
294 self.dim = 2
295 self.qd_dim = 2
296 self.bounds = Bounds([1]*self.dim, [40]*self.dim)
297 self.qd_bounds = Bounds([2.2E8, 5E8], [2.8E8, 6.3E8])
298 self.local = threading.local()
299
300 def get_model(self):
301 if not hasattr(self.local, 'model'):
302 self.create_model()
303 return self.local.model
304
305 def create_model(self):
306 self.local.model = PowerPlant()
307
308 def efficiency(self, x):
309 try:
310 with threadpoolctl.threadpool_limits(limits=1, user_api="blas"):
311 eff, desc = self.get_model().calculate_qd(x)
312 if not np.isfinite(eff): # model gets corrupted in case of an error
313 self.create_model() # we need to recreate the model
314 return 0, self.qd_bounds.lb
315 #print (eff, desc)
316 return eff, desc
317 except Exception as ex:
318 return 0, self.qd_bounds.lb
319
320 def qd_fitness(self, x):
321 y, desc = self.efficiency(x)
322 return 1-y, desc
323
324 problem = qd_problem()
325 opt_params0 = {'solver':'elites', 'popsize':128}
326 opt_params1 = {'solver':'CMA_CPP', 'max_evals':200, 'popsize':16, 'stall_criterion':3}
327 archive = diversifier.minimize(
328 mapelites.wrapper(problem.qd_fitness, 2, interval=1000), problem.bounds, problem.qd_bounds,
329 opt_params=[opt_params0, opt_params1], max_evals=25600)
330 print('final archive:', archive.info())
331 archive.save('powerplant_qd')
332 plot_archive(archive)
