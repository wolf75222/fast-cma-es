282 double getGaussian()
283 {
284 double q, u, v;
285
286 do
287 {
288 u = get();
289 v = get();
290
291 if( u == 0.0 || v == 0.0 )
292 {
293 u = 1.0;
294 v = 1.0;
295 }
296
297 v = 1.7156 * ( v - 0.5 );
298 const double x = u - 0.449871;
299 const double y = fabs( v ) + 0.386595;
300 q = x * x + y * ( 0.19600 * y - 0.25472 * x );
301
302 if( q < 0.27597 )
303 {
304 break;
305 }
306
307 } while( q > 0.27846 || v * v > -4.0 * log( u ) * u * u );
308

351 uint64_t advance()
352 {
353 if( rf != NULL )
354 {
355 uint64_t r = ( *rf )( rdata );
356 r |= (uint64_t) ( *rf )( rdata ) << 32;
357
358 return( r );
359 }
360
361 Seed *= lcg * 2 + 1;
362 const uint64_t rs = Seed >> 32 | Seed << 32;
363 Hash += rs + 0xAAAAAAAAAAAAAAAA;
364 lcg += Seed + 0x5555555555555555;
365 Seed ^= Hash;
366

493 void incr( CBiteRnd& rnd, const double v = 1.0 )
494 {
495 const int dp = (int) ( -Selp * v * v );
496
497 if( dp < 0 ) // Boost an efficient choice.
498 {
499 if( dp == -1 )
500 {
501 Sels[ Slot ][ Selp ] = Sels[ Slot ][ Selp - 1 ];
502 Sels[ Slot ][ Selp - 1 ] = Sel;
503 }
504 else
505 {
506 const int np = Selp + dp;
507 memmove( Sels[ Slot ] + np + 1, Sels[ Slot ] + np,
508 -dp * sizeof( Sels[ Slot ][ 0 ]));
509
510 Sels[ Slot ][ np ] = Sel;
511 }
512 }
513
514 SlotAccums[ Slot ] += AccumCoeff;
515
516 if( SlotAccums[ Slot ] >= 1.0 )
517 {
518 const double a = SlotAccums[ Slot ] - 1.0;
519
520 if( Slot > 0 )
521 {
522 int* const t = Sels[ Slot ];
523 Sels[ Slot ] = Sels[ Slot - 1 ];
524 Sels[ Slot - 1 ] = t;
525
526 SlotAccums[ Slot ] = SlotAccums[ Slot - 1 ];
527 SlotAccums[ Slot - 1 ] = a;
528 }
529 else
530 {
531 SlotAccums[ Slot ] = a;
532 }

795 void updateCentroid()
796 {
797 NeedCentUpdate = false;
798
799 const int BatchCount = ( 1 << IntOverBits ) - 1;
800 ptype* const cp = CentParams;
801 const double cm = 1.0 / PopSize;
802 ptype* const tp = TmpParams;
803 int i;
804 int j;
805
806 if( PopSize <= BatchCount )
807 {
808 copyParams( tp, PopParams[ 0 ]);
809
810 for( j = 1; j < PopSize; j++ )
811 {
812 const ptype* const p = PopParams[ j ];
813
814 for( i = 0; i < ParamCount; i++ )
815 {
816 tp[ i ] += p[ i ];
817 }
818 }
819
820 for( i = 0; i < ParamCount; i++ )
821 {
822 cp[ i ] = (ptype) ( tp[ i ] * cm );
823 }
824 }
825 else
826 {
827 // Batched centroid calculation, for more precision and no integer
828 // overflows.
829
830 int pl = PopSize;
831 j = 0;
832 bool DoCopy = true;
833
834 while( pl > 0 )
835 {
836 int c = ( pl > BatchCount ? BatchCount : pl );
837 pl -= c;
838 c--;
839
840 copyParams( tp, PopParams[ j ]);
841
842 while( c > 0 )
843 {
844 j++;
845 const ptype* const p = PopParams[ j ];
846
847 for( i = 0; i < ParamCount; i++ )
848 {
849 tp[ i ] += p[ i ];
850 }
851
852 c--;
853 }
854
855 if( DoCopy )
856 {
857 DoCopy = false;
858
859 for( i = 0; i < ParamCount; i++ )
860 {
861 cp[ i ] = (ptype) ( tp[ i ] * cm );
862 }
863 }
864 else
865 {
866 for( i = 0; i < ParamCount; i++ )
867 {
868 cp[ i ] += (ptype) ( tp[ i ] * cm );
869 }
870 }
871 }
872 }

1039 int updatePop( double UpdCost, const ptype* const UpdParams,
1040 const bool DoUpdateCentroid, const bool CanRejectCost = true )
1041 {

1043
1044 if( CurPopPos < PopSize )
1045 {
1046 ri = CurPopPos;
1047
1048 if( UpdCost != UpdCost ) // Handle NaN.
1049 {
1050 UpdCost = 1e300;
1051 }
1052 }
1053 else
1054 {
1055 ri = PopSize1;
1056
1057 if( UpdCost != UpdCost || // Check for NaN.
1058 UpdCost >= *getObjPtr( PopParams[ ri ]))
1059 {
1060 return( PopSize );
1061 }
1062 }
1063
1064 // Perform binary search of a cost within the population.
1065
1066 int p = 0;
1067 int i = ri;
1068
1069 while( p < i )
1070 {
1071 const int mid = ( p + i ) >> 1;
1072
1073 if( *getObjPtr( PopParams[ mid ]) >= UpdCost )
1074 {
1075 i = mid;
1076 }
1077 else
1078 {
1079 p = mid + 1;
1080 }
1081 }
1082
1083 if( CurPopPos < PopSize )
1084 {
1085 CurPopPos++;
1086 }
1087/*
1088 else
1089 {
1090 if( CanRejectCost )
1091 {
1092 // Reject same-cost solution using equality precision level.
1093 // This approach reduces search locality due to allowing older
1094 // solutions to remain in population.
1095
1096 static const double etol = 0x1p-52;
1097 const double c = *getObjPtr( PopParams[ p ]);
1098 const double cd = fabs( UpdCost - c );
1099
1100 if( cd == 0.0 )
1101 {
1102 return( PopSize );
1103 }
1104 else
1105 {
1106 const double cs = fabs( UpdCost ) + fabs( c );
1107
1108 if( cd < cs * etol )
1109 {
1110 return( PopSize );
1111 }
1112 }
1113 }
1114 }
1115*/

1120 memmove( pp + 1, pp, mc * sizeof( pp[ 0 ]));
1121
1122 *pp = rp;
1123 *getObjPtr( rp ) = UpdCost;
1124 *getRankPtr( rp ) = UpdCost;
1125
1126 if( rp != UpdParams )
1127 {
1128 if( DoUpdateCentroid )
1129 {

1131 const double lpc = CentLPC;
1132
1133 for( i = 0; i < ParamCount; i++ )
1134 {

1137 }
1138 }

1151 }
