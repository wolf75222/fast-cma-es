162def message_to_json(message):
163 """
164 Converts a given message object to its JSON representation.
165
166 This function serializes a message object based on its type. If the message type is
167 `Trial_param`, its `distribution` field is serialized as a JSON string before converting
168 the entire object to JSON format. For other message types, the object is directly
169 converted to JSON format.
170
171 Args:
172 message: The message object to be serialized to JSON.
173
174 Returns:
175 str: A JSON string representation of the message.
176 """
177 if isinstance(message, Trial_param):
178 data = asdict(message)
179 # Serialize 'distribution' field as JSON string
180 data['distribution'] = json.dumps(data['distribution'])
181 return json.dumps(data, separators=(',', ':'))
182 else:
183 data = asdict(message)
184 return json.dumps(data, separators=(',', ':'))

312 """
313 Handles logging of study, trial, parameter, and value information to a file.
314
315 This class facilitates logging structured data for studies and their associated
316 trials into a file. Each method corresponds to specific types of log entries such
317 as study start, trial start, trial parameters, and trial values. It ensures the
318 data is written in real-time and provides methods for handling batch operations
319 where multiple workers or trials need to be recorded.
320
321 Attributes:
322 filename (str): Path to the file where journal entries are logged.
323 file (TextIO): File object used for writing journal entries.
324 """
325 def __init__(self, filename, study_name, dir):
326 """
327 Initializes an instance for managing file operations and conducting initial study setup.
328
329 Args:
330 filename (str): Name of the file to be created and written into.
331 study_name (str): Name of the study to be initialized in the setup process.
332 dir (str): Directory path for the study workspace or initialization.
333 """
334 self.filename = filename
335 self.file = open(self.filename, 'w')
336 self.study("main", study_name, dir)

399 def write_x(self, worker_id, trial_id, x, bounds):
400 """
401 Writes parameter values for a given trial and worker.
402
403 This method iterates through the input vector `x` and assigns parameter values
404 for corresponding indices, using lower and upper bounds provided in `bounds`.
405
406 Args:

415 def write_xs(self, trial_id, xs, bounds):
416 """
417 Writes a set of parameter values (xs) for different workers associated with a trial.
418
419 This method distributes parameter values to workers by iterating through the
420 list `xs`. Each parameter value in `xs` is assigned to a worker identified by
421 an incremental ID. The `write_x` method is invoked for each worker, linking the
422 trial, worker, and provided bounds.
423
424 Args:
425 trial_id (int): Identifier for the trial to which the parameters belong.
426 xs (list): A list of parameter values to be assigned to workers.
427 bounds (tuple): Bounds within which the parameter values are constrained.
428 """
429 for worker_id, x in enumerate(xs):
430 self.write_x(str(worker_id+1), trial_id+worker_id, x, bounds)

432 def write_ys(self, trial_id, ys):
433 """
434 Writes a list of values to corresponding trial IDs for all workers.
435
436 Args:
437 trial_id (int): The base trial ID to which worker IDs and their corresponding
438 values will be associated.
439 ys (list): A list of values to be written for each worker. The values in this
440 list correspond to workers incrementally starting from worker ID 1.
441 """
442 for worker_id, y in enumerate(ys):

473class journal_wrapper(object):
474 """
475 A wrapper class for handling journaling operations related to batch processing in a
476 parallelized optimization or search study.
477
478 This class is responsible for managing concurrent journal entries that include trial
479 start information, parameters, and trial evaluation results. It coordinates the storage,
480 formatting, and writing of trial-related data into a journal file to support analytics
481 or dashboards for the study. Synchronization mechanisms ensure thread-safe operations
482 in multi-process environments.
483
484 Attributes:
485 fit (callable): A function that evaluates the input parameters `x` and returns the
486 corresponding result.
487 bounds (object): An object that contains lower (`lb`) and upper (`ub`) bounds for
488 parameters.
489 journal (Journal): An instance of the `Journal` class used for managing and writing
490 to the journal file.
491 study_id (str): A unique identifier for the entire study.
492 batch_size (int): The number of trials to process in a batch.
493 """

495 def __init__(self, fit, bounds, jfname, study_name, study_id, batch_size):
496 """
497 Initializes the object with the given parameters and sets up the required attributes.
498
499 Args:
500 fit: Callable to perform fitting or evaluation tasks.
501 bounds: Iterable specifying the bounds or constraints for the fitting
502 or evaluation process.
503 jfname: String specifying the file name for the journal.
504 study_name: String representing the name of the study.
505 study_id: Integer representing the identifier of the study.

579 def __call__(self, x):
580 """
581 Executes the callable functionality, managing and processing a batch of inputs
582 and outputs concurrently, while ensuring proper thread safety and data
583 handling.
584
585 Args:
586 x: Input value to process and fit.
587
588 Returns:
589 The processed output value as a result of `fit(x)`. If an exception occurs,
590 returns the maximum possible float value as defined in `sys.float_info.max`.
591
592 Raises:
593 Not explicitly raised in documentation but exceptions are caught
594 within the method.
595 """
596 try:
597 with self.lock:
598 n = self.batch_size
599 if len(self.ys) >= n:
600 for i in range(n):
601 self.journal.file.write(self.starts.pop(0))
602 for i in range(n):
603 self.journal.file.write(self.xs.pop(0))
604 for i in range(n):
605 self.journal.file.write(self.ys.pop(0))
606 self.journal.file.flush()
607 trial_id = self.trial_id.value
608 self.trial_id.value += 1
609
610 worker_id = str(trial_id % self.batch_size)
611 self.store_start(worker_id, self.study_id)
612 self.store_x(worker_id, trial_id, x)
613 y = self.fit(x)
614 self.store_y(worker_id, trial_id, y)
615 return y
616 except Exception as ex:
617 print(str(ex))
618 return sys.float_info.max
