6// Eigen based implementation of active CMA-ES
7
8// Supports parallel fitness function evaluation.
9//
10// For expensive objective functions (e.g. machine learning parameter optimization) use the workers
11// parameter to parallelize objective function evaluation. The workers parameter should be limited
12// the population size because otherwize poulation update is delayed.
13
14// Derived from http://cma.gforge.inria.fr/cmaes.m which follows
15// https://www.researchgate.net/publication/227050324_The_CMA_Evolution_Strategy_A_Comparing_Review
16
17// Requires Eigen version >= 3.4 because new slicing capabilities are used, see
18// https://eigen.tuxfamily.org/dox-devel/group__TutorialSlicingIndexing.html
19// requires https://github.com/bab2min/EigenRand for random number generation.

34static ivec inverse(const ivec &indices) {
35 ivec inverse = ivec(indices.size());
36 for (int i = 0; i < indices.size(); i++)
37 inverse(indices(i)) = i;
38 return inverse;
39}

41static vec sequence(double start, double end, double step) {
42 int size = (int) ((end - start) / step + 1);
43 vec d(size);
44 double value = start;
45 for (int r = 0; r < size; r++) {
46 d(r) = value;
47 value += step;
48 }
49 return d;
50}

56 AcmaesOptimizer(long runid_, Fitness *fitfun_, int popsize_, int mu_,
57 const vec &guess_, const vec &inputSigma_, int maxEvaluations_,
58 double accuracy_, double stopfitness_, double stopTolHistFun_,
59 int update_gap_, long seed) {
60 // runid used for debugging / logging
61 runid = runid_;
62 // fitness function to minimize
63 fitfun = fitfun_;
64 // initial guess for the arguments of the fitness function
65 guess = guess_;
66 // accuracy = 1.0 is default, > 1.0 reduces accuracy
67 accuracy = accuracy_;
68 // number of objective variables/problem dimension
69 dim = guess_.size();
70 // population size, offspring number. The primary strategy parameter to play
71 // with, which can be increased from its default value. Increasing the
72 // population size improves global search properties in exchange to speed.
73 // Speed decreases, as a rule, at most linearly with increasing population
74 // size. It is advisable to begin with the default small population size.
75 if (popsize_ > 0)
76 popsize = popsize_;
77 else
78 popsize = 4 + int(3. * log(dim));
79 // individual sigma values - initial search volume. inputSigma determines
80 // the initial coordinate wise standard deviations for the search. Setting
81 // SIGMA one third of the initial search region is appropriate.
82 if (inputSigma_.size() == 1)
83 inputSigma = vec::Constant(dim, inputSigma_[0]);
84 else
85 inputSigma = inputSigma_;
86 // overall standard deviation - search volume.
87 sigma = inputSigma.maxCoeff();
88 // termination criteria
89 // maximal number of evaluations allowed.
90 maxEvaluations = maxEvaluations_;
91 // limit for fitness value.
92 stopfitness = stopfitness_;
93 // stop if x-changes larger stopTolUpX.
94 stopTolUpX = 1e3 * sigma;
95 // stop if x-change smaller stopTolX.
96 stopTolX = 1e-11 * sigma * accuracy;
97 // stop if fun-changes smaller stopTolFun.
98 stopTolFun = 1e-12 * accuracy;
99 // stop if back fun-changes smaller stopTolHistFun.
100 stopTolHistFun = stopTolHistFun_ < 0 ? 1e-13 * accuracy : stopTolHistFun_;
101 // selection strategy parameters
102 // number of parents/points for recombination.
103 mu = mu_ > 0 ? mu_ : popsize / 2;
104 // array for weighted recombination.
105 weights = (log(sequence(1, mu, 1).array()) * -1.) + log(mu + 0.5);
106 double sumw = weights.sum();
107 double sumwq = weights.squaredNorm();
108 weights *= 1. / sumw;
109 // variance-effectiveness of sum w_i x_i.
110 mueff = sumw * sumw / sumwq;
111
112 // dynamic strategy parameters and constants
113 // cumulation constant.
114 cc = (4. + mueff / dim) / (dim + 4. + 2. * mueff / dim);
115 // cumulation constant for step-size.
116 cs = (mueff + 2.) / (dim + mueff + 3.);
117 // damping for step-size.
118 damps = (1. + 2. * std::max(0., sqrt((mueff - 1.) / (dim + 1.)) - 1.))
119 * max(0.3,
120 1. - // modification for short runs
121 dim / (1e-6 + (maxEvaluations/popsize)))
122 + cs; // minor increment
123 // learning rate for rank-one update.
124 ccov1 = 2. / ((dim + 1.3) * (dim + 1.3) + mueff);
125 // learning rate for rank-mu update'
126 ccovmu = min(1. - ccov1,
127 2. * (mueff - 2. + 1. / mueff)
128 / ((dim + 2.) * (dim + 2.) + mueff));
129 // expectation of ||N(0,I)|| == norm(randn(N,1)).
130 chiN = sqrt(dim) * (1. - 1. / (4. * dim) + 1 / (21. * dim * dim));
131 ccov1Sep = min(1., ccov1 * (dim + 1.5) / 3.);
132 ccovmuSep = min(1. - ccov1, ccovmu * (dim + 1.5) / 3.);
133 // lazy covariance update gap
134 lazy_update_gap =
135 update_gap_ >= 0 ?
136 update_gap_ :
137 1.0 / (ccov1 + ccovmu + 1e-23) / dim / 10.0;
138 // CMA internal values - updated each generation
139 // objective variables.
140 xmean = fitfun->encode(guess);
141 // evolution path.
142 pc = zeros(dim);
143 // evolution path for sigma.
144 ps = zeros(dim);
145 // norm of ps, stored for efficiency.
146 normps = ps.norm();
147 // coordinate system.
148 B = Eigen::MatrixXd::Identity(dim, dim);
149 // diagonal of sqrt(D), stored for efficiency.
150 diagD = inputSigma / sigma;
151 diagC = diagD.cwiseProduct(diagD);
152 // B*D, stored for efficiency.
153 BD = B.cwiseProduct(diagD.transpose().replicate(dim, 1));
154 // covariance matrix.
155 C = B * (Eigen::MatrixXd::Identity(dim, dim) * B.transpose());
156 // number of iterations.
157 iterations = 1;
158 // size of history queue of best values.
159 historySize = 10 + int(3. * 10. * dim / popsize);
160 // stop criteria
161 stop = 0;
162 // best value so far
163 bestValue = DBL_MAX;
164 // best parameters so far
165 bestX = guess;
166 // history queue of best values.
167 fitnessHistory = vec::Constant(historySize, DBL_MAX);
168 fitnessHistory(0) = bestValue;
169 rs = new pcg64(seed);
170
171 computeArz = true;
172 fitness = vec(popsize);
173 arx = mat(dim, popsize);
174 }

180 // param zmean weighted row matrix of the gaussian random numbers generating the current offspring
181 // param xold xmean matrix of the previous generation
182 // return hsig flag indicating a small correction
183
184 bool updateEvolutionPaths(const vec &zmean, const vec &xold) {
185 ps = ps * (1. - cs) + ((B * zmean) * sqrt(cs * (2. - cs) * mueff));
186 normps = ps.norm();
187 bool hsig = normps / sqrt(1. - pow(1. - cs, 2. * iterations)) / chiN
188 < 1.4 + 2. / (dim + 1.);
189 pc *= (1. - cc);
190 if (hsig)
191 pc += (xmean - xold) * (sqrt(cc * (2. - cc) * mueff) / sigma);
192 return hsig;
193 }
194
195 // param hsig flag indicating a small correction
196 // param bestArx fitness-sorted matrix of the argument vectors producing the current offspring
197 // param arz unsorted matrix containing the gaussian random values of the current offspring
198 // param arindex indices indicating the fitness-order of the current offspring
199 // param xold xmean matrix of the previous generation
200
201 double updateCovariance(bool hsig, const mat &bestArx, const mat &arz,
202 const ivec &arindex, const mat &xold) {
203 double negccov = 0;
204 if (ccov1 + ccovmu > 0) {
205 mat arpos = (bestArx - xold.replicate(1, mu)) * (1. / sigma); // mu difference vectors
206 mat roneu = pc * pc.transpose() * ccov1;
207 // minor correction if hsig==false
208 double oldFac = hsig ? 0 : ccov1 * cc * (2. - cc);
209 oldFac += 1. - ccov1 - ccovmu;
210 // Adapt covariance matrix C active CMA
211 negccov = (1. - ccovmu) * 0.25 * mueff
212 / (pow(dim + 2., 1.5) + 2. * mueff);
213 double negminresidualvariance = 0.66;
214 // keep at least 0.66 in all directions, small popsize are most critical
215 double negalphaold = 0.5; // where to make up for the variance loss,
216 // prepare vectors, compute negative updating matrix Cneg
217 ivec arReverseIndex = arindex.reverse();
218 mat arzneg = arz(Eigen::indexing::all, arReverseIndex.head(mu));
219 vec arnorms = arzneg.colwise().norm();
220 ivec idxnorms = sort_index(arnorms);
221 vec arnormsSorted = arnorms(idxnorms);
222 ivec idxReverse = idxnorms.reverse();
223 vec arnormsReverse = arnorms(idxReverse);
224 arnorms = arnormsReverse.cwiseQuotient(arnormsSorted);
225 vec arnormsInv = arnorms(inverse(idxnorms));
226 mat sqarnw = arnormsInv.cwiseProduct(arnormsInv).transpose()
227 * weights;
228 double negcovMax = (1. - negminresidualvariance) / sqarnw(0);
229 if (negccov > negcovMax)
230 negccov = negcovMax;
231 arzneg = arzneg.cwiseProduct(
232 arnormsInv.transpose().replicate(dim, 1));
233 mat artmp = BD * arzneg;
234 mat Cneg = artmp * weights.asDiagonal() * artmp.transpose();
235 oldFac += negalphaold * negccov;
236 C = (C * oldFac) + roneu
237 + (arpos * (ccovmu + (1. - negalphaold) * negccov)
238 * weights.replicate(1, dim).cwiseProduct(
239 arpos.transpose())) - (Cneg * negccov);
240 }
241 return negccov;
242 }
243
244 // Update B and diagD from C
245 // param negccov Negative covariance factor.
246
247 void updateBD(double negccov) {
248
249 if (ccov1 + ccovmu + negccov > 0
250 && (std::fmod(iterations,
251 1. / (ccov1 + ccovmu + negccov) / dim / 10.)) < 1.) {
252 // to achieve O(N^2) enforce symmetry to prevent complex numbers
253 mat triC = C.triangularView<Eigen::Upper>();
254 mat triC1 = C.triangularView<Eigen::StrictlyUpper>();
255 C = triC + triC1.transpose();
256 Eigen::SelfAdjointEigenSolver<mat> sades;
257 sades.compute(C);
258 // diagD defines the scaling
259 diagD = sades.eigenvalues();
260 B = sades.eigenvectors();
261 if (diagD.minCoeff() <= 0) {
262 for (int i = 0; i < dim; i++)
263 if (diagD(i, 0) < 0)
264 diagD(i, 0) = 0.;
265 double tfac = diagD.maxCoeff() / 1e14;
266 C += Eigen::MatrixXd::Identity(dim, dim) * tfac;
267 diagD += vec::Constant(dim, 1.0) * tfac;
268 }
269 if (diagD.maxCoeff() > 1e14 * diagD.minCoeff()) {
270 double tfac = diagD.maxCoeff() / 1e14 - diagD.minCoeff();
271 C += Eigen::MatrixXd::Identity(dim, dim) * tfac;
272 diagD += vec::Constant(dim, 1.0) * tfac;
273 }
274 diagC = C.diagonal();
275 diagD = diagD.cwiseSqrt(); // D contains standard deviations now
276 BD = B.cwiseProduct(diagD.transpose().replicate(dim, 1));
277 }
278 }

280 mat ask_all() { // undecoded
281 // generate popsize offspring.
282 arz = normal(dim, popsize, *rs);
283 mat xs(dim, popsize);
284 for (int k = 0; k < popsize; k++) {
285 vec delta = (BD * arz.col(k)) * sigma;
286 xs.col(k) = fitfun->getClosestFeasibleNormed(xmean + delta);
287 }
288 computeArz = false;
289 return xs;
290 }

315 vec ask() {
316 // ask for one new argument vector.
317 vec arz1 = normalVec(dim, *rs);
318 vec delta = (BD * arz1) * sigma;
319 vec arx1 = fitfun->getClosestFeasibleNormed(xmean + delta);
320 computeArz = true;
321 return arx1;
322 }
323
324 int tell(double y, const vec &x) {
325 //tell function value for a argument list retrieved by ask_one().
326 fitness[told] = isfinite(y) ? y : DBL_MAX;
327 arx.col(told) = x;
328 told++;
329 if (told >= popsize) {
330 xmean = fitfun->getClosestFeasibleNormed(xmean);
331 if (computeArz) {
332 try {
333 arz = (BD.inverse()
334 * ((arx - xmean.replicate(1, popsize)) / sigma));
335 } catch (std::exception &e) {
336 arz = normal(dim, popsize, *rs);
337 }
338 }
339 updateCMA();
340 told = 0;
341 iterations += 1;
342 }
343 return stop;
344 }

346 void updateCMA() {
347 // sort by fitness and compute weighted mean into xmean
348 ivec arindex = sort_index(fitness);
349 // calculate new xmean, this is selection and recombination
350 vec xold = xmean; // for speed up of Eq. (2) and (3)
351 ivec bestIndex = arindex.head(mu);
352 mat bestArx = arx(Eigen::indexing::all, bestIndex);
353 xmean = bestArx * weights;
354 mat bestArz = arz(Eigen::indexing::all, bestIndex);
355 mat zmean = bestArz * weights;
356 bool hsig = updateEvolutionPaths(zmean, xold);
357 // adapt step size sigma
358 sigma *= exp(min(1.0, (normps / chiN - 1.) * cs / damps));
359 double bestFitness = fitness(arindex(0));
360 double worstFitness = fitness(arindex(arindex.size() - 1));
361 if (bestValue > bestFitness) {
362 bestValue = bestFitness;
363 bestX = fitfun->decode(bestArx.col(0));
364 if (isfinite(stopfitness) && bestFitness < stopfitness) {
365 stop = 1;
366 return;
367 }
368 }
369 if (iterations >= last_update + lazy_update_gap) {
370 last_update = iterations;
371 double negccov = updateCovariance(hsig, bestArx, arz, arindex,
372 xold);
373 updateBD(negccov);
374 // handle termination criteria
375 vec sqrtDiagC = diagC.cwiseSqrt();
376 vec pcCol = pc;
377 for (int i = 0; i < dim; i++) {
378 if (sigma * (max(abs(pcCol[i]), sqrtDiagC[i])) > stopTolX)
379 break;
380 if (i >= dim - 1)
381 stop = 2;
382 }
383 if (stop > 0)
384 return;
385 for (int i = 0; i < dim; i++)
386 if (sigma * sqrtDiagC[i] > stopTolUpX)
387 stop = 3;
388 if (stop > 0)
389 return;
390 }
391 double historyBest = fitnessHistory.minCoeff();
392 double historyWorst = fitnessHistory.maxCoeff();
393 if (iterations > 2
394 && max(historyWorst, worstFitness)
395 - min(historyBest, bestFitness) < stopTolFun) {
396 stop = 4;
397 return;
398 }
399 if (iterations > fitnessHistory.size()
400 && historyWorst - historyBest < stopTolHistFun) {
401 stop = 5;
402 return;
403 }
404 // condition number of the covariance matrix exceeds 1e14
405 if (diagD.maxCoeff() / diagD.minCoeff() > 1e7 * 1.0 / sqrt(accuracy)) {
406 stop = 6;
407 return;
408 }
409 // adjust step size in case of equal function values (flat fitness)
410 if (bestValue == fitness[arindex[(int) (0.1 + popsize / 4.)]]) {
411 sigma *= exp(0.2 + cs / damps);
412 }
413 if (iterations > 2
414 && max(historyWorst, bestFitness)
415 - std::min(historyBest, bestFitness) == 0) {
416 sigma *= ::exp(0.2 + cs / damps);
417 }
418 // store best in history
419 for (int i = 1; i < fitnessHistory.size(); i++)
420 fitnessHistory[i] = fitnessHistory[i - 1];
421 fitnessHistory[0] = bestFitness;
422 }

424 int doOptimize() {
425
426 // -------------------- Generation Loop --------------------------------
427 iterations = 0;
428 fitfun->resetEvaluations();
429 while (fitfun->evaluations() < maxEvaluations && !fitfun->terminate()) {
430 // generate and evaluate popsize offspring
431 mat xs = ask_all();
432 vec ys(popsize);
433 fitfun->values(xs, ys); // decodes
434 told = 0;
435 for (int k = 0; k < popsize && stop == 0; k++)
436 tell(ys(k), xs.col(k)); // tell encoded
437 if (stop != 0)
438 return fitfun->evaluations();
439 }
440 return fitfun->evaluations();
441 }
442
443 int do_optimize_delayed_update(int workers) {
444 iterations = 0;
445 fitfun->resetEvaluations();
446 evaluator eval(fitfun, 1, workers);
447 vec evals_x[workers];
448 // fill eval queue with initial population
449 for (int i = 0; i < workers; i++) {
450 vec x = ask();
451 vec xdec = fitfun->decode(x);
452 eval.evaluate(xdec, i);
453 evals_x[i] = x; // encoded
454 }
455 while (fitfun->evaluations() < maxEvaluations) {
456 vec_id* vid = eval.result();
457 vec y = vec(vid->_v);
458 int p = vid->_id;
459 delete vid;
460 vec x = evals_x[p];
461 tell(y(0), x); // tell evaluated encoded x
462 if (fitfun->evaluations() >= maxEvaluations || stop != 0)
463 break;
464 x = ask();
465 eval.evaluate(x, p);
466 evals_x[p] = x;
467 }
468 return fitfun->evaluations();
469 }

560void optimizeACMA_C(long runid, callback_type func, callback_parallel func_par, int dim,
561 double *init, double *lower, double *upper, double *sigma,
562 int maxEvals, double stopfitness, double stopTolHistFun, int mu, int popsize, double accuracy,
563 long seed, bool normalize, bool use_delayed_update, int update_gap, int workers, double* res) {
564
565 vec guess(dim), lower_limit(dim), upper_limit(dim), inputSigma(dim);
566 for (int i = 0; i < dim; i++) {// guess is mandatory
567 guess[i] = init[i];
568 inputSigma[i] = sigma[i];
569 }
570 if (lower != NULL && upper != NULL) {
571 for (int i = 0; i < dim; i++) {
572 guess[i] = init[i];
573 lower_limit[i] = lower[i];
574 upper_limit[i] = upper[i];
575 }
576 } else {
577 lower_limit.resize(0);
578 upper_limit.resize(0);
579 normalize = false;
580 }
581
582 Fitness fitfun(func, func_par, dim, 1, lower_limit, upper_limit);
583 fitfun.setNormalize(normalize);
584
585 AcmaesOptimizer opt(runid, &fitfun, popsize, mu, guess, inputSigma,
586 maxEvals, accuracy, stopfitness, stopTolHistFun, update_gap, seed);
587 try {
588 int evals = 0;
589 if (workers > 1 && use_delayed_update)
590 evals = opt.do_optimize_delayed_update(workers);
591 else
592 evals = opt.doOptimize();
593 vec bestX = opt.getBestX();
594 double bestY = opt.getBestValue();
595 for (int i = 0; i < dim; i++)
596 res[i] = bestX[i];
597 res[dim] = bestY;
598 res[dim + 1] = evals;
599 res[dim + 2] = opt.getIterations();
600 res[dim + 3] = opt.getStop();
601 } catch (std::exception &e) {
602 cout << e.what() << endl;
603 }
604}

617 for (int i = 0; i < dim; i++) {

619 lower_limit[i] = lower[i];

633 return (uintptr_t) opt;

666int tellXACMA_C(uintptr_t ptr, double* ys, double* xs) {
