6# A compound gear train is to be designed to achieve a specific gear ratio between the
7# driver and driven shafts. The objective of the gear train design is to find the number
8# of teeth in each of the four gears so as to minimize
9# - the error between the obtained gear ratio and a required gear rafio of 1/6.931 and
10# - the maximum size of any of the four gears.
11# Since the number of teeth must be integers, all four variables are strictly integers.
12# This example is taken from https://link.springer.com/chapter/10.1007/3-540-45356-3_84
13
14# This example illustrates that for relatively simple tasks no dedicated mixed integer
15# solver is required. 500 solutions with different upper bounds are computed in about 80 sec
16# on an AMD 3950x 16 core processor. The continuous input vector is made discrete by
17# calling '_feasible(x)' in the objective function 'feasable_ratio()'. The second objective
18# is modeled as boxed boundary. By incrementally increasing the bounds on the teeth number
19# the whole pareto front is listed.
20
21# Tested using https://docs.conda.io/en/main/miniconda.html on Linux Mint 21.2

37fac = 1.0
38lowerbound = [12, 12, 12, 12]

40def ratio(x):
41 return (1 / 6.931 - x[0]*x[1] / (x[2]*x[3])) ** 2
42
43# map floats to discrete integer variables
44def discrete(x):
45 feasible_x = x - x % fac
46 if feasible_x < 12:
47 feasible_x += fac
48 if feasible_x > 60:
49 feasible_x = 60
50 return feasible_x
51
52def _feasible(x):
53 x = np.array(x)
54 return [discrete(v) for v in x]
55
56def feasable_ratio(x):
57 x = _feasible(x)
58 return ratio(x)
59
60def print_result(ret, best, t0, i):
61 x = _feasible(ret.x) # make sure result is _feasible / discrete
62 val = ratio(x)
63 if val < best:
64 best = min(val, best)
65 print("{0}: time = {1:.1f} best = {2:.3E} f(xmin) = {3:.3E} x = {4:s}"
66 .format(i, dtime(t0), best, val, str(x)))
67 return best

69def test_optimizer(fun, n):

75 ret = retry.minimize(fun, bounds, max_evaluations=10000, num_retries=100)
