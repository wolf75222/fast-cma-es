138 def apply_wrench(self, rlvel=(0, 0), ravel=0):
139
140 avel = self.hand.angularVelocity
141 delta_avel = ravel - avel
142 torque = self.hand.mass * delta_avel * 30.0
143 self.hand.ApplyTorque(torque, wake=True)
144
145 lvel = self.hand.linearVelocity
146 delta_lvel = b2Vec2(rlvel) - b2Vec2(lvel)
147 force = self.hand.mass * delta_lvel * self.forceunit
148 self.hand.ApplyForce(force, self.hand.position, wake=True)
149
150 def get_state(self, verbose=False):
151 state = list(self.hand.position) + [self.hand.angle] + \
152 list(self.hand.linearVelocity) + [self.hand.angularVelocity]
153 if verbose:
154 print_state = ["%.3f" % x for x in state]
155 print
156 "position, velocity: (%s), (%s) " % \
157 ((", ").join(print_state[:3]), (", ").join(print_state[3:]))
158
159 return state

223def run_simulation(world, body, body2, robot, robot2, xvel, yvel, \
224 xvel2, yvel2, rtor, rtor2, simulation_steps,
225 simulation_steps2):
226 # simulating push with fixed direction pointing from robot location to body location
227 desired_vel = np.array([xvel, yvel])
228 rvel = b2Vec2(desired_vel[0] + np.random.normal(0, 0.01), desired_vel[1] + np.random.normal(0, 0.01))
229
230 desired_vel2 = np.array([xvel2, yvel2])
231 rvel2 = b2Vec2(desired_vel2[0] + np.random.normal(0, 0.01), desired_vel2[1] + np.random.normal(0, 0.01))
232
233 tmax = np.max([simulation_steps, simulation_steps2])
234 for t in range(tmax + 100):
235 if t < simulation_steps:
236 robot.apply_wrench(rvel, rtor)
237 if t < simulation_steps2:
238 robot2.apply_wrench(rvel2, rtor2)
239 world.step()
240
241 return (list(body.position), list(body2.position))

243class PushReward:
244 def __init__(self, do_gui=False):
245
246 # domain of this function
247 self.xmin = [-5., -5., -10., -10., 2., 0., -5., -5., -10., -10., 2., 0., -5., -5.]
248 self.xmax = [5., 5., 10., 10., 30., 2.*np.pi, 5., 5., 10., 10., 30., 2.*np.pi, 5., 5.]
249
250 # starting xy locations for the two objects
251 self.sxy = (0, 2)
252 self.sxy2 = (0, -2)
253 # goal xy locations for the two objects
254 self.gxy = [4, 3.5]
255 self.gxy2 = [-4, 3.5]
256 self.do_gui = do_gui
257
258 @property
259 def f_max(self):
260 # maximum value of this function
261 return np.linalg.norm(np.array(self.gxy) - np.array(self.sxy)) \
262 + np.linalg.norm(np.array(self.gxy2) - np.array(self.sxy2))
263 @property
264 def dx(self):
265 # dimension of the input
266 return self._dx

268 def __call__(self, argv):
269 # returns the reward of pushing two objects with two robots
270 rx = float(argv[0])
271 ry = float(argv[1])
272 xvel = float(argv[2])
273 yvel = float(argv[3])
274 simu_steps = int(float(argv[4]) * 10)
275 init_angle = float(argv[5])
276 rx2 = float(argv[6])
277 ry2 = float(argv[7])
278 xvel2 = float(argv[8])
279 yvel2 = float(argv[9])
280 simu_steps2 = int(float(argv[10]) * 10)
281 init_angle2 = float(argv[11])
282 rtor = float(argv[12])
283 rtor2 = float(argv[13])
284
285 initial_dist = self.f_max
286
287 world = b2WorldInterface(self.do_gui)
288
289 oshape, osize, ofriction, odensity, bfriction, hand_shape, hand_size = \
290 'circle', 1, 0.01, 0.05, 0.01, 'rectangle', (1, 0.3)
291
292 base = make_base(500, 500, world)
293 body = create_body(base, world, 'rectangle', (0.5, 0.5), ofriction, odensity, self.sxy)
294 body2 = create_body(base, world, 'circle', 1, ofriction, odensity, self.sxy2)
295
296 robot = end_effector(world, (rx,ry), base, init_angle, hand_shape, hand_size)
297 robot2 = end_effector(world, (rx2,ry2), base, init_angle2, hand_shape, hand_size)
298 (ret1, ret2) = run_simulation(world, body, body2, robot, robot2, xvel, yvel, \
299 xvel2, yvel2, rtor, rtor2, simu_steps, simu_steps2)
300
301 ret1 = np.linalg.norm(np.array(self.gxy) - ret1)
302 ret2 = np.linalg.norm(np.array(self.gxy2) - ret2)
303 return -(initial_dist - ret1 - ret2) # negated because we minimize
