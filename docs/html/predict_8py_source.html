2'''
3This example is based on election prediction data from
4https://en.wikipedia.org/wiki/The_Keys_to_the_White_House
5
6The prediction model was developed by Allan Lichtman and Russian geophysicist Vladimir Keilis-Borok
7and uses 13 key properties to predict the outcome of an election.
8
9It can predict correctly almost all past presidential elections in the USA since 1860.
10
11This example proves, that it cannot reliably predict the 2024 election.
12
13Lichtmann assigns all his criteria the same weight (1.0) and defines a limit sum,
14which fits well with past elections.
15
16If we could apply different weights and thresholds to Lichtman's criteria,
17it suggests that the model could yield alternative predictions,
18especially for elections where the context varies from those on which the model
19was originally calibrated. This means that while the "13 Keys" may fit historical data well,
20other configurations might fit just as well but lead to different forecasts.
21Essentially, Lichtman’s model may not be the only equally valid model;
22alternative configurations could yield differing, yet valid, predictions,
23implying potential flexibility in interpreting election outcomes.
24
25To prove this statement we have to find weights for the 13 keys and corresponding limits, so that
26a) It is predicted that Harris wins 2024
27and other weights + limits, so that
28b) It is predicted that Trump wins 2024

185def read_data(fname):
186
187 with open(fname) as f:
188 lines = f.readlines()
189 data = []
190 for line in lines:
191 l = line.split()
192 if len(l) < 2:
193 continue
194 if len(l) < 6:
195 incumbant = l[-1]
196 else:
197 winner = l[-1]
198 prop_vals = np.array([is_false(s) for s in l if not is_false(s) is None])

200 #print(prop_vals, sum(prop_vals), incumbant == winner, (incumbant == winner) == (sum(prop_vals) > 5))

202 print("exclude", prop_vals, sum(prop_vals), incumbant == winner, (incumbant == winner) == (sum(prop_vals) > 5), file=sys.stderr)
203 else: # use only correct predictions:
204 data.append((prop_vals, incumbant == winner))

217# optimize the correct predictions for a given model defined by its weights + limit
218def fit(n, props, inc_is_winner, x):
219 weights = x[:-1]
220 limit = x[-1]
221 weights *= len(weights)/sum(weights) # normalize weights so that sum is num_props
222 weighted_props = weights*props
223 num_correct = sum( [ inc_is_winner[i] == (sum(weighted_props[i]) > limit) for i in range(n)] )
224 return -num_correct

256# optimize the correct predictions for a given model defined by its weights + limit
257# maximize an epsilon around the limit to improve robustness
258def fit_eps(n, props, inc_is_winner, x):
259 weights = x[:-2]
260 eps = x[-1]
261 limit = x[-2]
262 weights *= len(weights)/sum(weights) # normalize weights so that sum is num_props
263 weighted_props = weights*props
264 num_correct1 = sum( [ inc_is_winner[i] == (sum(weighted_props[i]) > limit + eps) for i in range(n)] )
265 num_correct2 = sum( [ (not inc_is_winner[i]) == (sum(weighted_props[i]) < limit - eps) for i in range(n)] )
266 return - (num_correct1 + num_correct2 + 0.1*eps)
