88 void doOptimize() {
89
90 // -------------------- Generation Loop --------------------------------
91 for (iterations = 1; fitfun->evaluations() < maxEvaluations;
92 iterations++) {
93 int stallCount = optimize(rnd);
94 if (getBestCost() < stopfitness) {
95 stop = 1;
96 break;
97 }
98 if (stallCriterion > 0 && stallCount > stallCriterion*128*dim) {
99 stop = 2;
100 break;
101 }
102 }
103 }
