20

25
26 Enables the comparison of DE and NSGA-II population update mechanism with everything else
27 kept completely identical.

30
31 Uses the following deviation from the standard DE algorithm:
32 a) oscillating CR/F parameters.
33
34 You may keep parameters F and CR at their defaults since this implementation works well with the given settings for most problems,
35 since the algorithm oscillates between different F and CR settings.
36

255 def add_result(self, x, y):
256 """
257 Adds a result to the storage, ensuring that the storage capacity is not exceeded.
258 If the storage reaches its capacity, it performs truncation and stores the new values accordingly.
259
260 Args:
261 x: The input data point to be added.
262 y: The corresponding result or label associated with `x`.
263 """
264 with self.add_mutex:
265 self.num_added.value += 1
266 if self.num_stored.value >= self.capacity:
267 self.get_front(update=True)
268 if self.num_stored.value >= self.capacity:
269 n = int(self.num_stored.value/2)

271 self.ys_view[:n] = self.ys_view[:2*n:2]
272 self.num_stored.value = n
273 i = self.num_stored.value
274 self.xs_view[i] = x
275 self.ys_view[i] = y[:self.nobj]
276 self.num_stored.value = i + 1

278 def add_results(self, xs, ys):
279 """
280 Adds the content of the provided `xs` and `ys` to the storage, while managing the storage
281 capacity and ensuring proper synchronization.
282
283 If the storage exceeds 90% of its capacity during the operation, the addition process will
284 terminate early to avoid overfilling. The addition process is thread-safe due to `add_mutex`.
285
286 Args:
287 xs: A sequence containing the elements to be added to the storage.
288 ys: A sequence of sequences, where each sub-sequence represents corresponding
289 entries to `xs` and is truncated to the first `nobj` elements during storage.
290 """
291 with self.add_mutex:
292 self.num_added.value += 1
293 i = self.num_stored.value
294 for j in range(len(xs)):
295 if i < self.capacity:
296 self.xs_view[i] = xs[j]
297 self.ys_view[i] = ys[j][:self.nobj]
298 i += 1
299 else:
300 self.get_front(update=True)
301 i = self.num_stored.value
302 if i > 0.9*self.capacity: # give up
303 return
304 self.num_stored.value = i

306 def get_front(self, update=False):
307 """
308 Retrieves the Pareto front from stored values of `xs_view` and `ys_view`. Optionally, updates the
309 values in-place if `update` is set to True.
310
311 Args:
312 update (bool): Determines whether the current views and stored count are updated with the
313 Pareto front values.
314
315 Returns:
316 tuple: Contains two elements `(xf, yf)` where:
317 `xf` - The x-coordinates of the Pareto front.
318 `yf` - The y-coordinates of the Pareto front.
319 """
320 stored = self.num_stored.value
321 xs = self.xs_view[:stored]
322 ys = self.ys_view[:stored]
323 xf, yf = moretry.pareto(xs, ys)
324 if update:
325 n = len(yf)
326 self.xs_view[:n] = xf
327 self.ys_view[:n] = yf
328 self.num_stored.value = n
329 return xf, yf

564 def minimize_ser(self,
565 fun: Callable[[ArrayLike], ArrayLike],
566 max_evaluations: Optional[int] = 100000) -> Tuple[np.ndarray, np.ndarray]:
567 """
568 Minimizes a given function using a custom optimization approach until the maximum number of evaluations
569 is reached. The function utilizes a population-based strategy for optimization, where potential solutions
570 are evaluated, and the results are used to iteratively improve the population.
571
572 Args:
573 fun: Callable function to be minimized. It takes an array-like input and returns an array-like output.
574 max_evaluations: Optional maximum number of function evaluations to perform. Defaults to 100000.
575
576 Returns:
577 A tuple consisting of the final population of solutions (xs) and their corresponding evaluated
578 results (ys).
579 """
580 evals = 0
581 while evals < max_evaluations:
582 xs = self.ask()
583 ys = np.array([fun(x) for x in xs])
584 self.tell(ys)
585 evals += self.popsize
586 return xs, ys

589 def minimize_par(self,
590 fun: Callable[[ArrayLike], ArrayLike],
591 max_evaluations: Optional[int] = 100000,
592 workers: Optional[int] = mp.cpu_count()) -> Tuple[np.ndarray, np.ndarray]:
593 """
594 Minimizes the given multi-objective function in parallel, using a specified number
595 of workers and an evaluation limit. This method leverages a parallel computation
596 framework to evaluate the function across multiple populations simultaneously,
597 improving efficiency for computationally expensive optimization tasks.
598
599 Args:
600 fun (Callable[[ArrayLike], ArrayLike]): A function to be minimized. It takes
601 an array-like input and returns an array-like output.
602 max_evaluations (Optional[int]): The maximum number of function evaluations
603 to perform. Defaults to 100,000 if not specified.
604 workers (Optional[int]): The number of parallel workers to use for function
605 evaluation. Defaults to the number of CPU cores available.
606
607 Returns:
608 Tuple[np.ndarray, np.ndarray]: A tuple containing the final population of
609 candidate solutions (xs) and the corresponding function evaluations (ys).
610 """

614 xs = self.ask()
615 ys = fit(xs)
616 self.tell(ys)
617 evals += self.popsize
618 fit.stop()
619 return xs, ys

621 def pop_update(self):
622 """
623 Updates the population by performing non-dominated sorting, crowding distance sorting,
624 and variation for optimization purposes. This function presorts objective values if
625 only one objective exists, reallocates population members based on domination level,
626 and sorts for crowding when necessary. It also applies variation operators to update
627 the population diversity.
628
629 Raises:
630 ValueError: If the total population size exceeds the specified `popsize`.
631
632 Args:
633 None
634
635 Returns:
636 None
637 """
638 y0 = self.y
639 x0 = self.x
640 if self.nobj == 1:
641 yi = np.flip(np.argsort(self.y[:,0]))
642 y0 = self.y[yi]
643 x0 = self.x[yi]
644 domination, self.ycon, self.eps = pareto_domination(y0, self.nobj, self.ncon, self.ycon, self.eps)
645 x = []
646 y = []
647 maxdom = int(max(domination))
648 for dom in range(maxdom, -1, -1):
649 domlevel = [p for p in range(len(domination)) if domination[p] == dom]
650 if len(domlevel) == 0:
651 continue
652 if len(x) + len(domlevel) <= self.popsize:
653 # whole level fits
654 x = [*x, *x0[domlevel]]
655 y = [*y, *y0[domlevel]]
656 else: # sort for crowding
657 nx = x0[domlevel]
658 ny = y0[domlevel]
659 si = [0]
660 if len(ny) > 1:
661 cd = crowd_dist(ny)
662 si = np.flip(np.argsort(cd))
663 for p in si:
664 if len(x) >= self.popsize:
665 break
666 x.append(nx[p])
667 y.append(ny[p])
668 break # we have filled popsize members
669 self.x[:self.popsize] = x[:self.popsize]
670 self.y[:self.popsize] = y[:self.popsize]
671 if self.nsga_update:
672 self.vx = variation(self.x[:self.popsize], self.lower, self.upper, self.rg,
673 pro_c = self.pro_c, dis_c = self.dis_c, pro_m = self.pro_m, dis_m = self.dis_m)

675 def _next_x(self, p):
676 """
677 Determines the next candidate solution vector based on the current population
678 and the DE (Differential Evolution) strategy, including NSGA-II or standard
679 DE/pareto/1 strategies.
680
681 This function applies specific update strategies to generate a feasible vector
682 that follows the optimization constraints and the mutation/crossover
683 mechanism. Depending on the selected strategy, it either takes from an elite
684 subset or uses various sampling methods from the entire population.
685 Differentiation is performed using a weighted mutation process on other
686 population members.
687
688 Args:
689 p (int): Index of the current individual in the population that is being
690 processed.
691
692 Returns:
693 numpy.ndarray: A feasible vector based on the mutation and crossover
694 operations, bounded within the defined lower and upper limits.
695 """
696 if self.nsga_update: # use NSGA-II update strategy.
697 x = self.vx[self.vp]
698 self.vp = (self.vp + 1) % self.popsize # only use the elite
699 return x
700 # use standard DE/pareto/1 strategy.
701 if p == 0: # switch FR / CR every generation
702 self.iterations += 1
703 self.Cr = 0.5*self.Cr0 if self.iterations % 2 == 0 else self.Cr0
704 self.F = 0.5*self.F0 if self.iterations % 2 == 0 else self.F0
705 while True:
706 if self.pareto_update > 0: # sample elite solutions
707 r1, r2 = self.rg.integers(0, self.popsize, 2)
708 rb = int(self.popsize * (self.rg.random() ** (1.0 + self.pareto_update)))
709 else:
710 # sample from whole population
711 r1, r2, rb = self.rg.integers(0, self.popsize, 3)
712 if r1 != p and r1 != rb and r1 != r2 and r2 != rb \
713 and r2 != p and rb != p:
714 break
715 xp = self.x[p]
716 xb = self.x[rb]
717 x1 = self.x[r1]
718 x2 = self.x[r2]
719 x = self._feasible(xb + self.F * (x1 - x2))
720 r = self.rg.integers(0, self.dim)
721 tr = np.array(
722 [i != r and self.rg.random() > self.Cr for i in range(self.dim)])
723 x[tr] = xp[tr]
724 if not self.modifier is None:
725 x = self.modifier(x)
726 return x.clip(self.lower, self.upper)

858 """
859 Computes the rank of objects based on their sum of ranks across all columns.

880def ranks(cons, feasible, eps):
881 """
882 Computes the ranks of constraint values, adjusted by their feasibility and scaled by the
883 fraction of violations for each constraint.
884
885 Args:
886 cons (numpy.ndarray): A 2D array of constraint values where rows represent different
887 samples and columns represent different constraints.
888 feasible (numpy.ndarray): A 1D boolean array indicating whether each row (sample) in
889 `cons` is feasible (True) or not (False).
890 eps (float): A scalar threshold value used to determine the violation of constraints.
891
892 Returns:
893 numpy.ndarray: A 1D array that contains the computed rank for each sample in `cons`.
894 """
895 ci = cons.argsort(axis=0)
896 rank = np.empty_like(ci)
897 ar = np.arange(cons.shape[0])
898 for i in range(cons.shape[1]):
899 rank[ci[:,i], i] = ar
900 rank[feasible] = 0
901 alpha = np.sum(np.greater(cons, eps), axis=1) / cons.shape[1] # violations
902 alpha = np.tile(alpha, (cons.shape[1],1)).T
903 rank = rank*alpha
904 rank = np.sum(rank, axis=1)
905 return rank

925def pareto_sort(x0, y0, nobj, ncon):
926 """
927 Sorts solutions based on Pareto dominance and crowding distance.
928
929 This function sorts a given set of solutions into levels based on Pareto
930 dominance. Within each level, solutions are further sorted by crowding
931 distance to ensure diversity. The input arrays `x0` and `y0` are reordered
932 accordingly, and the resulting sorted arrays are returned.
933
934 Args:
935 x0: Array-like structure representing the original decision variables of
936 the solutions.
937 y0: Array-like structure representing the objectives of the solutions.
938 nobj: Integer specifying the number of objectives for each solution.
939 ncon: Integer specifying the number of constraints for each solution.
940
941 Returns:
942 Tuple of two NumPy arrays:
943 - The reordered `x0` array based on Pareto dominance and crowding
944 distance.
945 - The reordered `y0` array based on Pareto dominance and crowding
946 distance.
947 """
948 domination, _, _ = pareto_domination(y0, nobj, ncon)
949 x = []
950 y = []
951 maxdom = int(max(domination))
952 for dom in range(maxdom, -1, -1):
953 domlevel = [p for p in range(len(domination)) if domination[p] == dom]
954 if len(domlevel) == 0:
955 continue
956 nx = x0[domlevel]
957 ny = y0[domlevel]
958 si = [0]
959 if len(ny) > 1:
960 cd = crowd_dist(ny)
961 si = np.flip(np.argsort(cd))
962 for p in si:
963 x.append(nx[p])
964 y.append(ny[p])
965 return np.array(x), np.array(y)

967def pareto_domination(ys, nobj, ncon, last_ycon = None, last_eps = 0):
968 """
969 Determines Pareto domination levels for a set of solutions considering objectives
970 and constraint violations. Divides solutions into feasible and infeasible sets,
971 calculates their dominance levels based on objectives and constraints, and
972 returns the relevant rankings.
973
974 Args:
975 ys (list or np.ndarray): A list or array of solutions where each solution
976 contains concatenated values of objectives and constraints.
977 nobj (int): Number of objectives in each solution.
978 ncon (int): Number of constraints in each solution.
979 last_ycon (np.ndarray, optional): Array representing constraint violations used
980 in the previous iteration for adjusting tolerance to small violations.
981 Defaults to None.
982 last_eps (float, optional): Tolerance level for constraint violations in the
983 last iteration. Defaults to 0.
984
985 Returns:
986 tuple: A tuple containing:
987 - np.ndarray: Array of Pareto domination levels for all solutions.
988 - np.ndarray: Array of adjusted constraint violations for each solution.
989 - float: Adjusted tolerance for constraint violations.
990
991 Raises:
992 None
993 """
994 if ncon == 0:
995 return pareto_levels(ys), None, 0
996 else:
997 eps = 0 # adjust tolerance to small constraint violations
998 if not last_ycon is None and np.amax(last_ycon) < 1E90:
999 eps = 0.5*(last_eps + 0.5*np.mean(last_ycon, axis=0))
1000 if np.amax(eps) < 1E-8: # ignore small eps
1001 eps = 0
1002
1003 yobj = np.array([y[:nobj] for y in ys])
1004 ycon = np.array([np.maximum(y[-ncon:], 0) for y in ys])
1005 popn = len(ys)
1006 feasible = np.less_equal(ycon, eps).all(axis=1)
1007
1008 csum = ranks(ycon, feasible, eps)
1009 if sum(feasible) > 0:
1010 csum += objranks(yobj)
1011
1012 ci = np.argsort(csum)
1013 domination = np.zeros(popn)
1014 # first pareto front of feasible solutions
1015 cy = np.fromiter((i for i in ci if feasible[i]), dtype=int)
1016 if len(cy) > 0:
1017 ypar = pareto_levels(yobj[cy])
1018 domination[cy] = ypar
1019
1020 # then constraint violations
1021 ci = np.fromiter((i for i in ci if not feasible[i]), dtype=int)
1022 if len(ci) > 0:
1023 cdom = np.arange(len(ci), 0, -1)
1024 domination[ci] += cdom
1025 if len(cy) > 0: # priorize feasible solutions
1026 domination[cy] += len(ci) + 1
1027
1028 return domination, ycon, eps

1030def pareto_levels(ys):
1031 """
1032 Determines the Pareto levels of a given set of points.
1033
1034 This function identifies the domination levels of a set of points, where the
1035 domination level of a point represents the number of other points dominating
1036 it. Domination is determined based on whether one point strictly dominates
1037 another in all dimensions.
1038
1039 Args:
1040 ys (numpy.ndarray): A 2D array where each row represents a point in
1041 a multi-dimensional space, and domination is evaluated across
1042 dimensions.
1043
1044 Returns:
1045 numpy.ndarray: A 1D array where each index corresponds to the domination
1046 level of the respective point in the input array.
1047 """
1048 popn = len(ys)
1049 pareto = np.arange(popn)
1050 index = 0 # Next index to search for
1051 domination = np.zeros(popn)
1052 while index < len(ys):
1053 mask = np.any(ys < ys[index], axis=1)
1054 mask[index] = True
1055 pareto = pareto[mask] # Remove dominated points
1056 domination[pareto] += 1
1057 ys = ys[mask]
1058 index = np.sum(mask[:index])+1
1059 return domination

1061def crowd_dist(y): # crowd distance for 1st objective
1062 """
1063 Calculates the crowding distance for the first objective in a multi-objective optimization problem.
1064
1065 This function determines for each solution how dense the neighborhood is based on the distances
1066 to its nearest neighbors in the sorted objective space. It assigns a numerical value to each
1067 solution reflecting this crowding distance.
1068
1069 Args:
1070 y (List[List[float]]): A list of solutions, where each solution is represented as a list
1071 of objective values. This function specifically uses the first objective value for
1072 the crowding distance calculation.
1073
1074 Returns:
1075 numpy.ndarray: An array of crowding distances for each solution in the input.
1076 """
1077 n = len(y)
1078 y0 = np.fromiter((yi[0] for yi in y), dtype=float)
1079 si = np.argsort(y0) # sort 1st objective
1080 y0_s = y0[si] # sorted
1081 d = y0_s[1:n] - y0_s[0:n-1] # neighbor distance
1082 if max(d) == 0:
1083 return np.zeros(n)
1084 dsum = np.zeros(n)
1085 dsum += np.array(list(d) + [0]) # distance to left
1086 dsum += np.array([0] + list(d)) # distance to right
1087 dsum[0] = 1E99 # keep borders
1088 dsum[-1] = 1E99
1089 ds = np.empty(n)
1090 ds[si] = dsum # inverse order
1091 return ds

1094def variation(pop, lower, upper, rg, pro_c = 1, dis_c = 20, pro_m = 1, dis_m = 20):
1095 """
1096 Applies genetic variation operations, including simulated binary crossover (SBX) and polynomial
1097 mutation, on a population to produce offspring. The method modifies the population based on crossover
1098 and mutation probabilities, as well as distribution control parameters.
1099
1100 Args:
1101 pop: numpy.ndarray
1102 Input population array, where each row represents an individual and each column represents
1103 a design variable.
1104 lower: numpy.ndarray
1105 Lower bounds for each design variable.
1106 upper: numpy.ndarray
1107 Upper bounds for each design variable.
1108 rg: numpy.random.Generator
1109 Random number generator instance.
1110 pro_c: float, optional
1111 Probability of performing crossover, default is 1.
1112 dis_c: float, optional
1113 Distribution index for crossover, default is 20.
1114 pro_m: float, optional
1115 Probability of performing mutation, default is 1.
1116 dis_m: float, optional
1117 Distribution index for mutation, default is 20.
1118
1119 Returns:
1120 numpy.ndarray
1121 Modified population (offspring) after applying crossover and mutation operations.
1122 """
1123 dis_c *= 0.5 + 0.5*rg.random() # vary spread factors randomly
1124 dis_m *= 0.5 + 0.5*rg.random()
1125 pop = pop[:(len(pop) // 2) * 2][:]
1126 (n, d) = np.shape(pop)
1127 parent_1 = pop[:n // 2, :]
1128 parent_2 = pop[n // 2:, :]
1129 beta = np.zeros((n // 2, d))
1130 mu = rg.random((n // 2, d))
1131 beta[mu <= 0.5] = np.power(2 * mu[mu <= 0.5], 1 / (dis_c + 1))
1132 beta[mu > 0.5] = np.power(2 * mu[mu > 0.5], -1 / (dis_c + 1))
1133 beta = beta * ((-1)** rg.integers(2, size=(n // 2, d)))
1134 beta[rg.random((n // 2, d)) < 0.5] = 1
1135 if pro_c < 1.0:
1136 beta[np.tile(rg.random((n // 2, 1)) > pro_c, (1, d))] = 1
1137 parent_mean = (parent_1 + parent_2) * 0.5
1138 parent_diff = (parent_1 - parent_2) * 0.5
1139 offspring = np.vstack((parent_mean + beta * parent_diff, parent_mean - beta * parent_diff))
1140 site = rg.random((n, d)) < pro_m / d
1141 mu = rg.random((n, d))
1142 temp = site & (mu <= 0.5)
1143 lower, upper = np.tile(lower, (n, 1)), np.tile(upper, (n, 1))
1144 norm = (offspring[temp] - lower[temp]) / (upper[temp] - lower[temp])
1145 offspring[temp] += (upper[temp] - lower[temp]) * \
1146 (np.power(2. * mu[temp] + (1. - 2. * mu[temp]) * np.power(1. - norm, dis_m + 1.),
1147 1. / (dis_m + 1)) - 1.)
1148 temp = site & (mu > 0.5)
1149 norm = (upper[temp] - offspring[temp]) / (upper[temp] - lower[temp])
1150 offspring[temp] += (upper[temp] - lower[temp]) * \
1151 (1. - np.power(
1152 2. * (1. - mu[temp]) + 2. * (mu[temp] - 0.5) * np.power(1. - norm, dis_m + 1.),
1153 1. / (dis_m + 1.)))
1154 offspring = np.clip(offspring, lower, upper)
1155 return offspring
