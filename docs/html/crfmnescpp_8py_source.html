53def minimize(fun: Callable[[ArrayLike], float],
54 bounds: Optional[Bounds] = None,
55 x0: Optional[ArrayLike] = None,
56 input_sigma = 0.3,
57 popsize = 32,
58 max_evaluations = 100000,
59 workers = None,
60 stop_fitness = -np.inf,
61 rg = Generator(PCG64DXSM()),
62 runid=0,
63 normalize = False,
64 use_constraint_violation = True,
65 penalty_coef = 1E5
66 ) -> OptimizeResult:
67
68 """
69 Minimizes a given objective function using the Covariance Matrix Adaptation Evolution
70 Strategy with Constraint Handling (CR-FM-NES). The optimization adjusts variables within the
71 provided bounds to achieve the minimum value of the given objective function.
72
73 Args:
74 fun: The objective function to be minimized. It should be a callable that takes an
75 array-like structure as input and returns a float value as output.
76 bounds: Optional bounds for the variables as a `Bounds` object. If not specified,
77 the search is unbounded.
78 x0: Optional initial guess for the independent variables as an array-like structure.
79 If not provided, it is created randomly within the bounds.
80 input_sigma: Initial step size(s) for the search. It can be a float or callable
81 providing an initial sigma value. If multi-dimensional, the mean is used.
82 popsize: The population size for the evolution. Must be an even number. If not
83 specified, defaults to 32.
84 max_evaluations: The maximum number of function evaluations allowed during optimization.
85 workers: Number of parallel workers to use for evaluation. If `None` or `workers` <= 1,
86 no parallelism is applied.
87 stop_fitness: The fitness value at which the optimization halts if surpassed.
88 Default is negative infinity.
89 rg: A random generator to control stochastic behavior. Defaults to an instance of
90 `Generator` with a `PCG64DXSM` bit generator.
91 runid: An identifier for this particular optimization run.
92 normalize: A boolean indicating whether to normalize the variables to the [0, 1] range
93 during optimization.
94 use_constraint_violation: Whether to apply penalty-based handling for constraint
95 violations during optimization. Defaults to True.
96 penalty_coef: Penalty coefficient for constraint violation handling. Default is 1E5.
97
98 Returns:
99 OptimizeResult: An object containing the optimization results, such as the optimized
100 variables (`x`), the function value at the solution (`fun`), the number of function
101 evaluations performed (`nfev`), the number of iterations (`nit`), the exit status code
102 (`status`), and whether the optimization was successful (`success`).
103
104 Raises:
105 Exception: Raises an exception if the optimization failed, and an empty `OptimizeResult`
106 is returned with default attributes indicating failure.
107 """
108
109 lower, upper, guess = _check_bounds(bounds, x0, rg)
110 dim = guess.size
111 if popsize is None:
112 popsize = 32
113 if popsize % 2 == 1: # requires even popsize
114 popsize += 1
115 if callable(input_sigma):
116 input_sigma=input_sigma()
117 if np.ndim(input_sigma) > 0:
118 input_sigma = np.mean(input_sigma)
119 array_type = ct.c_double * dim
120 parfun = None if (workers is None or workers <= 1) else parallel(fun, workers)
121 c_callback_par = call_back_par(callback_par(fun, parfun))
122 res = np.empty(dim+4)
123 res_p = res.ctypes.data_as(ct.POINTER(ct.c_double))
124 try:
125 optimizeCRFMNES_C(runid, c_callback_par, dim, array_type(*guess),
126 None if lower is None else array_type(*lower),
127 None if upper is None else array_type(*upper),
128 input_sigma, max_evaluations, stop_fitness,
129 popsize, int(rg.uniform(0, 2**32 - 1)), penalty_coef,
130 use_constraint_violation, normalize, res_p)
131 x = res[:dim]
132 val = res[dim]
133 evals = int(res[dim+1])
134 iterations = int(res[dim+2])
135 stop = int(res[dim+3])
136 res = OptimizeResult(x=x, fun=val, nfev=evals, nit=iterations, status=stop, success=True)
137 except Exception as ex:
138 res = OptimizeResult(x=None, fun=sys.float_info.max, nfev=0, nit=0, status=-1, success=False)
139 if not parfun is None:
140 parfun.stop()
141 return res

143class CRFMNES_C:
144 """
145 Minimization of a scalar function of one or more variables using a C++ CR-FM-NES
146 implementation, interfaced via Python using ctypes.
147
148 This class implements the Covariance Matrix Adaptation Evolution Strategy
149 (CR-FM-NES) algorithm for numerical optimization problems. It relies on the
150 underlying C++ implementation for the actual optimization and provides a Python
151 wrapper for convenient usage. The algorithm aims to find the minimum of a scalar
152 objective function and supports various configurations like bounds on variables,
153 initialization parameters, and constraint handling.
154
155 Attributes:
156 ptr (ctypes.POINTER): Pointer to the underlying C++ object handling the optimization.
157 popsize (int): Population size used by the CMA-ES algorithm.
158 dim (int): Dimension of the decision variable vector being optimized.
159 """
160 def __init__(self,
161 dim: int,
162 bounds: Optional[Bounds] = None,
163 x0: Optional[ArrayLike] = None,
164 input_sigma: Optional[Union[float, ArrayLike, Callable]] = 0.3,
165 popsize: Optional[int] = 32,
166 rg: Optional[Generator] = Generator(PCG64DXSM()),
167 runid: Optional[int] = 0,
168 normalize: Optional[bool] = False,
169 use_constraint_violation: Optional[bool] = True,
170 penalty_coef: Optional[float] = 1E5
171 ):
172
173 """
174 Initializes an evolutionary optimization algorithm with constrained bounds and penalty-based constraint
175 handling. Uses randomized initial guess for optimization and allows configuration of the population size
176 and mutation parameters.
177
178 Args:
179 dim (int): The dimensionality of the optimization problem.
180 bounds (Optional[Bounds]): The search space boundaries. If not specified, no limits are imposed.
181 x0 (Optional[ArrayLike]): Initial guess for the optimization problem. If None, a random guess is used.
182 input_sigma (Optional[Union[float, ArrayLike, Callable]]): Initial distribution width for mutation.
183 Default is 0.3. Callable functions are invoked and their result used.
184 popsize (Optional[int]): Size of the population. Default is 32. If an odd value is provided, it is
185 incremented to ensure compatibility.
186 rg (Optional[Generator]): Random generator for sampling. Default is Generator(PCG64DXSM()).
187 runid (Optional[int]): Unique identifier for the optimization run. Default is 0.
188 normalize (Optional[bool]): Indicates whether the search space should be normalized. Default is False.
189 use_constraint_violation (Optional[bool]): Enables constraint violation handling during optimization.
190 Default is True.
191 penalty_coef (Optional[float]): Coefficient of the penalty term for constraints. Default is 1E5.
192
193 """
194
195 lower, upper, guess = _get_bounds(dim, bounds, x0, rg)
196 if popsize is None:
197 popsize = 32
198 if popsize % 2 == 1: # requires even popsize
199 popsize += 1
200 if lower is None:
201 lower = [0]*dim
202 upper = [0]*dim
203 if callable(input_sigma):
204 input_sigma=input_sigma()
205 if np.ndim(input_sigma) > 0:
206 input_sigma = np.mean(input_sigma)
207 array_type = ct.c_double * dim
208 try:
209 self.ptr = initCRFMNES_C(runid, dim, array_type(*guess),
210 array_type(*lower), array_type(*upper),
211 input_sigma, popsize, int(rg.uniform(0, 2**32 - 1)), penalty_coef,
212 use_constraint_violation, normalize)
213 self.popsize = popsize
214 self.dim = dim
215 except Exception as ex:
216 print (ex)
217 pass

232 def ask(self) -> np.ndarray:
233 """
234 Generates a population of candidate solutions using the CR-FM-NES algorithm.
235
236 This method computes a new set of candidate solutions for the optimization
237 problem using internal state and algorithm properties. It uses the specified
238 population size and dimensionality of the problem.
239
240 Returns:
241 np.ndarray: A 2D array where each row corresponds to a candidate solution.
242
243 Raises:
244 Exception: If there is an error while generating the candidate solutions.
245 """
246 try:
247 lamb = self.popsize
248 n = self.dim
249 res = np.empty(lamb*n)
250 res_p = res.ctypes.data_as(ct.POINTER(ct.c_double))
251 askCRFMNES_C(self.ptr, res_p)
252 xs = np.empty((lamb, n))
253 for p in range(lamb):
254 xs[p,:] = res[p*n : (p+1)*n]
255 return xs
256 except Exception as ex:
257 print (ex)
258 return None

338 evals = int(res[self.dim+1])
339 iterations = int(res[self.dim+2])
340 stop = int(res[self.dim+3])
341 res = OptimizeResult(x=x, fun=val, nfev=evals, nit=iterations, status=stop, success=True)
342 except Exception as ex:
343 res = OptimizeResult(x=None, fun=sys.float_info.max, nfev=0, nit=0, status=-1, success=False)
