189class wrapper(object):
190 """Wrapper for optimization evaluation and logging.
191
192 This class serves as a wrapper around a user-defined objective function
193 (`fit`). It tracks the number of function evaluations, the best evaluation
194 value found so far, and provides logging functionality. The wrapper is
195 typically used in optimization tasks to facilitate performance tracking
196 and debug information.
197
198 Attributes:
199 fit (Callable[[ArrayLike], float]): The objective function to be
200 evaluated.
201 evals (multiprocessing.sharedctypes.Synchronized): A shared integer
202 that tracks the total number of evaluations.
203 best_y (multiprocessing.sharedctypes.Synchronized): A shared double
204 used to store the best result value found so far.
205 t0 (float): The timestamp (in seconds) when the wrapper is
206 instantiated. Used for tracking elapsed time.
207 """
208
209 def __init__(self,
210 fit: Callable[[ArrayLike], float]):
211 """
212 Initializes the class with a fitness evaluation function, shared evaluation counts,
213 best result achieved, and initial time measurement.
214
215 Args:
216 fit (Callable[[ArrayLike], float]): A callable function that evaluates the fitness of
217 an array-like input and returns a float score.
218 """
219 self.fit = fit
220 self.evals = mp.RawValue(ct.c_int, 0)
221 self.best_y = mp.RawValue(ct.c_double, np.inf)
222 self.t0 = time.perf_counter()
223
224 def __call__(self, x: ArrayLike) -> float:
225 """
226 Executes the callable object with the given input `x`, tracking its evaluations and logging improvements.
227
228 This method evaluates the function with the provided input array-like `x`, updates the best result seen so far,
229 and logs the details if an improvement occurs. If an exception is raised during execution, it prints the error
230 and returns the maximum float value.
231
232 Args:
233 x (ArrayLike): The input to be evaluated by the callable object.
234
235 Returns:
236 float: The result of the evaluation, or the maximum float value in case of an exception.
237 """
238 try:
239 self.evals.value += 1
240 y = self.fit(x)
241 y0 = y if np.isscalar(y) else sum(y)
242 if y0 < self.best_y.value:
243 self.best_y.value = y0
244 logger.info(
245 f'{dtime(self.t0)} {self.evals.value} {self.evals.value/(1E-9 + dtime(self.t0)):.1f} {self.best_y.value} {list(x)}'
246 )
247 return y
248 except Exception as ex:
249 print(str(ex))
250 return sys.float_info.max

313class Sequence(Optimizer):
314 """
315 A class for sequentially combining multiple optimizers.
316
317 The Sequence class is designed to execute a series of optimizers sequentially.
318 Each optimizer in the provided sequence takes over from the result of the
319 previous one, allowing for a composite approach to optimization. The class
320 inherits from the Optimizer base class.
321
322 Attributes:
323 optimizers (ArrayLike): List of optimizer instances to be executed
324 sequentially. Each optimizer must inherit from the base Optimizer
325 class.
326 max_evaluations (int): Total number of evaluations across all optimizers
327 in the sequence.
328 name (str): Concatenated name of all optimizers in the sequence, separated
329 by arrows (' -> ').
330 """
331
332 def __init__(self, optimizers: ArrayLike):
333 """
334 Initializes the class with a list of optimizers, calculating the total number of evaluations and preparing the combined
335 optimizer name.
336
337 Each optimizer from the provided list contributes its name and maximum evaluations to compose a combined metadata
338 representation for the initialized object.
339
340 Args:
341 optimizers (ArrayLike): A list or array-like object containing optimizer instances. Each optimizer should have a
342 `name` attribute and a `max_evaluations` attribute.
343 """
344 Optimizer.__init__(self)
345 self.optimizers = optimizers
346 self.max_evaluations = 0
347 for optimizer in self.optimizers:
348 self.name += optimizer.name + ' -> '
349 self.max_evaluations += optimizer.max_evaluations
350 self.name = self.name[:-4]

352 def minimize(self,
353 fun: Callable[[ArrayLike], float],
354 bounds: Bounds,
355 guess: Optional[ArrayLike] = None,
356 sdevs: Optional[Union[float, ArrayLike, Callable]] = None,
357 rg: Optional[Generator] = Generator(PCG64DXSM()),
358 store=None) -> Tuple[np.ndarray, float, int]:
359 """
360 Minimizes a given objective function using multiple optimizers and evaluates
361 respective results to find the optimal solution. This function iteratively
362 utilizes different optimizers to search for the minimum value of the objective
363 function within specified bounds while updating the best guess during the
364 process. The total function evaluations across all optimizers are accumulated.
365
366 Args:
367 fun (Callable[[ArrayLike], float]): The objective function to minimize.
368 Must accept a variable of type ArrayLike and return a float.
369 bounds (Bounds): The bounds within which the optimization is performed.
370 guess (Optional[ArrayLike]): An optional initial guess for the optimization.
371 sdevs (Optional[Union[float, ArrayLike, Callable]]): Optional standard
372 deviations or a callable defining standard deviation that supports
373 optimization.
374 rg (Optional[Generator]): Random number generator for stochastic techniques,
375 defaults to numpy's Generator(PCG64DXSM).
376 store: Optional parameter to store intermediate results or data related
377 to the optimization process.
378
379 Returns:
380 Tuple[np.ndarray, float, int]: A tuple containing:
381 - The optimal parameters as a numpy ndarray.
382 - The minimum value of the function found.
383 - Total function evaluations across all optimizers.
384 """
385 evals = 0
386 y = np.inf
387 for optimizer in self.optimizers:
388 ret = optimizer.minimize(fun, bounds, guess, sdevs, rg, store)
389 if ret[1] < y:
390 y = ret[1]
391 x = ret[0]
392 guess = x
393 evals += ret[2]
394 return x, y, evals

435 def minimize(self,
436 fun: Callable[[ArrayLike], float],
437 bounds: Bounds,
438 guess: Optional[ArrayLike] = None,
439 sdevs: Optional[Union[float, ArrayLike, Callable]] = None,
440 rg: Optional[Generator] = Generator(PCG64DXSM()),
441 store=None) -> Tuple[np.ndarray, float, int]:
442 """
443 Minimizes a given function using one of the available optimizers chosen at random.
444
445 This method selects an optimizer from the available pool of optimizers randomnly,
446 then uses the selected optimizer to minimize the given function. The function to
447 be minimized, along with other parameters required for the optimization, are passed
448 to the chosen optimizer. The result of the optimization, including the solution,
449 minimum value, and number of iterations, is returned.
450
451 Args:
452 fun (Callable[[ArrayLike], float]): The function to minimize. This should take
453 a single argument, which is a set of parameter values, and return a scalar
454 value representing the objective to minimize.
455 bounds (Bounds): The bounds for the optimization, specifying the feasible
456 range for each parameter in the input space.
457 guess (Optional[ArrayLike]): Initial guess for the optimization. Provides
458 a starting point for the optimization process. If not provided, defaults
459 to None.
460 sdevs (Optional[Union[float, ArrayLike, Callable]]): Standard deviations or
461 a function to generate them, influencing the exploration during the
462 optimization process. If not provided, defaults to None.
463 rg (Optional[Generator]): Random number generator instance for selecting a
464 random optimizer and potentially generating random values for the optimization.
465 Defaults to a `Generator` using `PCG64DXSM`.
466 store: Optional storage object for logging or saving optimization history. The
467 structure and usage are dependent on the specific optimizers used.
468
469 Returns:
470 Tuple[np.ndarray, float, int]: A tuple containing the optimization results:
471 - A NumPy array with the parameters corresponding to the solution.
472 - A float denoting the minimum value of the objective function found.
473 - An integer indicating the number of iterations used in the optimization.
474 """
475 choice = rg.integers(0, len(self.optimizers))
476 opt = self.optimizers[choice]
477 return opt.minimize(fun, bounds, guess, sdevs, rg, store)
