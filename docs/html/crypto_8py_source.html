53# compute the exponential mean average of a time series of prices
54def get_ema(prices, period):
55 return prices.ewm(span=period,min_periods=period,adjust=False,ignore_na=False).mean().to_numpy()

57# compute the simple mean average of a time series of prices
58def get_sma(prices, period):
59 return prices.rolling(period).mean().to_numpy()

63@njit(fastmath=True)
64def buy_all(cash, num_coins, price, date, logs):
65 num = int(cash / price)
66 if not date is None:
67 logs.append(date + " cash " + str(int(cash)) # no string formatting in numba :-(
68 + " buy " + str(num)
69 + " num_coins " + str(num_coins)
70 + " price " + str(int(100*price))+ " ct") # see https://github.com/numba/numba/issues/4758
71 return cash - num*price, num_coins + num

73@njit(fastmath=True)
74def sell_all(cash, num_coins, price, date, logs):
75 if not date is None:
76 logs.append(date + " cash " + str(int(cash))
77 + " sell " + str(num_coins)
78 + " num_coins " + str(num_coins)
79 + " price " + str(int(100*price)) + " ct")
80 return cash + num_coins*price, 0

82# simulate HODL strategy
83@njit(fastmath=True)
84def hodl(close, start_cash):
85 cash, num_coins = buy_all(start_cash, 0, close[0], None, None) # buy first day
86 cash, num_coins = sell_all(cash, num_coins, close[-1], None, None) # sell last day
87 return cash / start_cash # return cash gain factor

89# simulate a simple crossing EMA/SMA trading strategy, return the resulting cash factor
90@njit(fastmath=True)
91def strategy(close, start_cash, ema, sma, wait_buy, wait_sell, dates):
92 cash = start_cash
93 num_coins = 0
94 last_trade = 0
95 logs = List(); logs.append("") # needs to be String typed
96 num_trades = 0
97 # for all trading days
98 for i in range(len(close)):
99 c_ema = ema[i]; c_sma = sma[i]; price = close[i]
100 date = dates[i] if not dates is None else None
101 if np.isnan(c_ema) or np.isnan(c_sma):
102 continue
103 # buy if current ema > current sma and > wait_buy days since last trade
104 if num_coins == 0 and c_ema > c_sma and i > last_trade + wait_buy:
105 cash, num_coins = buy_all(cash, num_coins, price, date, logs)
106 last_trade = i
107 num_trades += 1
108 # sell if current ema < current sma and > wait_sell days since last trade
109 elif num_coins > 0 and c_ema < c_sma and i > last_trade + wait_sell:
110 cash, num_coins = sell_all(cash, num_coins, price, date, logs)
111 last_trade = i
112 num_trades += 1
113
114 cash, num_coins = sell_all(cash, num_coins, price, date, logs)
115 # return cash gain factor compared with HODL strategy
116 return cash / start_cash / hodl(close, start_cash), num_trades, logs

118def simulate(prices, ema_period, sma_period, wait_buy, wait_sell, dates=None):
119 close = prices.to_numpy()
120 ema = get_ema(prices, ema_period)
121 sma = get_sma(prices, sma_period)
122 return strategy(close, START_CASH, ema, sma, wait_buy, wait_sell, dates)

126 def __init__(self, tickers, start, end, max_trades = None):
127 self.evals = mp.RawValue(ct.c_int, 0)
128 self.best_y = mp.RawValue(ct.c_double, np.inf)
129 self.t0 = time.perf_counter()
130 self.tickers = tickers
131 self.max_trades = max_trades
132 self.histories = {}
133 self.closes = {}
134 self.dates = {}
135 self.hodls = {}
136 for ticker in tickers:
137 self.histories[ticker] = get_history(ticker, start=start, end=end)
138 self.closes[ticker] = self.histories[ticker].Close
139 self.dates[ticker] = np.array([d.strftime('%Y.%m.%d') for d in self.histories[ticker].index])
140 self.hodls[ticker] = hodl(self.closes[ticker].to_numpy(), START_CASH)
141 hodls = list(self.hodls.values())
142 logger.info("hodl = {0:.3f} {1:s}"
143 .format(np.prod(hodls) ** (1.0/len(hodls)), str([round(fi,1) for fi in hodls])))
144
145 def fun(self, x):
146 # simulate the EMS/SMA strategy for all tickers
147 factors = []
148 num_trades = []
149 for ticker in self.tickers:
150 # convert the optimization variables into integers and use them to configure the simulation
151 f, num, _ = simulate(self.closes[ticker], int(x[0]), int(x[1]), int(x[2]), int(x[3]))
152 factors.append(f)
153 num_trades.append(num)
154 factor = np.prod(factors) ** (1.0/len(factors)) # normalize the accumulated factor
155 y = -factor # our optimization algorithm minimizes the resulting value, we maximize factor
156 # book keeping and logging for parallel optimization
157 self.evals.value += 1
158 if y < self.best_y.value:
159 self.best_y.value = y
160 logger.info("nsim = {0}: time = {1:.1f} fac = {2:.3f} {3:s} ntr = {4:s} x = {5:s}"
161 .format(self.evals.value, dtime(self.t0), -y,
162 str([round(fi,1) for fi in factors]),
163 str([int(ntr) for ntr in num_trades]),
164 str([int(xi) for xi in x])))
165 return y, factors, num_trades
166
167 def __call__(self, x):
168 y, _, _ = self.fun(x)
169 return y
170
171 def mofun(self, x):
172 _, factors, num_trades = self.fun(x)
173 ys = [-f for f in factors] # higher factor is better
174 constraints = [ntr - self.max_trades for ntr in num_trades] # at most max_trades trades
175 return np.array(ys + constraints)
176
177 def ndfun(self, x):
178 y, factors, _ = self.fun(x)
179 return 5+y, factors # we need positive y values for tracking QD-Score

181 def get_trades(self, ticker, x):
182 _, _, log = simulate(self.closes[ticker], int(x[0]), int(x[1]), int(x[2]), int(x[3]), self.dates[ticker])
183 trades = []
184 for i in range(1,len(log)):
185 l = log[i].split()
186 trade = {'date': l[0],
187 'cash': int(l[2]),
188 'type': l[3],
189 'traded': int(l[4]),
190 'num_coins': int(l[6]),
191 'price': int(l[8])/1000}
192 trades.append(trade)
193 logger.info('\n' + ticker)
194 for l in log: logger.info(l)
195 return trades

197 def get_values(self, ticker, x):
198 trades = self.get_trades(ticker, x)
199 cash = START_CASH
200 num_coins = 0
201 closes = self.closes[ticker]
202 dates = self.dates[ticker]
203 values = []
204 i = 0
205 for date, price in zip(dates,closes):
206 trade = trades[i]
207 tdate = trade['date']
208 if date == tdate:
209 i += 1
210 traded = trade['traded']
211 if trade['type'] == 'buy':
212 num_coins, cash = num_coins + traded, cash - traded*price
213 else:
214 num_coins, cash = num_coins - traded, cash + traded*price
215 # print(trade)
216 values.append(cash + num_coins*price)
217 return np.array(np.fromiter((v/START_CASH*closes[0] for v in values), dtype=float)) # adjust to initial stock price

248def optimize_mo(tickers, start, end, nsga_update = True):
249 nobj = len(tickers) # number of objectives
250 ncon = nobj # number of constraints
251 max_trades = 8
252 fit = fitness(tickers, start, end, max_trades)
253 bounds = Bounds([20,50,10,10], [50,100,200,200])
254 xs, front = modecpp.retry(fit.mofun, len(tickers), ncon, bounds, num_retries=24, popsize = 48,
255 max_evaluations = 16000, nsga_update = nsga_update, workers=24)
256 for y, x in zip(front, xs):
257 print("fac " + str([-round(yi,2) for yi in y[:nobj]]) +
258 " trades " + str([int(max_trades+ci) for ci in y[nobj:]]) +
259 " x = " + str([int(xi) for xi in x]))
