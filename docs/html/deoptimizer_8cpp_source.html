6// Eigen based implementation of differential evolution using on the DE/best/1 strategy.
7// Uses two deviations from the standard DE algorithm:
8// a) temporal locality introduced in
9// https://www.researchgate.net/publication/309179699_Differential_evolution_for_protein_folding_optimization_based_on_a_three-dimensional_AB_off-lattice_model
10// b) reinitialization of individuals based on their age.
11//
12// Requires Eigen version >= 3.4 because new slicing capabilities are used, see
13// https://eigen.tuxfamily.org/dox-devel/group__TutorialSlicingIndexing.html
14// requires https://github.com/bab2min/EigenRand for random number generation.
15//
16// Supports parallel fitness function evaluation.
17//
18// You may keep parameters F and CR at their defaults since this implementation works well with the given settings for most problems,
19// since the algorithm oscillates between different F and CR settings.
20//
21// For expensive objective functions (e.g. machine learning parameter optimization) use the workers
22// parameter to parallelize objective function evaluation. The workers parameter is limited by the
23// population size.
24//
25// The ints parameter is a boolean array indicating which parameters are discrete integer values. This
26// parameter was introduced after observing non optimal DE-results for the ESP2 benchmark problem:
27// https://github.com/AlgTUDelft/ExpensiveOptimBenchmark/blob/master/expensiveoptimbenchmark/problems/DockerCFDBenchmark.py
28// If defined it causes a "special treatment" for discrete variables: They are rounded to the next integer value and
29// there is an additional mutation to avoid getting stuck at local minima.

50 DeOptimizer(long runid_, Fitness *fitfun_, int dim_, int seed_,
51 int popsize_, int maxEvaluations_, double keep_,
52 double stopfitness_, double F_, double CR_,
53 double min_mutate_, double max_mutate_, bool *isInt_,
54 const vec &guess_, const vec &inputSigma_, double minSigma_) {
55 // runid used to identify a specific run
56 runid = runid_;
57 // fitness function to minimize
58 fitfun = fitfun_;
59 // Number of objective variables/problem dimension
60 dim = dim_;
61 // Population size
62 popsize = popsize_ > 0 ? popsize_ : 15 * dim;
63 // maximal number of evaluations allowed.
64 maxEvaluations = maxEvaluations_ > 0 ? maxEvaluations_ : 50000;
65 // keep best young after each iteration.
66 keep = keep_ > 0 ? keep_ : 30;
67 // Limit for fitness value.
68 stopfitness = stopfitness_;
69 F = F0 = F_ > 0 ? F_ : 0.5;
70 CR = CR0 = CR_ > 0 ? CR_ : 0.9;
71 // Number of iterations already performed.
72 iterations = 0;
73 bestY = DBL_MAX;
74 // stop criteria
75 stop = 0;
76 pos = 0;
77 rs = new pcg64(seed_);
78 // Indicating which parameters are discrete integer values. If defined these parameters will be
79 // rounded to the next integer and some additional mutation of discrete parameters are performed.
80 isInt = isInt_;
81 // DE population update parameter used in connection with isInt. Determines
82 // the mutation rate for discrete parameters.
83 min_mutate = min_mutate_ > 0 ? min_mutate_ : 0.1;
84 max_mutate = max_mutate_ > 0 ? max_mutate_ : 0.5;
85
86 useNormal = guess_.size() > 0;
87 mean = guess_;
88 sigma = inputSigma_;
89 minSigmaVal = minSigma_;
90 init();
91 }

102 return (int) (max * distr_01(*rs));

105 vec sample() {
106 if (useNormal)
107 return fitfun->getClosestFeasible(mean + (normalVec(dim, *rs).array() * sigma.array()).matrix());
108 else
109 return fitfun->sample(*rs);
110 }
111
112 double sample_i(int i) {
113 if (useNormal)
114 return fitfun->getClosestFeasible_i(i, normreal(*rs, mean[i], sigma[i]));
115 else
116 return fitfun->sample_i(i, *rs);
117 }

119 void update_mean() {
120 if (useNormal) {
121 meanHist.col(meanHistIndex) = popX.col(bestI);
122 meanHistIndex = (meanHistIndex + 1) % meanHist.cols();
123 vec delta = meanHist.rowwise().maxCoeff() - meanHist.rowwise().minCoeff();
124 vec sigma_new = delta.cwiseMin(maxSigma).cwiseMax(minSigma);
125 sigma = sigma_new.mean() > sigma.mean() ? sigma_new : 0.9 * sigma + 0.1 * sigma_new;
126 mean = 0.9 * mean + 0.1 * popX.col(bestI);
127 }
128 }

130 vec nextX(int p, const vec &xp, const vec &xb) {
131 if (p == 0) {
132 iterations++;
133 CR = iterations % 2 == 0 ? 0.5 * CR0 : CR0;
134 F = iterations % 2 == 0 ? 0.5 * F0 : F0;
135 if (iterations > 2)
136 update_mean();
137 }
138 int r1, r2;
139 do {
140 r1 = rndInt(popsize);
141 } while (r1 == p || r1 == bestI);
142 do {
143 r2 = rndInt(popsize);
144 } while (r2 == p || r2 == bestI || r2 == r1);
145 vec x1 = popX.col(r1);
146 vec x2 = popX.col(r2);
147 vec x = xb + (x1 - x2) * F;
148 int r = rndInt(dim);
149 for (int j = 0; j < dim; j++)
150 if (j != r && rnd01() > CR)
151 x[j] = xp[j];
152 vec nextx = fitfun->getClosestFeasible(x);
153 modify(nextx);
154 return nextx;
155 }
156
157 vec next_improve(const vec &xb, const vec &x, const vec &xi) {
158 vec nextx = fitfun->getClosestFeasible(xb + ((x - xi) * F0));
159 modify(nextx);
160 return nextx;
161 }
162
163 void modify(vec &x) {
164 if (isInt == NULL)
165 return;
166 double n_ints = 0;
167 for (int i = 0; i < dim; i++)
168 if (isInt[i]) n_ints++;
169 double to_mutate = min_mutate + rnd01()*(max_mutate - min_mutate);
170 for (int i = 0; i < dim; i++) {
171 if (isInt[i]) {
172 if (rnd01() < to_mutate/n_ints)
173 x[i] = (int)sample_i(i); // resample
174 }
175 }
176 }

178 vec ask(int &p) {
179 // ask for one new argument vector.
180 if (improvesX.empty()) {
181 p = pos;
182 vec x = nextX(p, popX.col(p), popX.col(bestI));
183 pos = (pos + 1) % popsize;
184 return x;
185 } else {
186 p = improvesP.front();
187 vec x = improvesX.front();
188 improvesP.pop();
189 improvesX.pop();
190 return x;
191 }
192 }

194 int tell(double y, const vec &x, int p) {
195 //tell function value for a argument list retrieved by ask_one().
196 if (isfinite(y) && y < popY[p]) {
197 if (iterations > 1) {
198 // temporal locality
199 improvesP.push(p);
200 improvesX.push(next_improve(popX.col(bestI), x, popX0.col(p)));
201 }
202 popX0.col(p) = popX.col(p);
203 popX.col(p) = x;
204 popY[p] = y;
205 popIter[p] = iterations;
206 if (y < popY[bestI]) {
207 bestI = p;
208 if (y < bestY) {
209 bestY = y;
210 bestX = x;
211 if (isfinite(stopfitness) && bestY < stopfitness)
212 stop = 1;
213 }
214 }
215 } else {
216 // reinitialize individual
217 if (keep * rnd01() < iterations - popIter[p]) {
218 popX.col(p) = sample();
219 popY[p] = DBL_MAX;
220 }
221 }
222 return stop;
223 }
224
225 mat askAll() {
226 for (int i = 0; i < popsize;) {
227 int p;
228 vec x = ask(p);
229 askedP[i] = p;
230 askedX.col(i) = x;
231 i++;
232 }
233 return askedX;
234 }
235
236 int tellAll(vec &ys) {
237 for (int i = 0; i < popsize; i++) {
238 tell(ys[i], askedX.col(i), askedP[i]);
239 }
240 //std::cout << fitfun->evaluations() << " y " << ys.transpose() << std::endl;
241 return stop;
242 }

244 void doOptimize() {
245
246 // -------------------- Generation Loop --------------------------------
247 for (iterations = 1; fitfun->evaluations() < maxEvaluations
248 && !fitfun->terminate(); iterations++) {
249
250 if (iterations > 2)
251 update_mean();
252
253 CR = iterations % 2 == 0 ? 0.5 * CR0 : CR0;
254 F = iterations % 2 == 0 ? 0.5 * F0 : F0;
255
256 for (int p = 0; p < popsize; p++) {
257 vec xp = popX.col(p);
258 vec xb = popX.col(bestI);
259 int r1, r2;
260 do {
261 r1 = rndInt(popsize);
262 } while (r1 == p || r1 == bestI);
263 do {
264 r2 = rndInt(popsize);
265 } while (r2 == p || r2 == bestI || r2 == r1);
266 vec x1 = popX.col(r1);
267 vec x2 = popX.col(r2);
268 int r = rndInt(dim);
269 vec x = vec(xp);
270 for (int j = 0; j < dim; j++) {
271 if (j == r || rnd01() < CR) {
272 x[j] = xb[j] + F * (x1[j] - x2[j]);
273 if (!fitfun->feasible(j, x[j]))
274 x[j] = sample_i(j);
275 }
276 }
277 modify(x);
278 double y = fitfun->eval(x)(0);
279 if (isfinite(y) && y < popY[p]) {
280 // temporal locality
281 vec x2 = next_improve(xb, x, xp);
282 double y2 = fitfun->eval(x2)(0);
283 if (isfinite(y2) && y2 < y) {
284 y = y2;
285 x = x2;
286 }
287 popX.col(p) = x;
288 popY(p) = y;
289 popIter[p] = iterations;
290 if (y < popY[bestI]) {
291 bestI = p;
292 if (y < bestY) {
293 bestY = y;
294 bestX = x;
295 if (isfinite(stopfitness) && bestY < stopfitness) {
296 stop = 1;
297 return;
298 }
299 }
300 }
301 } else {
302 // reinitialize individual
303 if (keep * rnd01() < iterations - popIter[p]) {
304 popX.col(p) = sample();
305 popY[p] = DBL_MAX;
306 }
307 }
308 }
309 }
310 }

312 void do_optimize_delayed_update(int workers) {
313 iterations = 0;
314 fitfun->resetEvaluations();
315 workers = std::min(workers, popsize); // workers <= popsize
316 evaluator eval(fitfun, 1, workers);
317 int evals_size = popsize*10;
318 vec evals_x[evals_size];
319 int evals_p[evals_size];
320 int cp = 0;
321
322 // fill eval queue with initial population
323 for (int i = 0; i < workers; i++) {
324 int p;
325 vec x = ask(p);
326 eval.evaluate(x, cp);
327 evals_x[cp] = x;
328 evals_p[cp] = p;
329 cp = (cp + 1) % evals_size;
330 }
331 while (fitfun->evaluations() < maxEvaluations && !fitfun->terminate()) {
332 vec_id* vid = eval.result();
333 vec y = vec(vid->_v);
334 int id = vid->_id;
335 delete vid;
336 vec x = evals_x[id];
337 int p = evals_p[id];
338 tell(y(0), x, p); // tell evaluated x
339 if (isfinite(stopfitness) && bestY < stopfitness) {
340 stop = 1;
341 break;
342 }
343 if (fitfun->evaluations() >= maxEvaluations)
344 break;
345 x = ask(p);
346 eval.evaluate(x, cp);
347 evals_x[cp] = x;
348 evals_p[cp] = p;
349 cp = (cp + 1) % evals_size;
350 }
351 }

353 void init() {
354 popX = mat(dim, popsize);
355 popX0 = mat(dim, popsize);
356 popY = vec(popsize);
357 meanHist = mean.replicate(1,10);
358 meanHistIndex = 0;
359 maxSigma = sigma / (.1 + minSigmaVal);
360 minSigma = minSigmaVal * sigma;
361 for (int p = 0; p < popsize; p++) {
362 popX0.col(p) = popX.col(p) = sample();
363 popY[p] = DBL_MAX;
364 }
365 bestI = 0;
366 bestX = popX.col(bestI);
367 popIter = zeros(popsize);
368 askedX = mat(dim, popsize);
369 askedP = ivec(popsize);
370 }

451void optimizeDE_C(long runid, callback_type func, int dim, int seed,
452 double *lower, double *upper,
453 double *init, double *sigma, double minSigma,
454 bool *ints,
455 int maxEvals, double keep,
456 double stopfitness, int popsize, double F, double CR,
457 double min_mutate, double max_mutate,
458 int workers, double* res) {
459
460 vec guess(dim), lower_limit(dim), upper_limit(dim), inputSigma(dim);
461 if (init != NULL and sigma != NULL) {
462 for (int i = 0; i < dim; i++) {
463 guess[i] = init[i];
464 inputSigma[i] = sigma[i];
465 }
466 } else {
467 guess.resize(0);
468 inputSigma.resize(0);
469 minSigma = 0;
470 }
471 if (lower != NULL && upper != NULL) {
472 for (int i = 0; i < dim; i++) {
473 lower_limit[i] = lower[i];
474 upper_limit[i] = upper[i];
475 }
476 } else {
477 lower_limit.resize(0);
478 upper_limit.resize(0);
479 }
480
481 Fitness fitfun(func, noop_callback_par, dim, 1, lower_limit, upper_limit);
482 DeOptimizer opt(runid, &fitfun, dim, seed, popsize, maxEvals, keep,
483 stopfitness, F, CR, min_mutate, max_mutate,
484 ints, guess, inputSigma, minSigma);
485 try {
486 if (workers <= 1)
487 opt.doOptimize();
488 else
489 opt.do_optimize_delayed_update(workers);
490 vec bestX = opt.getBestX();
491 double bestY = opt.getBestValue();
492 for (int i = 0; i < dim; i++)
493 res[i] = bestX[i];
494 res[dim] = bestY;
495 res[dim + 1] = fitfun.evaluations();
496 res[dim + 2] = opt.getIterations();
497 res[dim + 3] = opt.getStop();
498 } catch (std::exception &e) {
499 cout << e.what() << endl;
500 }
501}
