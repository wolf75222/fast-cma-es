119FirstCall = True
120# adapted copy from https://github.com/ppgaluzio/MOBOpt/blob/master/mobopt/_NSGA2.py
121def nsgaII(NObj, objective, pbounds, seed=None, NGEN=20000, MU=400, CXPB=0.9):
122 random.seed(seed)
123
124 global FirstCall
125 if FirstCall:
126 creator.create('FitnessMin', base.Fitness, weights=(-1.0,)*NObj)
127 creator.create('Individual', array.array, typecode='d',
128 fitness=creator.FitnessMin)
129 FirstCall = False
130 toolbox = base.Toolbox()
131
132 NDIM = len(pbounds)
133
134 toolbox.register('attr_float', uniform, pbounds)
135
136 toolbox.register('individual',
137 tools.initIterate,
138 creator.Individual,
139 toolbox.attr_float)
140
141 toolbox.register('population', tools.initRepeat, list, toolbox.individual)
142
143 toolbox.register('evaluate', objective)
144
145 toolbox.register('mate',
146 tools.cxSimulatedBinaryBounded,
147 low=pbounds[:, 0].tolist(),
148 up=pbounds[:, 1].tolist(),
149 eta=20.0)
150
151 toolbox.register('mutate',
152 tools.mutPolynomialBounded,
153 low=pbounds[:, 0].tolist(),
154 up=pbounds[:, 1].tolist(),
155 eta=20.0,
156 indpb=1.0/NDIM)
157
158 toolbox.register('select', tools.selNSGA2)
159
160 stats = tools.Statistics(lambda ind: ind.fitness.values)
161 stats.register('min', np.min, axis=0)
162 stats.register('max', np.max, axis=0)
163
164 logbook = tools.Logbook()
165 logbook.header = 'gen', 'evals', 'std', 'min', 'avg', 'max'
166
167 pop = toolbox.population(n=MU)
168
169 # Evaluate the individuals with an invalid fitness
170 invalid_ind = [ind for ind in pop if not ind.fitness.valid]
171 fitnesses = toolbox.map(toolbox.evaluate, invalid_ind)
172 for ind, fit in zip(invalid_ind, fitnesses):
173 ind.fitness.values = fit
174 pop = toolbox.select(pop, len(pop))
175
176 record = stats.compile(pop)
177 logbook.record(gen=0, evals=len(invalid_ind), **record)
178
179 for gen in range(1, NGEN):
180 # Vary the population
181 offspring = tools.selTournamentDCD(pop, len(pop))
182 offspring = [toolbox.clone(ind) for ind in offspring]
183
184 for ind1, ind2 in zip(offspring[::2], offspring[1::2]):
185 if random.random() <= CXPB:
186 toolbox.mate(ind1, ind2)
187
188 toolbox.mutate(ind1)
189 toolbox.mutate(ind2)
190 del ind1.fitness.values, ind2.fitness.values
191
192 # Evaluate the individuals with an invalid fitness
193 invalid_ind = [ind for ind in offspring if not ind.fitness.valid]
194 fitnesses = toolbox.map(toolbox.evaluate, invalid_ind)
195 for ind, fit in zip(invalid_ind, fitnesses):
196 ind.fitness.values = fit
197
198 # Select the next generation population
199 pop = toolbox.select(pop + offspring, MU)
200 record = stats.compile(pop)
201 logbook.record(gen=gen, evals=len(invalid_ind), **record)
202 # print(logbook.stream)
203
204 front = np.array([ind.fitness.values for ind in pop])
205 return pop, logbook, front

207def nsgaII_test(problem, fname, NGEN=2000, MU=100, value_limits = None):
208 time0 = time.perf_counter() # optimization start time
209 name = problem.name
210 logger.info('optimize ' + name + ' nsgaII')
211 pbounds = np.array(list(zip(problem.bounds.lb, problem.bounds.ub)))
212 pop, logbook, front = nsgaII(2, problem.fun, pbounds, NGEN=NGEN, MU=MU)
213 logger.info(name + ' nsgaII time ' + str(dtime(time0)))
214 name = 'nsgaII_' + str(NGEN) + '_' + str(MU) + name + '_' + fname
215 np.savez_compressed(name, xs=pop, ys=front)
216 if not value_limits is None:
217 front = np.array(
218 [y for y in front if all([y[i] < value_limits[i] for i in range(len(y))])])
219 retry.plot(front, name)

221def plot_all(folder, fname):

231 retry.plot(ys, fname + '_all.png', interp=False)
232 retry.plot(front, fname + '_front.png')
