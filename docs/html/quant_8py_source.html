36def get_var_form(params):
37 qr = QuantumRegister(1, name="q")
38 cr = ClassicalRegister(1, name='c')
39 qc = QuantumCircuit(qr, cr)
40 qc.u(params[0], params[1], params[2], qr[0])
41 qc.measure(qr, cr[0])
42 return qc

44def get_probability_distribution(counts):
45 output_distr = [v / NUM_SHOTS for v in counts.values()]
46 if len(output_distr) == 1:
47 output_distr.append(1 - output_distr[0])
48 return output_distr

50def random_target_distr():

56def objective_function(params, target_distr):
57 # Obtain a quantum circuit instance from the parameters
58 qc = get_var_form(params)
59 # Execute the quantum circuit to obtain the probability distribution associated with the current parameters
60 t_qc = transpile(qc, backend)
61 qobj = assemble(t_qc, shots=NUM_SHOTS)
62 result = backend.run(qobj).result()
63 # Obtain the counts for each measured state, and convert those counts into a probability vector
64 output_distr = get_probability_distribution(result.get_counts(qc))
65 # Calculate the cost as the distance between the output distribution and the target distribution
66 cost = sum([np.abs(output_distr[i] - target_distr[i]) for i in range(2)])
67 return cost
