50def objectives_re(name, x, numVars, numObjs):
51 array_type = ct.c_double * numVars
52 try:
53 res = objectives_re_C(ct.create_string_buffer(name.encode('utf-8')), array_type(*x))
54 y = np.array(np.fromiter(res, dtype=np.float64, count=numObjs))
55 freemem(res)
56 return y
57 except Exception as ex:
58 return None

60def bounds_re(name):
61 try:
62 res = bounds_re_C(ct.create_string_buffer(name.encode('utf-8')))
63 numVars = int(res[0])
64 numObjs = int(res[1])
65 numConstr = int(res[2])
66 lower = np.array(res[3:3+numVars])
67 upper = np.array(res[3+numVars:3+2*numVars])
68 freemem(res)
69 return numVars, numObjs, numConstr, lower, upper
70 except Exception as ex:
71 return None
