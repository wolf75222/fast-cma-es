26def shift_indices(shifts):
27 shift_to_index = {}
28 days = []
29 locations = []
30 skills = []
31 sec_start = []
32 sec_end = []
33 for i in range(len(shifts)):
34 shift_to_index[str(shifts[i])] = i
35 day = shifts[i]['start'].split('T')[0]
36 dt_start = parse(shifts[i]['start'])
37 dt_end = parse(shifts[i]['end'])
38 sec_start.append(dt_start.timestamp())
39 sec_end.append(dt_end.timestamp())
40 location = shifts[i]['location']
41 required_skill = shifts[i]['required_skill']
42 days.append(day)
43 locations.append(location)
44 skills.append(required_skill)
45 return shift_to_index, days, locations, skills, \
46 np.array(sec_start), np.array(sec_end)

75def index_map(strs):
76 to_index = {}
77 ids = []
78 j = 0
79 for s in strs:
80 if not s in to_index:
81 to_index[s] = j
82 j += 1
83 ids.append(to_index[s])
84 return to_index, np.array(ids)

86def index_multi_map(lists):
87 to_index = {}
88 max_len = max([len(l) for l in lists])
89 ids = np.full((len(lists), max_len), -1, dtype=int)
90 k = 0
91 for i in range(len(lists)):
92 l = lists[i]
93 for j in range(len(l)):
94 s = l[j]
95 if not s in to_index:
96 to_index[s] = k
97 k += 1
98 ids[i,j] = to_index[s]
99 return to_index, np.array(ids)

106@njit(fastmath=True)
107def fitness_(employees_at_shift, day_ids, required_skill_ids, skill_set_ids,
108 avail_names_ids, avail_days_ids, avail_type_ids, sec_start, sec_end):
109 score = 0
110 num_employees = len(skill_set_ids)
111 employee_last_day = np.full(num_employees, -1, dtype=numba.int32)
112 employee_last_end = np.full(num_employees, -1, dtype=numba.int32)
113 employee_num_shifts = np.zeros(num_employees, dtype=numba.int32)
114 for shift in range(len(employees_at_shift)):
115 day = day_ids[shift]
116 employee = employees_at_shift[shift]
117 employee_num_shifts[employee] += 1
118 if employee_last_day[employee] == day:
119 score += 1000 # employee should only work once a day
120 employee_last_day[employee] = day
121 if sec_start[shift] - employee_last_end[employee] < 10*3600:
122 score += 1000 # employee should pause for 10 hours (and shifts should not overlap)
123 employee_last_end[employee] = sec_end[shift]
124 required_skill = required_skill_ids[shift]
125 skill_set = skill_set_ids[employee]
126 if not required_skill in skill_set:
127 score += 1000 # employee has wrong skill set
128 avail_ids = np.where(avail_names_ids == employee)
129 for avail_id in avail_ids[0]:
130 avail_day = avail_days_ids[avail_id]
131 if day == avail_day:
132 type = avail_type_ids[avail_id]
133 if type == UNDESIRED:
134 score += 100 # employee does not want to work this day
135 elif type == UNAVAILABLE:
136 score += 1000 # employee is unavailable
137 elif type == DESIRED:
138 score -= 100 # employee works at desired day
139 return score, employee_num_shifts
