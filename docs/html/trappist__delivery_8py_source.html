25"""
26 Extends ESAs verification code for the SpOC: Delivery Scheduling competition held in 2022.
27 Original code
28 https://api.optimize.esa.int/media/problems/spoc-delivery-scheduling-delivery-scheduling-1649698262955.py
29 is unmodified.
30 A second class fitness is added, which removes redundancies in the decision variable vector reducing
31 dimension to 352. Instead of only considering the station with minimal resources it uses a weighted
32 sum as fitness value. Performance is greatly improved by using https://numba.pydata.org/ enabling about
33 500000 fitness evaluations per second on a 16 core CPU.
34 Applies the https://github.com/avaneev/biteopt BiteOpt algorithm iteratively
35 using parallel restarts.

358 def _fitness_impl(
359 self,
360 x,
361 logging=False,
362 plotting=False,
363 ax=None,
364 path=None,
365 ):
366 """
367 Computes the constraints and the fitness of the provided chromosome.
368
369 1. Equality constraints:
370
371 1.1. Asteroid IDs: all asteroids in the database must be present in the chromosome.
372 1.2. Opportunity IDs: all opportunity IDs in the chromosome must correspond to opportunities in the database.
373
374 2. Inequality constraints:
375
376 2.1. Station gaps: all station gaps must be greater than a minimal time period (self.station_gap)
377 2.2. Arrival times: all asteroid arrival times must be between the start and end times of the corresponding station
378
379 3. Fitness:
380
381 3.1 Iterate over the chromosome and add the masses of the materials for all assigned asteroids with valid transfers
382 3.2 Find the minimum mass of each material per station. This is the final fitness.
383
384 Args:
385 x: A list of integers and floats in the following format:
386 - Station start and end times (2 x self.n_stations)
387 - The following items for all selected asteroids (3 x self.n_asteroids integers in total):
388 - Asteroid ID
389 - Station ID
390 - Opportunity ID
391
392 NOTE: The triplets do not have to be ordered by asteroid ID.
393
394 logging: Logging switch. Defaults to False.
395 plotting: Plotting switch. Defaults to False.
396 ax: Plot axes. Defaults to None.
397 path: File path for saving the plots. Defaults to None.
398
399 Returns:
400 A tuple containing:
401 - The fitness
402 - A list of equality constraints
403 - A list of inequality constraints
404 - Plot axes
405
406 """
407
408 eq_constraints = []
409 ineq_constraints = []
410
411 # Offset representing the time windows for stations (for convenience only)
412 station_times_offset = 2 * self.n_stations
413
414 # Extract a set of tuples of all the selections as tuples
415 asteroid_ids = [int(a_id) for a_id in x[station_times_offset::3]]
416 station_ids = [int(s_id) for s_id in x[(station_times_offset + 1) :: 3]]
417 assignments = [
418 int(assignment - 1) for assignment in x[(station_times_offset + 2) :: 3]
419 ]
420
421 triplets = tuple(zip(asteroid_ids, station_ids, assignments))
422
423 # Extract the start and end times for all stations from the chromosome
424 station_times = x[:station_times_offset]
425 station_start_times = station_times[0 : len(station_times) : 2]
426 station_end_times = station_times[1 : len(station_times) : 2]
427
428 """
429 1. Equality constraints.
430 """
431
432 # ==[ 1.1 Check asteroid IDs ]==
433
434 asteroid_id_violations = set(asteroid_ids).symmetric_difference(
435 set(self.db.keys())
436 )
437
438 eq_constraints.append(len(asteroid_id_violations))
439
440 # ==[ 1.2. Check opportunities ]==
441
442 # Check the opportunity IDs for asteroid / station assignments
443 # where the station ID is *not* 0 (i.e., the asteroid has
444 # been activated and selected for transfer to a valid station).
445 opportunity_id_violations = [
446 triplet
447 for triplet in triplets
448 if triplet not in self.flat_db and triplet[1] > 0
449 ]
450
451 eq_constraints.append(len(opportunity_id_violations))
452
453 """
454 2. Inequality constraints.
455 """
456
457 # First we sort and index station activation windows.
458 # We produce a list where each entry has the following format:
459 #
460 # [idx, [start, end]]
461 #
462 # where
463 # - idx: index in the *original* chromosome (= station ID)
464 # - (start, end): start and end times for visiting the corresponding station
465
466 station_windows = zip(station_start_times, station_end_times)
467 sorted_indexed_windows = sorted(
468 enumerate(station_windows, 1), key=lambda i: i[1]
469 )
470
471 # ==[ 2.1. Check station gaps ]==
472
473 # Compute the gaps between stations from the sorted array of station windows
474 gaps = [
475 s2[1][0] - s1[1][1]
476 for s1, s2 in zip(sorted_indexed_windows[:-1], sorted_indexed_windows[1:])
477 ]
478
479 gap_violations = self.station_gap - np.array(gaps, dtype=np.float32)
480
481 ineq_constraints.append(max(gap_violations))
482
483 # ==[ 2.2. Arrival times ]==
484
485 # Collect all asteroid times for each station
486 station_window_start_times = []
487 station_window_end_times = []
488
489 for s_id in station_ids:
490 if s_id == 0:
491 # Trick to make sure that unassigned asteroids don't trigger
492 # and inequality constraint violation
493 station_window_start_times.append(-1.0)
494 station_window_end_times.append(self.end_time + 1.0)
495 else:
496 station_window_start_times.append(station_start_times[s_id - 1])
497 station_window_end_times.append(station_end_times[s_id - 1])
498
499 # Find any instances where the arrival time is before the start time or
500 # after the end time of the corresponding station's window.
501 arrival_times = np.array(
502 [
503 self.flat_db[triplet][0]
504 if triplet in self.flat_db
505 else (self.end_time - self.start_time) / 2
506 for triplet in triplets
507 ],
508 dtype=np.float32,
509 )
510 station_window_start_times = np.array(
511 station_window_start_times, dtype=np.float32
512 )
513 station_window_end_times = np.array(station_window_end_times, dtype=np.float32)
514
515 # Find instances where the arrival times are earier than the
516 # start time for the corresponding station
517 arrival_time_violations = np.where(
518 arrival_times - station_window_start_times < 0.0, 1, 0
519 )
520
521 # Add to those any instances where the arrival times are later than the
522 # end time for the corresponding station
523 arrival_time_violations += np.where(
524 station_window_end_times - arrival_times < 0.0, 1, 0
525 )
526
527 # Find any violations (1s indicate a violation in either start time or end time)
528 arrival_time_violations = np.where(arrival_time_violations > 0, 1, 0)
529
530 ineq_constraints.append(arrival_time_violations.sum())
531
532 """
533 3. Fitness
534 """
535
536 # Compute the masses of all materials accumulated at each station
537 masses_per_station = {
538 s_id: np.array([0.0, 0.0, 0.0], dtype=np.float32)
539 for s_id in range(1, self.n_stations + 1)
540 }
541
542 for (triplet, violation) in zip(
543 triplets, arrival_time_violations.astype(bool).tolist()
544 ):
545 # If the asteroid is assigned to a valid station and
546 # its arrival time is within bounds...
547 if triplet[1] > 0 and not violation:
548 # Add the material masses to the stations
549 masses_per_station[triplet[1]] += np.array(
550 self.flat_db[triplet][1:],
551 dtype=np.float32,
552 )
553
554 # Collect all the masses per station into a single 2D array that is easy to manipulate
555 masses = np.array(
556 [masses_per_station[s] for s in range(1, self.n_stations + 1)]
557 )
558
559 # Final fitness computation.
560 # The objective is to maximise the minimum mass
561 # of any material across all stations.
562 fitness = -masses.min()
563
564 if logging:
565
566 print(
567 f"==[ Invalid asteroid IDs: {eq_constraints[0]} out of {len(self.db)}"
568 )
569 print(f"==[ Invalid arrival times: {eq_constraints[1]}")
570 print(f"==[ Minimal inter-station gap: {min(gaps):<2.4}")
571 print(f"==[ Invalid assignments: {ineq_constraints[1]}")
572 print(f"==[ Masses per station:")
573 print(
574 f"{'Station ID':>12} {self.materials[0]:>12} {self.materials[1]:>12} {self.materials[2]:>12}"
575 )
576 for stat_id, mass_dist in masses_per_station.items():
577 print(
578 f"{stat_id:>12} {mass_dist[0]:>12.6f} {mass_dist[1]:>12.6f} {mass_dist[2]:>12.6f}"
579 )
580
581 for idx, item in enumerate(gap_violations):
582 if item > 0:
583 # Get the station IDs in the original chromosome
584 station_1_id = sorted_indexed_windows[idx][0]
585 station_2_id = sorted_indexed_windows[idx + 1][0]
586 gap = gaps[idx]
587 if gap <= self.station_gap:
588 print(
589 f"==[\tThe gap between stations {station_1_id} and {station_2_id} is {gap:3.3f} (should be >= {self.station_gap:3.3f})."
590 )
591 else:
592 print(
593 f"==[\tThe windows for stations {station_1_id} and {station_2_id} overlap by {-gap:3.3f} days."
594 )
595
596 print(f"==[ Total fitness: {fitness}")
597
598 # Plotting
599 if plotting:
600 schedule = {
601 s_id: {
602 "opportunities": [],
603 "assignments": [],
604 "window": [],
605 }
606 for s_id in range(1, self.n_stations + 1)
607 }
608
609 for (_, s_id, _), val in self.flat_db.items():
610 schedule[s_id]["opportunities"].append(val[0])
611
612 for triplet in triplets:
613 if triplet[1] > 0:
614 schedule[triplet[1]]["assignments"].append(self.flat_db[triplet][0])
615
616 for s_id, window in sorted_indexed_windows:
617 if s_id > 0:
618 schedule[s_id]["window"] = window
619
620 ax = self._plot(masses, schedule, ax=ax, path=path)
621
622 return (fitness, eq_constraints, ineq_constraints, ax)

624 def fitness(
625 self,
626 x,
627 ):
628 """
629 A wrapper for the fitness function (called for evaluation only).
630
631 #################### IMPORTANT ######################
632 - The chromosome has the following format:
633
634 - Start and end times for each station, *in order of Station ID*
635 - Asteroid / station assignments with the corresponding arrival times.
636
637 Format:
638 [
639 Station 1 start time, Station 1 end time, |
640 Station 2 start time, Station 2 end time, |
641 ... | 2 x number of stations
642 Station 11 start time, Station 11 end time, |
643 Station 12 start time, Station 12 end time, |
644 Asteroid ID, Station ID, Opportunity ID, |
645 Asteroid ID, Station ID, Opportunity ID, |
646 ... | number of asteroids
647 Asteroid ID, Station ID, Opportunity ID, |
648 Asteroid ID, Station ID, Opportunity ID |
649 ]
650
651 - All IDs (for asteroids, stations and opportunities) are 1-based.
652 - This is particularly relevant for the opportunity IDs
653 since they are converted to 0-based indices in the fitness
654 evaluation function by subtracting 1.
655
656 - Stations must be activated *sequentially* (*not* in parallel) but not necessarily in order of their ID.
657
658 - There must be a minimal gap (called 'station gap') between the end time of one station
659 and the start time of the next.
660
661 - Every asteroid must be either asigned to a station or unassigned.
662
663 - The asteroid / station assignments do not have to be in any particular order,
664 but all asteroid IDs must appear in the chromosome, even if some asteroids are unassigned.
665
666 - Assigned asteroids must have corresponding Station IDs between 1 and the number of stations.
667 - Unassigned asteroids must have a Station ID 0.
668
669 ######################################################
670
671 Args:
672 x: A chromosome in the format specified above.
673
674 Returns:
675 A tuple containing the fitness followed by the equality and inequality constraints.
676
677 """
678
679 (fitness, eq_constraints, ineq_constraints, ax) = self._fitness_impl(x)
680
681 return (fitness, *eq_constraints, *ineq_constraints)

724 def example(self):
725 """
726 An example method returning a minimal chromosome that assigns
727 a single asteroid to each station.
728
729 Returns:
730 A valid chromosome.
731 """
732
733 assignments = {}
734
735 # Iterate over a random permutation of asteroid IDs
736 for ast_id in list(self.db.keys()):
737
738 if len(assignments) == self.n_stations:
739 break
740
741 # Create a set of station IDs.
742 stations = set(self.db[ast_id].keys()).difference(set(assignments.keys()))
743 while len(stations) > 0:
744 stat_id = stations.pop()
745
746 # Check if the asteroid / station pair
747 # is already assigned.
748 if stat_id in assignments:
749 break
750
751 opps = self.db[ast_id][stat_id]
752 # Check if there are any opportunities
753 for opp_idx, opp in enumerate(opps, 1):
754 # Check if the arrival time of the opportunity conflicts with
755 # the assigned asteroid / station build times
756 conflict = False
757 for _, vals in assignments.items():
758 if ast_id == vals[0] or (vals[2] - 2 * self.station_gap) <= opp[
759 0
760 ] <= (vals[3] + 2 * self.station_gap):
761 conflict = True
762 break
763 if not conflict:
764 assignments[stat_id] = [
765 ast_id,
766 opp_idx,
767 opp[0] - self.station_gap,
768 opp[0] + self.station_gap,
769 ]
770 stations = set()
771 break
772
773 # Build the chromosome
774 windows = []
775 schedule = []
776 for stat_id in range(1, self.n_stations + 1):
777 # Station start time
778 windows.append(assignments[stat_id][2])
779 # Station end time
780 windows.append(assignments[stat_id][3])
781 # Asteroid / station assignment
782 schedule.extend([assignments[stat_id][0], stat_id, assignments[stat_id][1]])
783
784 # Stitch the windows and schedule together and create
785 # a complete chromosome from this partial solution.
