72class Fitness {
73
74public:
75
76 vec lower;
77 vec upper;
78
79 Fitness(callback_type pfunc, vec *lower_limit, vec *upper_limit,
80 long maxEvals_) {
81 func = pfunc;
82 lower = *lower_limit;
83 upper = *upper_limit;
84 if (lower.size() > 0) // bounds defined
85 scale = (upper - lower);
86 maxEvals = maxEvals_;
87 }
88
89 vec getClosestFeasible(const vec &X) const {
90 if (lower.size() > 0) {
91 return X.cwiseMin(1.0).cwiseMax(-1.0);
92 }
93 return X;
94 }
95
96 double eval(const vec &X) {
97 int n = X.size();
98 double res = func(n, X.data());
99 evaluationCounter++;
100 return res;
101 }
102
103 double value(const vec &X) {
104 double res = DBL_MAX;
105 if (lower.size() > 0)
106 res = eval(decode(getClosestFeasible(X)));
107 else
108 res = eval(X);
109 if (res < bestY) {
110 bestY = res;
111 bestX = vec(X);
112 }
113 return res;
114 }
115
116 const double LS_MAXITER_RATIO = 6;
117 const double LS_MAXITER_MIN = 100;
118 const double LS_MAXITER_MAX = 1000;
119
120 double local_search(const vec &x0, double currval, vec &res) {
121 vec init = getClosestFeasible(x0);
122 bestY = DBL_MAX;
123 int maxIter = LS_MAXITER_RATIO * x0.size();
124 if (maxIter > LS_MAXITER_MAX)
125 maxIter = LS_MAXITER_MAX;
126 if (maxIter < LS_MAXITER_MIN)
127 maxIter = LS_MAXITER_MIN;
128 minLBFGS(this, init, maxIter);
129 if (bestY < DBL_MAX) {
130 for (int i = 0; i < res.size(); i++)
131 res[i] = bestX(i);
132 }
133 return bestY;
134 }
135
136 vec encode(const vec &X) const {
137 if (lower.size() > 0)
138 return (X - lower).array() / scale.array();
139 else
140 return X;
141 }
142
143 vec decode(const vec &X) const {
144 if (lower.size() > 0)
145 return X.cwiseProduct(scale) + lower;
146 else
147 return X;
148 }
149
150 int getEvaluations() {
151 return evaluationCounter;
152 }
153
154 bool maxEvalReached() {
155 return evaluationCounter >= maxEvals;
156 }
157
158private:
159 callback_type func;
160 long evaluationCounter = 0;
161 long maxEvals;
162 vec scale;
163 double bestY = DBL_MAX;
164 vec bestX;
165};

167class LBFGSFunc {
168private:
169 Fitness *func;
170 int dim;
171
172public:
173
174 LBFGSFunc(Fitness *Fitness_, int dim_) {
175 func = Fitness_;
176 dim = dim_;
177 }
178
179 double operator()(const vec &x, vec &grad) {
180 if (!x.allFinite())
181 return DBL_MAX;
182 double eps = 1E-6;
183 vec arg = vec(dim);
184 for (int i = 0; i < dim; i++)
185 arg[i] = x(i);
186 for (int i = 0; i < dim; i++) {
187 vec x1 = vec(arg);
188 vec x2 = vec(arg);
189 double e1 = eps;
190 double e2 = eps;
191 x1[i] += eps;
192 if (x1[i] > 1) {
193 x1[i] = 1;
194 e1 = 1 - arg[i];
195 }
196 x2[i] -= eps;
197 if (x2[i] < 0) {
198 x2[i] = 0;
199 e2 = arg[i];
200 }
201 double f1 = func->value(x1);
202 double f2 = func->value(x2);
203 grad[i] = (f1 - f2) / (e1 + e2);
204 }
205 double f = func->value(arg);
206 return f;
207 }
208};
209
210double minLBFGS(Fitness *fitfun, vec &X0, int maxIterations) {
211 int dim = X0.size();
212 LBFGSFunc fun = LBFGSFunc(fitfun, dim);
213
214 LBFGSBParam<double> param;
215 param.max_iterations = maxIterations;
216 LBFGSBSolver<double> solver(param);
217 vec lb = vec::Constant(dim, 0.0);
218 vec ub = vec::Constant(dim, 1.0);
219 // Initial values
220 vec x = vec::Constant(dim, 0);
221 for (int i = 0; i < dim; i++)
222 x[i] = X0[i];
223 double fx;
224 int niter;
225 try {
226 niter = solver.minimize(fun, x, fx, lb, ub);
227 } catch (std::exception &e) {
228 //cout << e.what() << endl;
229 return DBL_MAX;
230 }
231 return fx;
232}

234class VisitingDistribution {
235
236 //Class used to generate new coordinates based on the distorted
237 //Cauchy-Lorentz distribution. Depending on the steps within the Markov
238 //chain, the class implements the strategy for generating new location
239 //changes.
240
241public:
242
243 VisitingDistribution(int dim, double visiting_param_, pcg64 *rs_) {
244 _visiting_param = visiting_param_;
245 rs = rs_;
246
247 // these are invariant numbers unless visiting_param changes
248 double factor2 = exp(
249 (4.0 - _visiting_param) * log(_visiting_param - 1.0));
250 double factor3 = exp(
251 (2.0 - _visiting_param) * log(2.0) / (_visiting_param - 1.0));
252 _factor4_p = sqrt(M_PI) * factor2 / (factor3 * (3.0 - _visiting_param));
253
254 double factor5 = 1.0 / (_visiting_param - 1.0) - 0.5;
255 double d1 = 2.0 - factor5;
256 _factor6 = M_PI * (1.0 - factor5) / sin(M_PI * (1.0 - factor5))
257 / exp(lgamma(d1));
258 }
259
260 vec visiting(const vec &x, int step, double temperature) {
261 //Based on the step in the strategy chain, new coordinated are
262 //generated by changing all components is the same time or only
263 //one of them, the new values are computed with visit_fn method
264
265 int dim = x.size();
266 if (step < dim) {
267 // Changing all coordinates with a new visiting value
268 double upper_sample = distr_01(*rs);
269 double lower_sample = distr_01(*rs);
270 vec visits = visit_fn(temperature, dim);
271 for (int i = 0; i < dim; i++) {
272 if (visits[i] > TAIL_LIMIT)
273 visits[i] = TAIL_LIMIT * upper_sample;
274 else if (visits[i] < -TAIL_LIMIT)
275 visits[i] = -TAIL_LIMIT * lower_sample;
276 }
277 vec x_visit = visits + x;
278 vec a = x_visit;
279 vec b = vec(dim);
280 for (int i = 0; i < dim; i++) {
281 b[i] = fmod(a[i], 1) + 1;
282 x_visit[i] = fmod(b[i], 1);
283 if (abs(x_visit[i]) < MIN_VISIT_BOUND)
284 x_visit[i] += 1.e-10;
285 }
286 //cerr << step << " " << temperature << endl;// << x_visit << endl;
287 return x_visit;
288 } else {
289 // Changing only one coordinate at a time based on strategy
290 // chain step
291 vec x_visit = vec(x);
292 double visit = visit_fn(temperature, 1)[0];
293 if (visit > TAIL_LIMIT)
294 visit = TAIL_LIMIT * distr_01(*rs);
295 else if (visit < -TAIL_LIMIT)
296 visit = -TAIL_LIMIT * distr_01(*rs);
297 int index = step - dim;
298 x_visit[index] = visit + x[index];
299 double a = x_visit[index];
300 double b = fmod(a, 1) + 1;
301 x_visit[index] = fmod(b, 1);
302 if (abs(x_visit[index]) < MIN_VISIT_BOUND)
303 x_visit[index] += MIN_VISIT_BOUND;
304 //cerr << step << " " << temperature << endl;// << x_visit << endl;
305 return x_visit;
306 }
307 }
308
309 vec visit_fn(double temperature, int dim) {
310
311 //Formula Visita from p. 405 of reference [2]
312 vec x = normalVec(dim, *rs);
313 vec y = normalVec(dim, *rs);
314 ;
315
316 double factor1 = exp(log(temperature) / (_visiting_param - 1.0));
317 double factor4 = _factor4_p * factor1;
318
319 // sigmax
320 x = x
321 * exp(
322 -(_visiting_param - 1.0) * log(_factor6 / factor4)
323 / (3.0 - _visiting_param));
324
325 vec den = expv(
326 logv(y.cwiseAbs() * (_visiting_param - 1.0))
327 / (3.0 - _visiting_param));
328 return x.cwiseQuotient(den);
329 }
330
331private:
332
333 pcg64 *rs;
334 double _visiting_param;
335 double _factor4_p;
336 double _factor6;
337
338 const double TAIL_LIMIT = 1.e8;
339 const double MIN_VISIT_BOUND = 1.e-10;
340};

350class EnergyState {
351
352 //Class used to record the energy state-> At any time, it knows what is the
353 //currently used coordinates and the most recent best location
354public:
355
356 double ebest;
357 vec xbest;
358 double current_energy;
359 vec current_location;
360
361 EnergyState(int dim_) {
362 dim = dim_;
363 ebest = DBL_MAX;
364 xbest = { };
365 current_energy = DBL_MAX;
366 current_location = { };
367 }
368
369 void reset(Fitness *owf, pcg64 *rs, const vec &x0) {
370 if (x0.size() == 0)
371 current_location = normalVec(dim, *rs);
372 else
373 current_location = vec(x0);
374 bool init_error = true;
375 int reinit_counter = 0;
376 while (init_error) {
377 current_energy = owf->value(current_location);
378 if (current_energy >= BIG_VALUE || isnan(current_energy)) {
379 if (reinit_counter >= MAX_REINIT_COUNT) {
380 init_error = false;
381 throw naneexc;
382 }
383 current_location = uniformVec(dim, *rs);
384 reinit_counter++;
385 } else
386 init_error = false;
387 // If first time reset, initialize ebest and xbest
388 if (ebest == DBL_MAX && xbest.size() == 0) {
389 ebest = current_energy;
390 xbest = vec(current_location);
391 }
392 // Otherwise, keep them in case of reannealing reset
393 }
394 }
395
396 void update_best(double e, const vec &x) {
397 ebest = e;
398 xbest = vec(x);
399 }
400
401 void update_current(double e, const vec &x) {
402 current_energy = e;
403 current_location = vec(x);
404 }
405
406private:
407 // Maximimum number of trials for generating a valid starting point
408 int MAX_REINIT_COUNT = 1000;
409 int dim;
410};

412class StrategyChain {
413 // Class used for the Markov chain and related strategy for local search
414 // decision
415public:
416
417 StrategyChain(double acceptance_param_, VisitingDistribution *vd_,
418 Fitness *ofw_, pcg64 *rs_, EnergyState *state_) {
419 // Global optimizer state
420 state = state_;
421 // Local markov chain minimum energy and location
422 emin = state->current_energy;
423 xmin = vec(state->current_location);
424 // Acceptance parameter
425 acceptance_param = acceptance_param_;
426 // Visiting distribution instance
427 vd = vd_;
428 // Wrapper to objective function and related local minimizer
429 ofw = ofw_;
430 not_improved_idx = 0;
431 not_improved_max_idx = 1000;
432 rs = rs_;
433 temperature_step = 0;
434 K = 100 * (state->current_location).size();
435 }
436
437 void accept_reject(int j, double e, const vec &x_visit) {
438 double r = distr_01(*rs);
439 double pqv_temp = (acceptance_param - 1.0) * (e - state->current_energy)
440 / (temperature_step + 1.);
441 double pqv = 0;
442 if (pqv_temp < 0.)
443 pqv = 0.;
444 else
445 pqv = exp(log(pqv_temp) / (1. - acceptance_param));
446 if (r <= pqv) {
447 // We accept the new location and update state
448 state->update_current(e, x_visit);
449 xmin = vec(state->current_location);
450 }
451 // No improvement since long time
452 if (not_improved_idx >= not_improved_max_idx) {
453 if (j == 0 || state->current_energy < emin) {
454 emin = state->current_energy;
455 xmin = vec(state->current_location);
456 }
457 }
458 }
459
460 void run(int step, double temperature) {
461 temperature_step = temperature / (double) (step + 1);
462 not_improved_idx += 1;
463 for (unsigned int j = 0; j < (state->current_location).size() * 2;
464 j++) {
465 if (j == 0)
466 state_improved = false;
467 if (step == 0 && j == 0)
468 state_improved = true;
469 vec x_visit = vd->visiting(state->current_location, j, temperature);
470 // Calling the objective function
471 double e = ofw->value(x_visit);
472 if (e < state->current_energy) {
473 // We have got a better energy value
474 state->update_current(e, x_visit);
475 if (e < state->ebest) {
476 state->update_best(e, x_visit);
477 state_improved = true;
478 not_improved_idx = 0;
479 }
480 } else {
481 // We have not improved but do we accept the new location?
482 accept_reject(j, e, x_visit);
483 }
484 if (ofw->maxEvalReached())
485 return;
486 } // End of StrategyChain loop
487 }
488
489 void local_search() {
490 // Decision making for performing a local search
491 // based on Markov chain results
492 // If energy has been improved or no improvement since too long,
493 // performing a local search with the best Markov chain location
494 int dim = state->xbest.size();
495 if (state_improved) {
496 // Global energy has improved, let's see if LS improved further
497 vec x = vec(dim);
498 double e = ofw->local_search(state->xbest, state->ebest, x);
499 if (e < state->ebest) {
500 not_improved_idx = 0;
501 state->update_best(e, x);
502 state->update_current(e, x);
503 if (ofw->maxEvalReached())
504 return;
505 }
506 }
507 // Check probability of a need to perform a LS even if no improvment
508 bool do_ls = false;
509 if (K < 90 * state->current_location.size()) {
510 double pls = exp(
511 K * (state->ebest - state->current_energy)
512 / temperature_step);
513 if (pls >= distr_01(*rs))
514 do_ls = true;
515 }
516 // Global energy not improved, let's see what LS gives
517 // on the best strategy chain location
518 if (not_improved_idx >= not_improved_max_idx)
519 do_ls = true;
520 if (do_ls) {
521 vec x = vec(dim);
522 double e = ofw->local_search(xmin, state->ebest, x);
523 xmin = vec(x);
524 emin = e;
525 not_improved_idx = 0;
526 not_improved_max_idx = state->current_location.size();
527 if (e < state->ebest) {
528 state->update_best(emin, xmin);
529 state->update_current(e, x);
530 }
531 }
532 }
533
534private:
535
536 double emin;
537 vec xmin;
538 EnergyState *state;
539 double acceptance_param;
540 VisitingDistribution *vd;
541 int not_improved_idx;
542 int not_improved_max_idx;
543 pcg64 *rs;
544 Fitness *ofw;
545 double temperature_step;
546 double K;
547 bool state_improved = false;
548};

556class DARunner {
557
558public:
559
560 DARunner(Fitness *fun_, vec &x0_, long seed_, bool use_local_search_) {
561 owf = fun_;
562 if (x0_.size() > 0 && x0_.size() != owf->lower.size())
563 throw sizeeexc;
564 //Initialization of RandomState for reproducible runs if seed provided
565 rs = new pcg64(seed_);
566 use_local_search = use_local_search_;
567 // Initialization of the energy state
568 es = new EnergyState(owf->lower.size());
569 es->reset(owf, rs, x0_);
570 // VisitingDistribution instance
571 vd = new VisitingDistribution(owf->lower.size(), qv, rs);
572 // Markov chain instance
573 sc = new StrategyChain(qa, vd, owf, rs, es);
574 }
575
576 ~DARunner() {
577 delete rs;
578 delete vd;
579 delete sc;
580 delete es;
581 }
582
583 void search() {
584 iter = 0;
585 double t1 = exp((qv - 1) * log(2.0)) - 1.0;
586 for (;;) {
587 for (int i = 0; i < maxsteps; i++) {
588 // Compute temperature for this step
589 double s = i + 2.0;
590 double t2 = exp((qv - 1) * log(s)) - 1.0;
591 double temperature = temperature_start * t1 / t2;
592 if (iter++ >= maxsteps)
593 return;
594 // Need a re-annealing process?
595 if (temperature < temperature_restart) {
596 es->reset(owf, rs, emptyVec);
597 break;
598 }
599 // starting strategy chain
600 sc->run(i, temperature);
601 if (owf->maxEvalReached())
602 return;
603 if (use_local_search) {
604 sc->local_search();
605 if (owf->maxEvalReached())
606 return;
607 }
608 }
609 }
610 }
611
612 vec bestX() {
613 return es->xbest;
614 }
615
616 double bestY() {
617 return es->ebest;
618 }
619
620private:
621
622 int MAX_REINIT_COUNT = 1000;
623 double temperature_start = 5230;
624 double qv = 2.62;
625 double qa = -5.0;
626 bool use_local_search;
627 // maximum number of step (main iteration)
628 double maxsteps = 1000;
629 // minimum value of annealing temperature reached to perform
630 // re-annealing temperature_start
631 double temperature_restart = 0.1;
632 Fitness *owf;
633 pcg64 *rs;
634 EnergyState *es;
635 StrategyChain *sc;
636 VisitingDistribution *vd;
637 int iter = 0;
638};

640double minimize(Fitness *fun, vec &x0, long seed, bool use_local_search,
641 vec &X) {
642 DARunner gr = DARunner(fun, x0, seed, use_local_search);
643 gr.search();
644 int dim = x0.size();
645 vec bx = gr.bestX();
646 for (int i = 0; i < dim; i++)
647 X[i] = bx[i];
648 return gr.bestY();
649}

655void optimizeDA_C(long runid, callback_type func, int dim, int seed,
656 double *init, double *lower, double *upper, int maxEvals,
657 bool use_local_search, double* res) {
658 int n = dim;
659 vec guess(n), lower_limit(n), upper_limit(n);
660 for (int i = 0; i < n; i++)
661 guess[i] = init[i];
662 if (lower != NULL && upper != NULL) {
663 for (int i = 0; i < n; i++) {
664 lower_limit[i] = lower[i];
665 upper_limit[i] = upper[i];
666 }
667 } else {
668 lower_limit.resize(0);
669 upper_limit.resize(0);
670 }
671 if (maxEvals <= 0)
672 maxEvals = 1E7;
673 Fitness fitfun(func, &lower_limit, &upper_limit, maxEvals);
674
675 try {
676 vec X = zeros(dim);
677 vec enc = fitfun.encode(guess);
678 double bestY = minimize(&fitfun, enc, seed, use_local_search, X);
679 vec bestX = fitfun.decode(X);
680 for (int i = 0; i < n; i++)
681 res[i] = bestX[i];
682 res[n] = bestY;
683 res[n + 1] = fitfun.getEvaluations();
684 res[n + 2] = 0;
685 res[n + 3] = 0;
686 } catch (std::exception &e) {
687 cerr << e.what() << endl;
688 }
689}
