6# A cylindrical vessel is capped at both ends by hemispherical heads. The objective is to
7# minimize the total weight, including the weight of material, forming and welding. There are four design variables:
8
9# - x1: thickness of the shell
10# - x2: thickness of the head
11# - x3: inner radius
12# - x4 length of cylindrical section of the vessel, not including the head
13
14# x1 and x2 are integer multiples of 0.0625 inch, witch are
15# the available thickness of rolled steel plates,
16# and x3 and x4 are continuous.
17# x1 and x2 are made discrete by calling '_feasible(x)' in the objective function 'feasable_cost_penalty()
18# This works well with derivative free optimizers, but never do this with scipy.minimize.
19# Try 'test_minimize_SLSQP' to see why. But this can be fixed by using 'cost_int_penalty' as
20# objective function which adds a penalty for the "multiples of 0.0625" violation.
21# With fcmaes retry usually you can just "discretize" arguments.
22
23# This example is taken from https://www.sciencedirect.com/science/article/abs/pii/S0096300306015098
24
25# Tested using https://docs.conda.io/en/main/miniconda.html on Linux Mint 21.
26

41fac = 0.0625
42bounds = Bounds([1.1, 0.6, 0, 0], [240, 240, 240, 240]) # six inequalities
43#bounds = Bounds([0, 0, 0, 0], [240, 240, 240, 240]) # four inequalities
44
45def discrete(x):
46 feasible_x = x - x % fac
47 if feasible_x < bounds.lb[0]:
48 feasible_x += fac
49 return feasible_x
50
51def _feasible(x):
52 x = np.array(x)
53 x[0] = discrete(x[0])
54 x[1] = discrete(x[1])
55 return np.maximum(np.minimum(x, bounds.ub), bounds.lb)
56
57def constraint_ineq(x):
58 return [x[0] - 0.0193*x[2],
59 x[1] - 0.00954*x[2],
60 math.pi*x[2]**2 * x[3] + (4/3)*math.pi*x[2]**3 - 1296000]
61
62def weight(x):
63 return 0.6224*x[0]*x[2]*x[3] + 1.7781*x[1]*x[2]**2 \
64 + 3.1661 * x[0]**2 * x[3] + 19.84*x[0]**2 * x[2]
65
66def penalty(x):
67 return -np.sum(np.minimum(constraint_ineq(x), 0))
68
69def feasable_cost(x):
70 x = _feasible(x)
71 return weight(x)
72
73def feasable_cost_penalty(x):
74 return feasable_cost(x) + 100000*penalty(x)
75
76def penalty_int(x):
77 feasible_x = x - x % fac
78 if x - feasible_x < 0.5 * fac:
79 return x - feasible_x
80 else:
81 return feasible_x + fac - x
82
83def penalty_int_all(x):
84 return penalty_int(x[0]) + penalty_int(x[1])
85
86def cost_int_penalty(x):
87 return feasable_cost(x) + 100000*penalty_int_all(x)
88
89def print_result(ret, best, t0, i):
90 val = feasable_cost_penalty(ret.x)
91 x = _feasible(ret.x) # make sure result is _feasible
92 if val < best:
93 best = val
94 print("{0}: time = {1:.1f} best = {2:.8f} f(xmin) = {3:.5f} ineq = {4:.8f} x = {5:s}"
95 .format(i+1, dtime(t0), best, weight(x), penalty(x), str(x)))
96 return best

98def test_minimize_SLSQP(fun, num):
99 ineq_cons = {'type': 'ineq', 'fun' : constraint_ineq}
100 best = np.inf
101 t0 = time.perf_counter();
102 for i in range(num):
103 guess = random_x(bounds.lb, bounds.ub)
104 ret = minimize(fun, x0 = guess, bounds = bounds,
105 method='SLSQP', constraints=[ineq_cons])
106 if ret.success:
107 best = print_result(ret, best, t0, i)
