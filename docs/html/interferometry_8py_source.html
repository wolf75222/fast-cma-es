35def archipelago():
36 print('interferometer sga archipelago')
37 uda = pg.sga(gen = 50000)
38 # instantiate an unconnected archipelago
39 archi = pg.archipelago(t = pg.topologies.unconnected())
40 t = time()
41 for _ in range(8):
42 alg = pg.algorithm(uda)
43 #alg.set_verbosity(1)
44 prob = pg.problem(udp)
45 pop = pg.population(prob, 20)
46 isl = pg.island(algo=alg, pop=pop)
47 archi.push_back(isl)
48
49 archi.evolve()
50 archi.wait_check()
51 print(f'archi: {time() - t:0.3f}s')
